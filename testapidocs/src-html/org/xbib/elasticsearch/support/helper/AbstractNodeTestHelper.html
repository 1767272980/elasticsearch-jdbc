001package org.xbib.elasticsearch.support.helper;
002
003import org.elasticsearch.action.admin.cluster.health.ClusterHealthResponse;
004import org.elasticsearch.action.admin.cluster.node.info.NodesInfoRequest;
005import org.elasticsearch.action.admin.cluster.node.info.NodesInfoResponse;
006import org.elasticsearch.client.Client;
007import org.elasticsearch.common.logging.ESLogger;
008import org.elasticsearch.common.logging.ESLoggerFactory;
009import org.elasticsearch.common.network.NetworkUtils;
010import org.elasticsearch.common.settings.ImmutableSettings;
011import org.elasticsearch.common.settings.Settings;
012import org.elasticsearch.common.transport.InetSocketTransportAddress;
013import org.elasticsearch.common.transport.LocalTransportAddress;
014import org.elasticsearch.node.Node;
015import org.testng.Assert;
016import org.testng.annotations.AfterMethod;
017import org.testng.annotations.BeforeMethod;
018
019import java.io.IOException;
020import java.util.Map;
021import java.util.concurrent.atomic.AtomicInteger;
022
023import static org.elasticsearch.common.collect.Maps.newHashMap;
024import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
025import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
026
027public abstract class AbstractNodeTestHelper extends Assert {
028
029    protected final static ESLogger logger = ESLoggerFactory.getLogger("test");
030
031    private final static AtomicInteger clusterCount = new AtomicInteger();
032
033    protected String cluster;
034
035    // note, this must be same name as in json river specs
036    protected final String index = "my_jdbc_river_index";
037
038    protected final String type = "my_jdbc_river_type";
039
040    private Map<String, Node> nodes = newHashMap();
041
042    private Map<String, Client> clients = newHashMap();
043
044    protected void setClusterName() {
045        this.cluster = "test-jdbc-cluster-" + NetworkUtils.getLocalAddress().getHostName() + "-" + clusterCount.incrementAndGet();
046    }
047
048    protected String getClusterName() {
049        return cluster;
050    }
051
052    protected Settings getNodeSettings() {
053        return ImmutableSettings
054                .settingsBuilder()
055                .put("cluster.name", getClusterName())
056                .put("index.number_of_shards", 1)
057                .put("index.number_of_replica", 0)
058                .put("cluster.routing.schedule", "50ms")
059                .put("gateway.type", "none")
060                .put("index.store.type", "ram")
061                .put("http.enabled", false)
062                .put("discovery.zen.multicast.enabled", false)
063                .build();
064    }
065
066    public void startNodes() throws Exception {
067        setClusterName();
068
069        // we need more than one node, for the river state actions to test
070        startNode("1");
071        startNode("2");
072
073        // find node address
074        NodesInfoRequest nodesInfoRequest = new NodesInfoRequest().transport(true);
075        NodesInfoResponse response = client("1").admin().cluster().nodesInfo(nodesInfoRequest).actionGet();
076        Object obj = response.iterator().next().getTransport().getAddress().publishAddress();
077        if (obj instanceof InetSocketTransportAddress) {
078            InetSocketTransportAddress address = (InetSocketTransportAddress) obj;
079            // ... do we need our transport address?
080        }
081        if (obj instanceof LocalTransportAddress) {
082            LocalTransportAddress address = (LocalTransportAddress) obj;
083            // .... do we need local transport?
084        }
085    }
086
087    public Node startNode(String id) {
088        return buildNode(id).start();
089    }
090
091    public Node buildNode(String id) {
092        String settingsSource = getClass().getName().replace('.', '/') + ".yml";
093        Settings finalSettings = settingsBuilder()
094                .loadFromClasspath(settingsSource)
095                .put(getNodeSettings())
096                .put("name", id)
097                .build();
098        Node node = nodeBuilder().local(true).settings(finalSettings).build();
099        Client client = node.client();
100        nodes.put(id, node);
101        clients.put(id, client);
102        return node;
103    }
104
105    public void waitForYellow(String id) throws IOException {
106        logger.info("wait for healthy cluster...");
107        ClusterHealthResponse clusterHealthResponse = client(id).admin().cluster().prepareHealth().setWaitForYellowStatus().execute().actionGet();
108        if (clusterHealthResponse.isTimedOut()) {
109            throw new IOException("error, cluster health is " + clusterHealthResponse.getStatus().name());
110        }
111        logger.info("cluster health is {}", clusterHealthResponse.getStatus().name());
112    }
113
114    public void assertHits(String id, int expectedHits) {
115        client(id).admin().indices().prepareRefresh(index).execute().actionGet();
116        logger.debug("{} refreshed", index);
117        long hitsFound = client(id).prepareSearch(index).setTypes(type).execute().actionGet().getHits().getTotalHits();
118        logger.info("{}/{} = {} hits", index, type, hitsFound);
119        assertEquals(hitsFound, expectedHits);
120    }
121
122    public Client client(String id) {
123        Client client = clients.get(id);
124        if (client == null) {
125            client = nodes.get(id).client();
126            clients.put(id, client);
127        }
128        return client;
129    }
130
131    public void stopNodes() {
132        for (Client client : clients.values()) {
133            client.close();
134        }
135        clients.clear();
136        for (Node node : nodes.values()) {
137            node.stop();
138            node.close();
139        }
140        nodes.clear();
141    }
142
143}
