053    @BeforeMethod
054    @Parameters({"starturl", "user", "password", "create"})
055    public void beforeMethod(String starturl, String user, String password, @Optional String resourceName)
056            throws Exception {
057        startNodes();
058
059        logger.info("nodes started");
060
061        waitForYellow("1");
062        try {
063            // create river index
064            client("1").admin().indices().create(new CreateIndexRequest("_river")).actionGet();
065            logger.info("river index created");
066        } catch (IndexAlreadyExistsException e) {
067            logger.warn(e.getMessage());
068        }
069        source = getRiverSource()
070                .setUrl(starturl)
071                .setUser(user)
072                .setPassword(password);
073        context = getRiverContext()
074                .setRiverSource(source)
075                .setRetries(1)
076                .setMaxRetryWait(TimeValue.timeValueSeconds(5))
077                .setLocale("en");
078        context.contextualize();
079        logger.info("create table {}", resourceName);
080        if (resourceName == null || "".equals(resourceName)) {
081            return;
082        }
083        Connection connection = source.getConnectionForWriting();
084        if (connection == null) {
085            throw new IOException("no connection");
086        }
087        sqlScript(connection, resourceName);
088        source.closeWriting();
089    }

091    @AfterMethod
092    @Parameters({"stopurl", "user", "password", "delete"})
093    public void afterMethod(String stopurl, String user, String password, @Optional String resourceName)
094            throws Exception {
095
096        logger.info("remove table {}", resourceName);
097        if (resourceName == null || "".equals(resourceName)) {
098            return;
099        }
100        // before dropping tables, open read connection must be closed to avoid hangs in mysql/postgresql
101        logger.debug("closing reads...");
102        source.closeReading();
103
104        logger.debug("connecting for close...");
105        Connection connection = source.getConnectionForWriting();
106        if (connection == null) {
107            throw new IOException("no connection");
108        }
109        logger.debug("cleaning...");
110        // clean up tables
111        sqlScript(connection, resourceName);
112        logger.debug("closing writes...");
113        source.closeWriting();
114
115        // some driver can drop database by a magic 'stop' URL
116        source = getRiverSource()
117                .setUrl(stopurl)
118                .setUser(user)
119                .setPassword(password);
120        try {
121            logger.info("connecting to stop URL...");
122            // activate stop URL
123            source.getConnectionForWriting();
124        } catch (Exception e) {
125            // exception is expected, ignore
126        }
127        // close open write connection
128        source.closeWriting();
129        logger.info("stopped");
130
131        try {
132            client("1").admin().indices().deleteMapping(new DeleteMappingRequest()
133                    .indices(new String[]{"_river"}).types("my_jdbc_river")).actionGet();
134            logger.info("river deleted");
135        } catch (Exception e) {
136            logger.warn(e.getMessage());
137        }
138        // delete test index
139        try {
140            client("1").admin().indices().delete(new DeleteIndexRequest(index)).actionGet();
141            logger.info("index {} deleted", index);
142        } catch (IndexMissingException e) {
143            logger.warn(e.getMessage());
144        }
145
146        stopNodes();
147    }

149    protected void createRiver(String resource) {
150        try {
151            waitForYellow("1");
152            Map<String, Object> map = XContentHelper.convertToMap(Streams.copyToByteArray(getClass().getResourceAsStream(resource)), false).v2();
153            XContentBuilder builder = jsonBuilder()
154                    .startObject()
155                    .field("type", "jdbc")
156                    .field("jdbc", map.get("jdbc"))
157                    .endObject();
158            logger.debug("river spec = {}", builder.string());
159            IndexRequest indexRequest = Requests.indexRequest("_river").type("my_jdbc_river").id("_meta")
160                    .source(builder.string());
161            client("1").index(indexRequest).actionGet();
162            client("1").admin().indices().prepareRefresh("_river").execute().actionGet();
163            logger.info("river is created");
164            waitForRiverEnabled(client("1"), "my_jdbc_river", 15);
165            logger.info("river is enabled");
166            waitForRiverActive(client("1"), "my_jdbc_river", 15);
167            logger.info("river is active");
168        } catch (InterruptedException e) {
169            logger.error(e.getMessage());
170        } catch (IOException e) {
171            logger.error(e.getMessage());
172        }
173    }

187    protected void createRandomProducts(String sql, int size)
188            throws SQLException {
189        Connection connection = source.getConnectionForWriting();
190        for (int i = 0; i < size; i++) {
191            long amount = Math.round(Math.random() * 1000);
192            double price = (Math.random() * 10000) / 100.00;
193            add(connection, sql, UUID.randomUUID().toString().substring(0, 32), amount, price);
194        }
195        if (!connection.getAutoCommit()) {
196            connection.commit();
197        }
198        source.closeWriting();
199    }

242    private void sqlScript(Connection connection, String resourceName) throws Exception {
243        InputStream in = getClass().getResourceAsStream(resourceName);
244        BufferedReader br = new BufferedReader(new InputStreamReader(in, "UTF-8"));
245        String sql;
246        while ((sql = br.readLine()) != null) {
247
248            try {
249                logger.trace("executing {}", sql);
250                Statement p = connection.createStatement();
251                p.execute(sql);
252                p.close();
253            } catch (SQLException e) {
254                // ignore
255                logger.error(sql + " failed. Reason: " + e.getMessage());
256            } finally {
257                connection.commit();
258            }
259        }
260        br.close();
261    }

263    public static void waitForRiverEnabled(Client client, String riverName, int seconds) throws InterruptedException, IOException {
264        RiverStateRequest riverStateRequest = new RiverStateRequest()
265                .setRiverName(riverName);
266        RiverStateResponse riverStateResponse = client
267                .execute(RiverStateAction.INSTANCE, riverStateRequest).actionGet();
268        logger.info("enabled? riverStateResponse={}", riverStateResponse.getStates());
269        while (seconds-- > 0 && !isEnabled(riverName, riverStateResponse)) {
270            Thread.sleep(1000L);
271            try {
272                riverStateResponse = client.execute(RiverStateAction.INSTANCE, riverStateRequest).actionGet();
273                logger.info("enabled? riverStateResponse={}", riverStateResponse.getStates());
274            } catch (IndexMissingException e) {
275                // ignore
276            }
277        }
278    }
279
280    private static boolean isEnabled(String riverName, RiverStateResponse riverStateResponse) {
281        if (riverStateResponse == null) {
282            return false;
283        }
284        if (riverStateResponse.getStates() == null) {
285            return false;
286        }
287        if (riverStateResponse.getStates().isEmpty()) {
288            return false;
289        }
290        for (RiverState state : riverStateResponse.getStates()) {
291            if (state.getName().equals(riverName)) {
292                return state.isEnabled();
293            }
294        }
295        return false;
296    }

298    public static void waitForRiverActive(Client client, String riverName, int seconds) throws InterruptedException, IOException {
299        RiverStateRequest riverStateRequest = new RiverStateRequest()
300                .setRiverName(riverName);
301        RiverStateResponse riverStateResponse = client
302                .execute(RiverStateAction.INSTANCE, riverStateRequest).actionGet();
303
304        logger.info("active? riverStateResponse={}", riverStateResponse.getStates());
305        seconds *= 10;
306        while (seconds-- > 0 && !isActive(riverName, riverStateResponse)) {
307            Thread.sleep(100L);
308            try {
309                riverStateResponse = client.execute(RiverStateAction.INSTANCE, riverStateRequest).actionGet();
310                logger.info("active? riverStateResponse={}", riverStateResponse.getStates());
311            } catch (IndexMissingException e) {
312                //
313            }
314        }
315        if (seconds < 0) {
316            throw new IOException("timeout waiting for active river");
317        }
318    }

320    public static void waitForRiverInactive(Client client, String riverName, int seconds) throws InterruptedException, IOException {
321        RiverStateRequest riverStateRequest = new RiverStateRequest()
322                .setRiverName(riverName);
323        RiverStateResponse riverStateResponse = client
324                .execute(RiverStateAction.INSTANCE, riverStateRequest).actionGet();
325
326        logger.info("inactive? riverStateResponse={}", riverStateResponse.getStates());
327        seconds *= 10;
328        while (seconds-- > 0 && isActive(riverName, riverStateResponse)) {
329            Thread.sleep(100L);
330            try {
331                riverStateResponse = client.execute(RiverStateAction.INSTANCE, riverStateRequest).actionGet();
332                logger.info("inactive? riverStateResponse={}", riverStateResponse.getStates());
333            } catch (IndexMissingException e) {
334                //
335            }
336        }
337        if (seconds < 0) {
338            throw new IOException("timeout waiting for inactive river");
339        }
340    }

342    private static boolean isActive(String riverName, RiverStateResponse riverStateResponse) {
343        if (riverStateResponse == null) {
344            return false;
345        }
346        if (riverStateResponse.getStates() == null) {
347            return false;
348        }
349        if (riverStateResponse.getStates().isEmpty()) {
350            return false;
351        }
352        for (RiverState state : riverStateResponse.getStates()) {
353            if (state.getName().equals(riverName)) {
354                return state.isActive();
355            }
356        }
357        return false;
358    }
