004package org.xbib.elasticsearch.river.jdbc.strategy.table;
005
006import java.util.Arrays;
007import java.util.List;
008
009import org.xbib.elasticsearch.river.jdbc.strategy.mock.MockRiverMouth;
010import org.testng.Assert;
011import org.testng.annotations.Test;
012
013/**
014 * @author simon00t
015 *
016 */
017public class TableValueTests extends Assert {
018        
019    @Test
020    public void testFilterColumns() throws Exception {
021        List<String> columns = Arrays.asList("source_operation", "_id", "label");
022        List<String> row1 = Arrays.asList("index", "1", "label1");
023        List<String> row2 = Arrays.asList("index", "2", "label2");
024        List<String> row3 = Arrays.asList("index", "3", "label3");
025        List<String> row4 = Arrays.asList("index", "4", "label4");
026        MockRiverMouth target = new MockRiverMouth();
027        new TableValueListener()
028                .target(target)
029                .begin()
030                .keys(columns)
031                .values(row1)
032                .values(row2)
033                .values(row3)
034                .values(row4)
035                .end();
036        assertEquals(target.data().size(), 4, "Number of inserted objects");
037        assertEquals(target.data().toString(),"{index/null/null/1 {label=\"label1\"}={\"label\":\"label1\"}, " +
038                        "index/null/null/2 {label=\"label2\"}={\"label\":\"label2\"}, " +
039                                        "index/null/null/3 {label=\"label3\"}={\"label\":\"label3\"}, " +
040                                                        "index/null/null/4 {label=\"label4\"}={\"label\":\"label4\"}}");
041    }
042}
