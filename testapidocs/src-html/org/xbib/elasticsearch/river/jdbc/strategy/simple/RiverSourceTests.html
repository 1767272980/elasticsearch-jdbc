051    public void testSimpleSQL(String sql) throws Exception {
052        PreparedStatement statement = source.prepareQuery(sql);
053        ResultSet results = source.executeQuery(statement);
054        for (int i = 0; i < 5; i++) {

061    @Test
062    @Parameters({"sql2", "n"})
063    public void testSimpleStarQuery(String sql, @Optional Integer n) throws Exception {
064        List<Object> params = new LinkedList<Object>();
065        RiverMouth output = new MockRiverMouth() {
066            @Override
067            public void index(IndexableObject object, boolean create) throws IOException {
068                logger.debug("object={}", object);
069            }
070        };
071        PreparedStatement statement = source.prepareQuery(sql);
072        source.bind(statement, params);
073        ResultSet results = source.executeQuery(statement);
074        KeyValueStreamListener listener = new StringKeyValueStreamListener()
075                .output(output);
076        long rows = 0L;
077        source.beforeRows(results, listener);
078        while (source.nextRow(results, listener)) {
079            rows++;
080        }
081        source.afterRows(results, listener);
082        assertEquals(rows, n == null ? 5 : n);
083        source.close(results);
084        source.close(statement);
085    }

087    @Test
088    @Parameters({"sql3"})
089    public void testSimpleNullInteger(String sql) throws Exception {
090        List<Object> params = new LinkedList<Object>();
091        RiverMouth mouth = new MockRiverMouth() {
092            @Override
093            public void index(IndexableObject object, boolean create) throws IOException {
094                if (object == null || object.source() == null) {
095                    throw new IllegalArgumentException("object missing");
096                }
097                Values o = (Values) object.source().get("amount");
098                if (o == null) {
099                    o = (Values) object.source().get("AMOUNT"); // hsqldb is uppercase
100                }
101                if (!o.isNull()) {
102                    throw new IllegalArgumentException("amount not null??? " + o.getClass().getName());
103                }
104            }
105        };
106        PreparedStatement statement = source.prepareQuery(sql);
107        source.bind(statement, params);
108        ResultSet results = source.executeQuery(statement);
109        KeyValueStreamListener listener = new StringKeyValueStreamListener()
110                .output(mouth);
111        long rows = 0L;
112        source.beforeRows(results, listener);
113        if (source.nextRow(results, listener)) {
114            // only one row
115            rows++;
116        }
117        source.afterRows(results, listener);
118        assertEquals(rows, 1);
119        source.close(results);
120        source.close(statement);
121    }

123    /**
124     * Test JDBC Array to structured object array
125     *
126     * @param sql the array select statement
127     * @throws Exception if test fails
128     */
129    @Test
130    @Parameters({"sql4", "res1", "res2"})
131    public void testSimpleArray(@Optional String sql, @Optional String res1, @Optional String res2) throws Exception {
132        if (sql == null) {
133            return;
134        }
135        List<Object> params = new LinkedList<Object>();
136        final List<IndexableObject> result = new LinkedList<IndexableObject>();
137        RiverMouth mouth = new MockRiverMouth() {
138            @Override
139            public void index(IndexableObject object, boolean create) throws IOException {
140                if (object == null || object.source() == null) {
141                    throw new IllegalArgumentException("object missing");
142                }
143                result.add(object);
144            }
145        };
146        PreparedStatement statement = source.prepareQuery(sql);
147        source.bind(statement, params);
148        ResultSet results = source.executeQuery(statement);
149        KeyValueStreamListener listener = new StringKeyValueStreamListener()
150                .output(mouth);
151        long rows = 0L;
152        source.beforeRows(results, listener);
153        while (source.nextRow(results, listener)) {
154            rows++;
155        }
156        source.afterRows(results, listener);
157        assertEquals(rows, 2);
158        source.close(results);
159        source.close(statement);
160        Iterator<IndexableObject> it = result.iterator();
161        assertEquals(it.next().source().toString(), res1);
162        assertEquals(it.next().source().toString(), res2);
163    }
