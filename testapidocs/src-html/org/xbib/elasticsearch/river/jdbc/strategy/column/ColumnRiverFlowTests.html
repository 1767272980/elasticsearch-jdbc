030public class ColumnRiverFlowTests extends AbstractRiverNodeTest {

032    @Override

042    @Test()
043    @Parameters({"river-existedWhereClause"})
044    public void testWriteLastRiverRunTimeToCustomRiverInfo(String riverResource) throws IOException {
045        final Client client = client("1");
046        setupContext(new MockRiverSource() {
047            @Override
048            public void fetch() {
049            }
050        }, riverResource);
051        Map<String, Object> spec = (Map<String, Object>) riverSettings(riverResource).settings().get("jdbc");
052        Map<String, String> loadedSettings = new JsonSettingsLoader().load(jsonBuilder().map(spec).string());
053        Settings settings = settingsBuilder().put(loadedSettings).build();
054        RiverFlow flow = new ColumnRiverFlow();
055        flow.setRiverContext(context);
056        flow.getFeeder()
057                .setRiverState(new RiverState())
058                .setSpec(spec)
059                .setSettings(settings)
060                .setClient(client)
061                .run();

063        //assertLastRiverRunTimeExists(client);
064    }
065
066    @Test()
067    @Parameters({"river-existedWhereClause"})
068    public void testReadLastRiverRunTimeFromCustomRiverInfo(String riverResource) throws IOException {
069        final Client client = client("1");
070        final TimeValue lastRunAt = new TimeValue(600);
071        setupContext(new MockRiverSource() {
072            @Override
073            public void fetch() {
074                assertLastRunDateEquals(lastRunAt);
075            }
076        }, riverResource);
077        writeLastRunDateToCustomRiverInfo(client, lastRunAt);
078        Map<String, Object> spec = (Map<String, Object>) riverSettings(riverResource).settings().get("jdbc");
079        Map<String, String> loadedSettings = new JsonSettingsLoader().load(jsonBuilder().map(spec).string());
080        Settings settings = settingsBuilder().put(loadedSettings).build();
081        RiverFlow flow = new ColumnRiverFlow();
082        flow.setRiverContext(context);
083        flow.getFeeder()
084                .setRiverState(new RiverState())
085                .setSpec(spec)
086                .setSettings(settings)
087                .setClient(client)
088                .run();
089    }
090
091    private void setupContext(RiverSource riverSource, String riverResource) throws IOException {
092        RiverSettings riverSettings = riverSettings(riverResource);
093        context.setRiverName(new RiverName(index, type).getName());
094        context.setRiverMouth(new MockRiverMouth());
095        context.setRiverSource(riverSource);
096        context.setRiverSettings(riverSettings.settings());
097        context.columnEscape(true);
098    }
099
100    private void assertLastRiverRunTimeExists(Client client) {
101        try {
102            GetResponse get = client.prepareGet("_river",
103                    context.getRiverName(),
104                    ColumnRiverFlow.DOCUMENT).execute().actionGet();
105            logger.info("get response = {}", get.getSourceAsMap());
106            Map map = (Map) get.getSourceAsMap().get("jdbc");
107            assertNotNull(map, "jdbc key not exists in custom info");
108            assertNotNull(map.get(ColumnRiverFlow.LAST_RUN_TIME), "last run time should not be null");
109        } catch (IndexMissingException e) {
110            fail("custom info not found");
111        }
112    }
113
114    private void assertLastRunDateEquals(TimeValue expectedLastRunAt) {
115        Map map = (Map) context.getRiverSettings().get("jdbc");
116        assertNotNull(map);
117        assertEquals(map.get(ColumnRiverFlow.LAST_RUN_TIME), expectedLastRunAt);
118    }
119
120    private void writeLastRunDateToCustomRiverInfo(Client client, TimeValue lastRunAt) throws IOException {
121        XContentBuilder builder = XContentFactory.jsonBuilder()
122                .startObject()
123                .startObject("jdbc")
124                .field(ColumnRiverFlow.LAST_RUN_TIME, lastRunAt.millis())
125                .endObject()
126                .endObject();
127        client.prepareIndex().setIndex("_river").setType(context.getRiverName()).setId(ColumnRiverFlow.DOCUMENT)
128                .setSource(builder.string()).execute().actionGet();
129        client.admin().indices().refresh(Requests.refreshRequest("_river")).actionGet();
130    }
