043    @Test

052    @Test

054    public void testSQL(String sql) throws Exception {
055        PreparedStatement statement = source.prepareQuery(sql);
056        ResultSet results = source.executeQuery(statement);
057        for (int i = 0; i < 5; i++) {
058            assertTrue(results.next());
059        }
060        source.close(results);

064    @Test
065    @Parameters({"sql2", "n"})
066    public void testStarQuery(String sql, @Optional Integer n) throws Exception {
067        List<? extends Object> params = newLinkedList();
068        RiverMouth output = new MockRiverMouth() {
069            @Override
070            public void index(IndexableObject object, boolean create) throws IOException {
071                logger.debug("object={}", object);
072            }
073        };
074        PreparedStatement statement = source.prepareQuery(sql);
075        source.bind(statement, params);
076        ResultSet results = source.executeQuery(statement);
077        KeyValueStreamListener listener = new RiverKeyValueStreamListener()
078                .output(output);
079        long rows = 0L;
080        source.beforeRows(results, listener);
081        while (source.nextRow(results, listener)) {
082            rows++;
083        }
084        source.afterRows(results, listener);
085        assertEquals(rows, n == null ? 5 : n);
086        source.close(results);
087        source.close(statement);
088    }

090    @Test
091    @Parameters({"sql3"})
092    public void testNullInteger(String sql) throws Exception {
093        List<? extends Object> params = newLinkedList();
094        RiverMouth mouth = new MockRiverMouth() {
095            @Override
096            public void index(IndexableObject object, boolean create) throws IOException {
097                if (object == null || object.source() == null) {
098                    throw new IllegalArgumentException("object missing");
099                }
100                Values o = (Values)object.source().get("amount");
101                if (o == null) {
102                    o = (Values)object.source().get("AMOUNT"); // hsqldb is uppercase
103                }
104                else if (!o.isNull()) {
105                    throw new IllegalArgumentException("amount not null??? " + o.getClass().getName() );
106                }
107            }
108        };
109        PreparedStatement statement = source.prepareQuery(sql);
110        source.bind(statement, params);
111        ResultSet results = source.executeQuery(statement);
112        KeyValueStreamListener listener = new RiverKeyValueStreamListener()
113                .output(mouth);
114        long rows = 0L;
115        source.beforeRows(results, listener);
116        if (source.nextRow(results, listener)) {
117            // only one row
118            rows++;
119        }
120        source.afterRows(results, listener);
121        assertEquals(rows, 1);
122        source.close(results);
123        source.close(statement);
124    }

127    /**
128     * Test JDBC Array to structured object array
129     *
130     * @param sql the array select statement
131     * @throws Exception
132     */
133    @Test
134    @Parameters({"sql4", "res1", "res2"})
135    public void testArray(@Optional String sql, @Optional String res1, @Optional String res2) throws Exception {
136        if (sql == null) {
137            return;
138        }
139        List<? extends Object> params = newLinkedList();
140        final List<IndexableObject> result = newLinkedList();
141        RiverMouth mouth = new MockRiverMouth() {
142            @Override
143            public void index(IndexableObject object, boolean create) throws IOException {
144                if (object == null || object.source() == null) {
145                    throw new IllegalArgumentException("object missing");
146                }
147                result.add(object);
148            }
149        };
150        PreparedStatement statement = source.prepareQuery(sql);
151        source.bind(statement, params);
152        ResultSet results = source.executeQuery(statement);
153        KeyValueStreamListener listener = new RiverKeyValueStreamListener()
154                .output(mouth);
155        long rows = 0L;
156        source.beforeRows(results, listener);
157        while (source.nextRow(results, listener)) {
158            rows++;
159        }
160        source.afterRows(results, listener);
161        assertEquals(rows, 2);
162        source.close(results);
163        source.close(statement);
164        Iterator<IndexableObject> it = result.iterator();
165        assertEquals(it.next().source().toString(), res1);
166        assertEquals(it.next().source().toString(), res2);
167    }
