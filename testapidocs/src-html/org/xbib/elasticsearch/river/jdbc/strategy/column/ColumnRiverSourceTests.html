083    private void verifyCreateObjects(String riverResource, String sql)
084            throws SQLException, IOException, InterruptedException {
085        final int newRecordsOutOfTimeRange = 3;
086        final int newRecordsInTimeRange = 2;
087        final int updatedRecordsInTimeRange = 4;
088        testColumnRiver(new MockRiverMouth(), riverResource, sql, new ProductFixture[]{
089                new ProductFixture(newRecordsOutOfTimeRange)
090                        .createdAt(oldTimestamp()),
091                new ProductFixture(newRecordsInTimeRange)
092                        .createdAt(okTimestamp()),
093                new ProductFixture(updatedRecordsInTimeRange)
094                        .createdAt(oldTimestamp())
095                        .updatedAt(okTimestamp()),
096        }, newRecordsInTimeRange + updatedRecordsInTimeRange);
097    }
098
099    private void verifyDeleteObjects(String riverResource, String insertSql)
100            throws IOException, SQLException, InterruptedException {
101        MockRiverMouth riverMouth = new MockRiverMouth();
102        boolean[] shouldProductsBeDeleted = new boolean[]{true, true, false};
103        ProductFixtures productFixtures = createFixturesAndPopulateMouth(shouldProductsBeDeleted, riverMouth);
104        testColumnRiver(riverMouth, riverResource, insertSql,
105                productFixtures.fixtures,
106                productFixtures.expectedCount);
107    }
108
109    private ProductFixtures createFixturesAndPopulateMouth(boolean[] shouldProductsBeDeleted, MockRiverMouth riverMouth)
110            throws IOException {
111        ProductFixture[] fixtures = new ProductFixture[shouldProductsBeDeleted.length];
112        int expectedExistsCountAfterRiverRun = 0;
113        for (int i = 0; i < shouldProductsBeDeleted.length; i++) {
114            IndexableObject p = new PlainIndexableObject()
115                    .id(Integer.toString(i))
116                    .source(createSource(i))
117                    .optype("delete");
118            riverMouth.index(p, false);
119            Timestamp deletedAt;
120            if (shouldProductsBeDeleted[i]) {
121                deletedAt = okTimestamp();
122            } else {
123                deletedAt = oldTimestamp();
124                expectedExistsCountAfterRiverRun++;
125            }
126            fixtures[i] = ProductFixture.one()
127                    .setId(i)
128                    .createdAt(oldTimestamp())
129                    .updatedAt(oldTimestamp())
130                    .deletedAt(deletedAt);
131        }
132        return new ProductFixtures(fixtures, expectedExistsCountAfterRiverRun);
133    }
134
135    private Map<String, Object> createSource(int id) {
136        Map<String, Object> map = new HashMap<String, Object>();
137        map.put("id", id);
138        map.put("name", null);
139        return map;
140    }

142    private void testColumnRiver(MockRiverMouth riverMouth, String riverResource, String sql, ProductFixture[] fixtures, int expectedHits)
143            throws SQLException, IOException, InterruptedException {
144        createData(sql, fixtures);
145        context.setRiverMouth(riverMouth);
146        setContextSettings(riverResource);
147        source.fetch();
148        assertEquals(riverMouth.data().size(), expectedHits);
149    }
150
151    private void setContextSettings(String riverResource) throws IOException {
152        RiverSettings riverSettings = riverSettings(riverResource);
153        Map<String, Object> settings = (Map<String, Object>) riverSettings.settings().get("jdbc");
154        settings.put(ColumnRiverFlow.LAST_RUN_TIME, LAST_RUN_TIME);
155        settings.put(ColumnRiverFlow.CURRENT_RUN_STARTED_TIME, new TimeValue(new Date().getTime()));
156        context.setRiverSettings(riverSettings.settings());
157        context.setStatements(SQLCommand.parse(settings));
158        context.columnCreatedAt(XContentMapValues.nodeStringValue(settings.get("column_created_at"), null));
159        context.columnUpdatedAt(XContentMapValues.nodeStringValue(settings.get("column_updated_at"), null));
160        context.columnDeletedAt(XContentMapValues.nodeStringValue(settings.get("column_deleted_at"), null));
161        context.columnEscape(true);
162    }
163
164    private Timestamp okTimestamp() {
165        return new Timestamp(LAST_RUN_TIME.getMillis() + 1000);
166    }
167
168    private Timestamp oldTimestamp() {
169        return new Timestamp(LAST_RUN_TIME.getMillis() - 1000);
170    }
171
172    private void createData(String sql, ProductFixture[] fixtures) throws SQLException {
173        Connection conn = source.getConnectionForWriting();
174        for (ProductFixture fixture : fixtures) {
175            createData(conn, sql, fixture);
176        }
177        source.closeWriting();
178    }
179
180    private void createData(Connection connection, String sql, ProductFixture fixture) throws SQLException {
181        PreparedStatement stmt = connection.prepareStatement(sql);
182        logger.debug("timestamps: [" + fixture.createdAt + ", " + fixture.updatedAt + ", " + fixture.deletedAt + "]");
183        for (int i = 0; i < fixture.size; i++) {
184            int id = fixture.id >= 0 ? fixture.id : random.nextInt();
185            logger.debug("id={}", id);
186            stmt.setInt(1, id);
187            stmt.setNull(2, Types.VARCHAR);
188            stmt.setInt(3, 1);
189            stmt.setDouble(4, 1.1);
190            if (fixture.createdAt != null) {
191                stmt.setTimestamp(5, fixture.createdAt);
192            } else {
193                stmt.setNull(5, Types.TIMESTAMP);
194            }
195            if (fixture.updatedAt != null) {
196                stmt.setTimestamp(6, fixture.updatedAt);
197            } else {
198                stmt.setNull(6, Types.TIMESTAMP);
199            }
200            if (fixture.deletedAt != null) {
201                stmt.setTimestamp(7, fixture.deletedAt);
202            } else {
203                stmt.setNull(7, Types.TIMESTAMP);
204            }
205            stmt.execute();
206        }
207    }
