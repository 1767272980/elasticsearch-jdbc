002package org.xbib.elasticsearch.river.jdbc.strategy.simple;
003
004import org.elasticsearch.client.Client;
005import org.elasticsearch.river.RiverName;
006import org.elasticsearch.river.RiverSettings;
007
008import org.testng.annotations.Parameters;
009import org.testng.annotations.Test;
010
011import org.xbib.elasticsearch.river.jdbc.JDBCRiver;
012import org.xbib.elasticsearch.river.jdbc.RiverSource;
013import org.xbib.elasticsearch.river.jdbc.support.AbstractRiverNodeTest;
014import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
015
016import java.sql.Connection;
017import java.sql.PreparedStatement;
018import java.sql.SQLException;
019import java.util.ArrayList;
020import java.util.List;
021import java.util.UUID;
022
023public class SimpleRiverDataTests extends AbstractRiverNodeTest {
024
025    private Client client;
026
027    @Override
028    public RiverSource getRiverSource() {
029        return new SimpleRiverSource();
030    }
031
032    @Override
033    public RiverContext getRiverContext() {
034        return new RiverContext();
035    }
036
037    /**
038     * Product table star select
039     *
040     * @param riverResource
041     * @throws Exception
042     */
043    @Test
044    @Parameters({"river2", "sql1"})
045    public void testSimpleRiverRandom(String riverResource, String sql) throws Exception {
046        Connection connection = source.connectionForWriting();
047        createRandomProducts(connection, sql, 100);
048        source.closeWriting();
049        client = client("1");
050        RiverSettings settings = riverSettings(riverResource);
051        JDBCRiver river = new JDBCRiver(new RiverName(INDEX, TYPE), settings, client);
052        river.start();
053        Thread.sleep(10000L); // let the river run
054        assertEquals(client.prepareSearch(INDEX).execute().actionGet().getHits().getTotalHits(), 104);
055        river.close();
056    }
057
058    /**
059     * Product table
060     *
061     * @param riverResource the river
062     * @param sql the SQL statement
063     * @throws Exception
064     */
065    @Test
066    @Parameters({"river3", "sql1"})
067    public void testSimpleRiverMaxrows(String riverResource, String sql) throws Exception {
068        Connection connection = source.connectionForWriting();
069        createRandomProducts(connection, sql, 100);
070        source.closeWriting();
071        client = client("1");
072        RiverSettings settings = riverSettings(riverResource);
073        JDBCRiver river = new JDBCRiver(new RiverName(INDEX, TYPE), settings, client);
074        river.start();
075        Thread.sleep(5000L); // let the river flow...
076        river.once();
077        Thread.sleep(5000L); // let the river flow...
078        assertEquals(client.prepareSearch(INDEX).execute().actionGet().getHits().getTotalHits(), 208);
079        river.close();
080    }
081
082    private void createRandomProducts(Connection connection, String sql, int size)
083            throws SQLException {
084        for (int i = 0; i < size; i++) {
085            long amount = Math.round(Math.random() * 1000);
086            double price = (Math.random() * 10000) / 100.00;
087            addData(connection, sql, UUID.randomUUID().toString().substring(0, 32), amount, price);
088        }
089        if (!connection.getAutoCommit()) {
090            connection.commit();
091        }
092    }
093
094    private void addData(Connection connection, String sql, final String name, final long amount, final double price)
095            throws SQLException {
096        PreparedStatement stmt = connection.prepareStatement(sql);
097        List<Object> params = new ArrayList<Object>() {{
098            add(name);
099            add(amount);
100            add(price);
101        }};
102        source.bind(stmt, params);
103        stmt.execute();
104    }
105
106}
