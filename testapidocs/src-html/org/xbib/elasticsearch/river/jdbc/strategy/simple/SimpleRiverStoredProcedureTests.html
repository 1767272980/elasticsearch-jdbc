021    import java.util.List;
022    import java.util.UUID;
023    
024    public class SimpleRiverStoredProcedureTests extends AbstractRiverNodeTest {
025    
026        @Override
027        public RiverSource getRiverSource() {
028            return new SimpleRiverSource();
029        }
030    
031        @Override
032        public RiverContext getRiverContext() {
033            return new RiverContext();
034        }
035    
036        @Test
037        @Parameters({"river5", "sql1", "sql2"})
038        public void testStoredProcedure(String riverResource, String sql, String storedProcSQL)
039                throws Exception {
040            Connection connection = source.connectionForWriting();
041            createRandomProducts(connection, sql, 100);
042    
043            // create stored procedure
044            Statement statement = connection.createStatement();
045            statement.execute(storedProcSQL);
046            statement.close();
047    
048            source.closeWriting();
049            Client client = client("1");
050            RiverSettings settings = riverSettings(riverResource);
051            JDBCRiver river = new JDBCRiver(new RiverName(INDEX, TYPE), settings, client);
052            river.start();
053            Thread.sleep(10000L); // let the river run
054            System.err.println(client.prepareSearch(INDEX).execute().actionGet().getHits().getTotalHits());
055            river.close();
056        }
057    
058        private void createRandomProducts(Connection connection, String sql, int size)
059                throws SQLException {
060            for (int i = 0; i < size; i++) {
061                long amount = Math.round(Math.random() * 1000);
062                double price = (Math.random() * 10000) / 100.00;
063                addData(connection, sql, UUID.randomUUID().toString().substring(0, 32), amount, price);
064            }
065            if (!connection.getAutoCommit()) {
066                connection.commit();
067            }
068        }
069    
070        private void addData(Connection connection, String sql, final String name, final long amount, final double price)
071                throws SQLException {
072            PreparedStatement stmt = connection.prepareStatement(sql);
073            List<Object> params = new ArrayList<Object>() {{
074                add(name);
075                add(amount);
076                add(price);
077            }};
078            source.bind(stmt, params);
079            stmt.execute();
080        }
081    }
