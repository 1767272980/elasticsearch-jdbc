026    public class SimpleRiverJobTests extends AbstractRiverNodeTest {
027    
028        @Override
029        public RiverSource getRiverSource() {
030            return new SimpleRiverSource();
031        }
032    
033        @Override
034        public RiverContext getRiverContext() {
035            return new RiverContext();
036        }
037    
038        @Test
039        @Parameters({"river1", "sql1", "sql2"})
040        public void testSimpleRiverJob(String riverResource, String sql1, String sql2)
041                throws IOException, InterruptedException, SQLException {
042            Connection connection = source.connectionForWriting();
043            createRandomProductsJob(connection, sql2, 100);
044            source.closeWriting();
045            connection = source.connectionForReading();
046            ResultSet results = connection.createStatement().executeQuery(sql1);
047            if (!connection.getAutoCommit()) {
048                connection.commit();
049            }
050            int count = results.next() ? results.getInt(1) : -1;
051            source.close(results);
052            source.closeReading();
053            assertEquals(count, 100);
054            Client client = client("1");
055            assertEquals(client.prepareSearch(INDEX).execute().actionGet().getHits().getTotalHits(), 0);
056            RiverSettings settings = riverSettings(riverResource);
057            JDBCRiver river = new JDBCRiver(new RiverName(INDEX, TYPE), settings, client);
058            river.once();
059            Thread.sleep(10000L); // let the river flow
060            assertEquals(client.prepareSearch(INDEX).setTypes(TYPE).execute().actionGet().getHits().getTotalHits(), 100);
061            connection = source.connectionForReading();
062            // sql1 = select count(*)
063            results = connection.createStatement().executeQuery(sql1);
064            if (!connection.getAutoCommit()) {
065                connection.commit();
066            }
067            count = results.next() ? results.getInt(1) : -1;
068            results.close();
069            assertEquals(count, 0);
070            river.close();
071        }
072    
073        private void createRandomProductsJob(Connection connection, String sql, int size)
074                throws SQLException {
075            for (int i = 0; i < size; i++) {
076                long amount = Math.round(Math.random() * 1000);
077                double price = (Math.random() * 10000) / 100.00;
078                add(connection, sql, "1", UUID.randomUUID().toString().substring(0, 32), amount, price);
079            }
080            if (!connection.getAutoCommit()) {
081                connection.commit();
082            }
083        }
084    
085        private void add(Connection connection, String sql, final String job, final String name, final long amount, final double price)
086                throws SQLException {
087            PreparedStatement stmt = connection.prepareStatement(sql);
088            List<Object> params = new ArrayList<Object>() {
089                {
090                    add(job);
091                    add(name);
092                    add(amount);
093                    add(price);
094                }
095            };
096            source.bind(stmt, params);
097            stmt.execute();
098        }
099    }
