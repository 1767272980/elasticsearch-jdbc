001package org.xbib.elasticsearch.river.jdbc.strategy;
002
003import org.elasticsearch.common.logging.ESLogger;
004import org.elasticsearch.common.logging.ESLoggerFactory;
005import org.elasticsearch.common.unit.TimeValue;
006import org.xbib.elasticsearch.river.jdbc.RiverSource;
007import org.xbib.elasticsearch.river.jdbc.strategy.simple.SimpleRiverSource;
008import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
009import org.testng.Assert;
010import org.testng.annotations.Test;
011
012import java.sql.Connection;
013import java.sql.PreparedStatement;
014import java.sql.ResultSet;
015
016public class CSVTests extends Assert {
017
018    private final ESLogger logger = ESLoggerFactory.getLogger(CSVTests.class.getName());
019
020    @Test
021    public void testCSV() throws Exception {
022        RiverSource source = new SimpleRiverSource()
023                .driver("org.xbib.jdbc.csv.CsvDriver")
024                .url("jdbc:csv:src/test/resources/csv?separator=;")
025                .user("")
026                .password("");
027        RiverContext context = new RiverContext()
028                .riverSource(source)
029                .retries(1)
030                .maxRetryWait(TimeValue.timeValueSeconds(3));
031        context.contextualize();
032        Connection connection = source.connectionForReading();
033        assertNotNull(connection);
034
035        // fetch some data
036        PreparedStatement statement = source.prepareQuery("select * from Gesamtfinanzplan");
037        ResultSet results = source.executeQuery(statement);
038        for (int i = 0; i < 36; i++) {
039            assertTrue(results.next());
040            logger.info("{} {} {} {} {} {} {} ",
041                    results.getString(1),
042                    results.getString(2),
043                    results.getString(3),
044                    results.getString(4),
045                    results.getString(5),
046                    results.getString(6),
047                    results.getString(7)
048            );
049
050        }
051        source.close(results);
052        source.close(statement);
053
054
055        source.closeReading();
056    }
057}
