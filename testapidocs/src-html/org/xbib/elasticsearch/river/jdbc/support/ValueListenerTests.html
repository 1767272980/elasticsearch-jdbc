074    @Test
075    public void testNestedObjects() throws Exception {
076        List<String> columns = Arrays.asList("_id", "person.salary", "person.name", "person.position.name", "person.position.since");
077        List<String> row1 = Arrays.asList("1", "$1000", "Joe Doe", "Worker", "2012-06-12");
078        List<String> row2 = Arrays.asList("2", "$2000", "Bill Smith", "Boss", "2012-06-13");
079        MockRiverMouth output = new MockRiverMouth();
080        new StringKeyValueStreamListener()
081                .output(output)
082                .begin()
083                .keys(columns)
084                .values(row1)
085                .values(row2)
086                .end();
087        assertEquals(output.data().size(), 2, "Number of inserted objects");
088        assertEquals(output.data().toString(),
089                "{[null/null/null/1]->{person={name=\"Joe Doe\", position={name=\"Worker\", since=\"2012-06-12\"}, salary=\"$1000\"}}={\"person\":{\"name\":\"Joe Doe\",\"position\":{\"name\":\"Worker\",\"since\":\"2012-06-12\"},\"salary\":\"$1000\"}}, [null/null/null/2]->{person={name=\"Bill Smith\", position={name=\"Boss\", since=\"2012-06-13\"}, salary=\"$2000\"}}={\"person\":{\"name\":\"Bill Smith\",\"position\":{\"name\":\"Boss\",\"since\":\"2012-06-13\"},\"salary\":\"$2000\"}}}"
090        );

249    public void testDoubleScientificNotation() throws Exception {
250        List<String> columns = Arrays.asList("_id", "lat", "lon");
251        List<Object> row1 = new LinkedList<Object>();
252        row1.add("1");
253        row1.add(50.940664);
254        row1.add(6.9599115);
255        MockRiverMouth output = new MockRiverMouth();
256        new RiverMouthKeyValueStreamListener<String,Object>()
257                .output(output)
258                .begin()
259                .keys(columns)
260                .values(row1)
261                .end();
262        assertEquals(output.data().toString(),
263                "{[null/null/null/1]->{lat=50,940664000000, lon=6,959911500000}={\"lat\":50.940664,\"lon\":6.9599115}}"
