002package org.xbib.elasticsearch.river.jdbc.support;
003
004import org.elasticsearch.action.admin.indices.create.CreateIndexRequest;
005import org.elasticsearch.action.admin.indices.delete.DeleteIndexRequest;
006import org.elasticsearch.client.Client;
007import org.elasticsearch.common.logging.ESLogger;
008import org.elasticsearch.common.logging.Loggers;
009import org.elasticsearch.common.network.NetworkUtils;
010import org.elasticsearch.common.settings.ImmutableSettings;
011import org.elasticsearch.common.settings.Settings;
012import org.elasticsearch.indices.IndexMissingException;
013import org.elasticsearch.node.Node;
014import org.testng.annotations.AfterMethod;
015import org.testng.annotations.BeforeMethod;
016
017import java.util.Map;
018import java.util.concurrent.atomic.AtomicLong;
019
020import static org.elasticsearch.common.collect.Maps.newHashMap;
021import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
022import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
023
024public abstract class AbstractRiverNodeTest extends AbstractRiverTest {
025
026    private static final ESLogger logger = Loggers.getLogger(AbstractRiverNodeTest.class.getSimpleName());
027
028    public final String INDEX = "my_jdbc_river";
029
030    public final String TYPE = "my_jdbc_river";
031
032    private final static AtomicLong counter = new AtomicLong();
033
034    private Map<String, Node> nodes = newHashMap();
035
036    private Map<String, Client> clients = newHashMap();
037
038    private Settings defaultSettings() {
039        return ImmutableSettings
040            .settingsBuilder()
041            .put("cluster.name", "testing-jdbc-river-on-" + NetworkUtils.getLocalAddress().getHostName() + "-" + counter.incrementAndGet())
042            .build();
043    }
044
045    @BeforeMethod
046    public void createIndices() throws Exception {
047        startNode("1").client();
048        client("1").admin().indices().create(new CreateIndexRequest(INDEX)).actionGet();
049    }
050
051    @AfterMethod
052    public void deleteIndices() {
053        try {
054            // clear test index
055            client("1").admin().indices()
056                    .delete(new DeleteIndexRequest().indices(INDEX))
057                    .actionGet();
058        } catch (IndexMissingException e) {
059            logger.error(e.getMessage());
060        }
061        try {
062            // clear rivers
063            client("1").admin().indices()
064                    .delete(new DeleteIndexRequest().indices("_river"))
065                    .actionGet();
066        } catch (IndexMissingException e) {
067            logger.error(e.getMessage());
068        }
069        closeNode("1");
070        closeAllNodes();
071    }
072
073
074    public Node startNode(String id) {
075        return buildNode(id).start();
076    }
077
078    public Node buildNode(String id) {
079        return buildNode(id, defaultSettings());
080    }
081
082    public Node buildNode(String id, Settings settings) {
083        String settingsSource = getClass().getName().replace('.', '/') + ".yml";
084        Settings finalSettings = settingsBuilder()
085                .loadFromClasspath(settingsSource)
086                .put(settings)
087                .put("name", id)
088                .build();
089
090        if (finalSettings.get("gateway.type") == null) {
091            // default to non gateway
092            finalSettings = settingsBuilder().put(finalSettings).put("gateway.type", "none").build();
093        }
094
095        Node node = nodeBuilder()
096                .settings(finalSettings)
097                .build();
098        nodes.put(id, node);
099        clients.put(id, node.client());
100        return node;
101    }
102
103    public void closeNode(String id) {
104        Client client = clients.remove(id);
105        if (client != null) {
106            client.close();
107        }
108        Node node = nodes.remove(id);
109        if (node != null) {
110            node.close();
111        }
112    }
113
114    public Client client(String id) {
115        return clients.get(id);
116    }
117
118    public void closeAllNodes() {
119        for (Client client : clients.values()) {
120            client.close();
121        }
122        clients.clear();
123        for (Node node : nodes.values()) {
124            node.close();
125        }
126        nodes.clear();
127    }
128}
