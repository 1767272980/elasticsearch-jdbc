002    package org.xbib.elasticsearch.river.jdbc.support;
003    
004    import org.elasticsearch.common.io.Streams;
005    import org.elasticsearch.common.logging.ESLogger;
006    import org.elasticsearch.common.logging.ESLoggerFactory;
007    import org.elasticsearch.common.settings.ImmutableSettings;
008    import org.elasticsearch.common.unit.TimeValue;
009    import org.elasticsearch.common.xcontent.XContentHelper;
010    import org.elasticsearch.river.RiverSettings;
011    import org.xbib.elasticsearch.river.jdbc.RiverSource;
012    import org.testng.Assert;
013    import org.testng.annotations.AfterMethod;
014    import org.testng.annotations.BeforeMethod;
015    import org.testng.annotations.Optional;
016    import org.testng.annotations.Parameters;
017    
018    import java.io.BufferedReader;
019    import java.io.IOException;
020    import java.io.InputStream;
021    import java.io.InputStreamReader;
022    import java.sql.Connection;
023    import java.sql.SQLException;
024    import java.sql.Statement;
025    
026    public abstract class AbstractRiverTest extends Assert {
027    
028        private static final ESLogger logger = ESLoggerFactory.getLogger(AbstractRiverTest.class.getName());
029    
030        protected static RiverSource source;
031    
032        protected static RiverContext context;
033    
034        public abstract RiverSource getRiverSource();
035    
036        public abstract RiverContext getRiverContext();
037    
038        @BeforeMethod
039        @Parameters({"driver", "starturl", "user", "password"})
040        public void createClient(String driver, String starturl, String user, String password)
041                throws Exception {
042            source = getRiverSource()
043                    .url(starturl)
044                    .user(user)
045                    .password(password);
046            context = getRiverContext()
047                    .riverSource(source)
048                    .setRetries(1)
049                    .setMaxRetryWait(TimeValue.timeValueSeconds(5))
050                    .locale("en");
051            context.contextualize();
052        }
053    
054        @BeforeMethod(dependsOnMethods = {"createClient"})
055        @Parameters({"create"})
056        public void createTable(@Optional String resourceName) throws Exception {
057            if (resourceName == null || "".equals(resourceName)) {
058                return;
059            }
060            Connection connection = source.connectionForWriting();
061            if (connection == null) {
062                throw new IOException("no connection");
063            }
064            sqlScript(connection, resourceName);
065            source.closeWriting();
066        }
067    
068        @AfterMethod
069        @Parameters({"delete"})
070        public void removeTable(@Optional String resourceName) throws Exception {
071            if (resourceName == null || "".equals(resourceName)) {
072                return;
073            }
074            // before dropping tables, open read connection must be closed to avoid hangs in mysql/postgresql
075            logger.debug("closing reads...");
076            source.closeReading();
077    
078            logger.debug("connecting for close...");
079            Connection connection = source.connectionForWriting();
080            if (connection == null) {
081                throw new IOException("no connection");
082            }
083            logger.debug("cleaning...");
084            // clean up tables
085            sqlScript(connection, resourceName);
086            logger.debug("closing writes...");
087            source.closeWriting();
088        }
089    
090        @AfterMethod(dependsOnMethods = {"removeTable"})
091        @Parameters({"driver", "starturl", "stopurl", "user", "password"})
092        public void removeClient(String driver, String starturl, String stopurl, String user, String password)
093                throws Exception {
094            // some driver can drop database by a magic 'stop' URL
095            source = getRiverSource()
096                    .url(stopurl)
097                    .user(user)
098                    .password(password);
099            try {
100                logger.debug("connecting to stop URL...");
101                // activate stop URL
102                source.connectionForWriting();
103            } catch (Exception e) {
104                // exception is expected, ignore
105            }
106            // close open write connection
107            source.closeWriting();
108            logger.debug("stopped");
109        }
110    
111        protected RiverSettings riverSettings(String resource)
112                throws IOException {
113            InputStream in = getClass().getResourceAsStream(resource);
114            return new RiverSettings(ImmutableSettings.settingsBuilder()
115                    .build(), XContentHelper.convertToMap(Streams.copyToByteArray(in), false).v2());
116        }
117    
118        private void sqlScript(Connection connection, String resourceName) throws Exception {
119            InputStream in = getClass().getResourceAsStream(resourceName);
120            BufferedReader br = new BufferedReader(new InputStreamReader(in, "UTF-8"));
121            String sql;
122            while ((sql = br.readLine()) != null) {
123    
124                try {
125                    logger.trace("executing {}", sql);
126                    Statement p = connection.createStatement();
127                    p.execute(sql);
128                    p.close();
129                } catch (SQLException e) {
130                    // ignore
131                    logger.error(sql + " failed. Reason: " + e.getMessage());
132                } finally {
133                    connection.commit();
134                }
135            }
136            br.close();
137        }
138    
139    }
