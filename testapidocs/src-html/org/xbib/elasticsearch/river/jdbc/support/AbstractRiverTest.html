002package org.xbib.elasticsearch.river.jdbc.support;
003
004import org.elasticsearch.common.io.Streams;
005import org.elasticsearch.common.logging.ESLogger;
006import org.elasticsearch.common.logging.Loggers;
007import org.elasticsearch.common.settings.ImmutableSettings;
008import org.elasticsearch.common.unit.TimeValue;
009import org.elasticsearch.common.xcontent.XContentHelper;
010import org.elasticsearch.river.RiverSettings;
011import org.xbib.elasticsearch.river.jdbc.RiverSource;
012import org.testng.Assert;
013import org.testng.annotations.AfterMethod;
014import org.testng.annotations.BeforeMethod;
015import org.testng.annotations.Optional;
016import org.testng.annotations.Parameters;
017
018import java.io.BufferedReader;
019import java.io.IOException;
020import java.io.InputStream;
021import java.io.InputStreamReader;
022import java.sql.Connection;
023import java.sql.SQLException;
024import java.sql.Statement;
025
026public abstract class AbstractRiverTest extends Assert {
027
028    private static final ESLogger logger = Loggers.getLogger(AbstractRiverTest.class.getSimpleName());
029
030    protected static RiverSource source;
031
032    protected static RiverContext context;
033
034    public abstract RiverSource getRiverSource();
035
036    public abstract RiverContext getRiverContext();
037
038    @BeforeMethod
039    @Parameters({"driver", "starturl", "user", "password"})
040    public void createClient(String driver, String starturl, String user, String password)
041            throws Exception {
042        source = getRiverSource()
043                .driver(driver)
044                .url(starturl)
045                .user(user)
046                .password(password);
047        context = getRiverContext()
048                .riverSource(source)
049                .retries(1)
050                .maxRetryWait(TimeValue.timeValueSeconds(5))
051                .locale("en");
052        context.contextualize();
053    }
054
055    @BeforeMethod(dependsOnMethods = {"createClient"})
056    @Parameters({"create"})
057    public void createTable(@Optional String resourceName) throws Exception {
058        if (resourceName == null || "".equals(resourceName)) {
059            return;
060        }
061        Connection connection = source.connectionForWriting();
062        if (connection == null) {
063            throw new IOException("no connection");
064        }
065        sqlScript(connection, resourceName);
066        source.closeWriting();
067    }
068
069    @AfterMethod
070    @Parameters({"delete"})
071    public void removeTable(@Optional String resourceName) throws Exception {
072        if (resourceName == null || "".equals(resourceName)) {
073            return;
074        }
075        // before dropping tables, open read connection must be closed to avoid hangs in mysql/postgresql
076        logger.debug("closing reads...");
077        source.closeReading();
078
079        logger.debug("connecting for close...");
080        Connection connection = source.connectionForWriting();
081        if (connection == null) {
082            throw new IOException("no connection");
083        }
084        logger.debug("cleaning...");
085        // clean up tables
086        sqlScript(connection, resourceName);
087        logger.debug("closing writes...");
088        source.closeWriting();
089    }
090
091    @AfterMethod(dependsOnMethods = {"removeTable"})
092    @Parameters({"driver", "starturl", "stopurl", "user", "password"})
093    public void removeClient(String driver, String starturl, String stopurl, String user, String password)
094            throws Exception {
095        // some driver can drop database by magic URL
096        source = getRiverSource()
097                .driver(driver)
098                .url(stopurl)
099                .user(user)
100                .password(password);
101        try {
102            logger.debug("connecting to stop URL...");
103            // activate stop URL
104            source.connectionForWriting();
105        } catch (Exception e) {
106            // exception is expected, ignore
107        }
108        // close open write connection
109        source.closeWriting();
110        logger.debug("stopped");
111    }
112
113    protected RiverSettings riverSettings(String resource)
114            throws IOException {
115        InputStream in = getClass().getResourceAsStream(resource);
116        return new RiverSettings(ImmutableSettings.settingsBuilder()
117                .build(), XContentHelper.convertToMap(Streams.copyToByteArray(in), false).v2());
118    }
119
120    private void sqlScript(Connection connection, String resourceName) throws Exception {
121        InputStream in = getClass().getResourceAsStream(resourceName);
122        BufferedReader br = new BufferedReader(new InputStreamReader(in, "UTF-8"));
123        String sql;
124        while ((sql = br.readLine()) != null) {
125            try {
126                logger.debug("executing {}", sql);
127                Statement p = connection.createStatement();
128                p.execute(sql);
129                p.close();
130            } catch (SQLException e) {
131                // ignore
132                logger.error(sql + " failed. Reason: " + e.getMessage());
133            }
134        }
135        br.close();
136    }
137
138}
