001package org.xbib.elasticsearch.river.jdbc.support;
002
003import org.xbib.elasticsearch.plugin.jdbc.util.IndexableObject;
004import org.xbib.elasticsearch.plugin.jdbc.util.PlainKeyValueStreamListener;
005import org.xbib.elasticsearch.river.jdbc.RiverMouth;
006
007import java.io.IOException;
008
009/**
010 * This class consumes pairs from a key/value stream
011 * and transports them to the river mouth.
012 */
013public class StringKeyValueStreamListener extends PlainKeyValueStreamListener<String, String> {
014
015    private RiverMouth output;
016
017    public StringKeyValueStreamListener output(RiverMouth output) {
018        this.output = output;
019        return this;
020    }
021
022    /**
023     * The object is complete. Push it to the river mouth.
024     *
025     * @param object the object
026     * @return this value listener
027     * @throws IOException
028     */
029    public StringKeyValueStreamListener end(IndexableObject object) throws IOException {
030        if (object.isEmpty()) {
031            return this;
032        }
033        if (output != null) {
034            if (object.optype() == null) {
035                output.index(object, false);
036            } else if ("index".equals(object.optype())) {
037                output.index(object, false);
038            } else if ("create".equals(object.optype())) {
039                output.index(object, true);
040            } else if ("delete".equals(object.optype())) {
041                output.delete(object);
042            } else {
043                throw new IllegalArgumentException("unknown optype: " + object.optype());
044            }
045        }
046        return this;
047    }
048
049}
