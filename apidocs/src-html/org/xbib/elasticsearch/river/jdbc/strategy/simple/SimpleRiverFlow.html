002package org.xbib.elasticsearch.river.jdbc.strategy.simple;
003
004import org.elasticsearch.client.Client;
005import org.elasticsearch.common.logging.ESLogger;
006import org.elasticsearch.common.logging.ESLoggerFactory;
007
008import org.xbib.elasticsearch.river.jdbc.support.cron.CronExpression;
009import org.xbib.elasticsearch.river.jdbc.support.cron.CronThreadPoolExecutor;
010import org.xbib.elasticsearch.river.jdbc.RiverFlow;
011import org.xbib.elasticsearch.river.jdbc.RiverMouth;
012import org.xbib.elasticsearch.river.jdbc.RiverSource;
013import org.xbib.elasticsearch.river.jdbc.RiverState;
014import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
015
016import java.util.Date;
017import java.util.concurrent.ScheduledThreadPoolExecutor;
018import java.util.concurrent.TimeUnit;
019
020/**
021 * Simple river flow
022 */
023public class SimpleRiverFlow implements RiverFlow {
024
025    private final ESLogger logger = ESLoggerFactory.getLogger(SimpleRiverFlow.class.getName());
026
027    protected Thread thread;
028
029    protected ScheduledThreadPoolExecutor executor;
030
031    protected RiverContext context;
032
033    protected RiverState state;
034
035    protected boolean abort = false;
036
037    protected ESLogger logger() {
038        return logger;
039    }
040
041    @Override
042    public String strategy() {
043        return "simple";
044    }
045
046    @Override
047    public SimpleRiverFlow riverContext(RiverContext context) {
048        this.context = context;
049        return this;
050    }
051
052    @Override
053    public RiverContext riverContext() {
054        return context;
055    }
056
057    @Override
058    public void schedule(Thread thread) {
059        this.thread = thread;
060        if (context.getSchedule() != null) {
061            // cron-like execution
062            CronExpression cron = new CronExpression(context.getSchedule());
063            CronThreadPoolExecutor executor = new CronThreadPoolExecutor(context.getPoolSize());
064            this.executor = executor;
065            executor.schedule(thread, cron);
066        } else if (context.getInterval() != null && context.getInterval().millis() > 0) {
067            // interval execution
068            ScheduledThreadPoolExecutor executor = new ScheduledThreadPoolExecutor(context.getPoolSize());
069            this.executor = executor;
070            executor.scheduleAtFixedRate(thread, 0L, context.getInterval().millis(), TimeUnit.MILLISECONDS);
071        } else {
072            // single execution
073            thread.start();
074        }
075    }
076
077    @Override
078    public void once(Thread thread) {
079        this.thread = thread;
080        thread.start();
081    }
082
083    /**
084     * The river task run
085     */
086    @Override
087    public void run() {
088        move();
089    }
090
091    /**
092     * A river move.
093     */
094    @Override
095    public void move() {
096        try {
097            if (logger().isDebugEnabled()) {
098                 logger().debug("begin of river move");
099            }
100            RiverSource source = context.riverSource();
101            RiverMouth riverMouth = context.riverMouth();
102            Client client = context.riverMouth().client();
103            riverMouth.waitForCluster();
104            if (state == null) {
105                // create state object
106                this.state = new RiverState()
107                        .name("jdbc")
108                        .setIndex("_river")
109                        .setType(riverContext().riverName())
110                        .setId(ID_INFO_RIVER_INDEX)
111                        .started(new Date().getTime())
112                        .info(context.asMap());
113            }
114            state.load(client);
115            // increment state counter
116            Long counter = state.counter() + 1;
117            this.state = state.counter(counter)
118                .active(true)
119                .timestamp(new Date().getTime());
120            state.save(client);
121            if (logger().isDebugEnabled()) {
122                logger().debug("state saved before fetch");
123            }
124            // set the job number to the state counter
125            context.job(Long.toString(counter));
126            if (logger().isDebugEnabled()) {
127                logger().debug("trying to fetch ...");
128            }
129            source.fetch();
130            if (logger().isDebugEnabled()) {
131                logger().debug("... fetched, flushing");
132            }
133            riverMouth.flush();
134            this.state = state.active(false)
135                .timestamp(new Date().getTime());
136            state.save(client);
137        } catch (Exception e) {
138            logger().error(e.getMessage(), e);
139            // if any error occurs, abort the river
140            abort();
141        } finally {
142            if (logger().isDebugEnabled()) {
143                logger().debug("end of river move");
144            }
145        }
146    }
147
148    /**
149     * Abort the river immediately.
150     */
151    @Override
152    public void abort() {
153        // do not abort twice
154        if (abort) {
155            return;
156        }
157        this.abort = true;
158        logger().warn("aborting river");
159        if (thread != null) {
160            // kill thread
161            thread.interrupt();
162        }
163        if (executor != null) {
164            // do not wait for termination
165            executor.shutdownNow();
166        }
167    }
168
169    @Override
170    public boolean isActive() {
171        return thread != null && thread.isAlive() && !abort;
172    }
173
174    @Override
175    public RiverState riverState() {
176        return state;
177    }
178}
