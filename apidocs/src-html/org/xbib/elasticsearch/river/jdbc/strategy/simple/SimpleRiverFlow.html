045/**
046 * A river flow implementation for the 'simple' strategy.
047 * <p/>
048 * This river flow runs fetch actions in a loop and waits before the next cycle begins.
049 * <p/>
050 * A version counter is incremented each time a fetch move is executed.
051 * <p/>
052 * A checksum is computed over the fetched data. If it differs between runs, it is assumed
053 * the data has changed, and a houskeeper is run to clean up the documents which have
054 * a smaller version than the river state.
055 * <p/>
056 * The state of the river flow is saved between runs. So, in case of a restart, the
057 * river flow will recover with the last known state of the river.
058 *
059 * @author JÃ¶rg Prante <joergprante@gmail.com>
060 */
061public class SimpleRiverFlow implements RiverFlow {
062
063    private final ESLogger logger = ESLoggerFactory.getLogger(SimpleRiverFlow.class.getName());
064
065    protected RiverContext context;
066
067    protected Date startDate;
068
069    protected boolean abort = false;
070
071    @Override
072    public String strategy() {
073        return "simple";
074    }
075
076    @Override
077    public SimpleRiverFlow riverContext(RiverContext context) {
078        this.context = context;
079        return this;
080    }
081
082    @Override
083    public RiverContext riverContext() {
084        return context;
085    }
086
087    /**
088     * Set a start date
089     *
090     * @param creationDate the creation date
091     */
092    @Override
093    public SimpleRiverFlow startDate(Date creationDate) {
094        this.startDate = creationDate;
095        return this;
096    }
097
098    /**
099     * Return the start date of the river task
100     *
101     * @return the creation date
102     */
103    @Override
104    public Date startDate() {
105        return startDate;
106    }
107
108    /**
109     * Delay the connector for poll millis, and notify a reason.
110     *
111     * @param reason the reason for the dealy
112     */
113    @Override
114    public SimpleRiverFlow delay(String reason) {
115        TimeValue poll = context.pollingInterval();
116        if (poll.millis() > 0L) {
117            logger.info("{}, waiting {}", reason, poll);
118            try {
119                Thread.sleep(poll.millis());
120            } catch (InterruptedException e) {
121                logger.debug("Thread interrupted while waiting, stopping");
122                abort();
123            }
124        }
125        return this;
126    }
127
128    /**
129     * Triggers flag to abort the connector down at next run.
130     */
131    @Override
132    public void abort() {
133        this.abort = true;
134    }
135
136    /**
137     * The river task loop. Execute move, check if the task must be aborted, continue with next run after a delay.
138     */
139    @Override
140    public void run() {
141        while (!abort) {
142            move();
143            if (abort) {
144                return;
145            }
146            delay("next run");
147        }
148    }
149
150    /**
151     * A single river move.
152     */
153    @Override
154    public void move() {
155        try {
156            RiverSource source = context.riverSource();
157            RiverMouth target = context.riverMouth();
158            Client client = context.riverMouth().client();
159            Number version;
160            String digest;
161            GetResponse get = null;
162            try {
163                // read state from _custom
164                client.admin().indices().prepareRefresh(context.riverIndexName()).execute().actionGet();
165                get = client.prepareGet(context.riverIndexName(), context.riverName(), ID_INFO_RIVER_INDEX).execute().actionGet();
166            } catch (IndexMissingException e) {
167                logger.warn("river state missing: {}/{}/{}", context.riverIndexName(), context.riverName(), ID_INFO_RIVER_INDEX);
168            }
169            if (get != null && get.isExists()) {
170                Map jdbcState = (Map) get.getSourceAsMap().get("jdbc");
171                if (jdbcState != null) {
172                    version = (Number) jdbcState.get("version");
173                    version = version.longValue() + 1; // increase to next version
174                    digest = (String) jdbcState.get("digest");
175                } else {
176                    throw new IOException("can't retrieve previously persisted state from " + context.riverIndexName() + "/" + context.riverName());
177                }
178            } else {
179                version = 1L;
180                digest = null;
181            }
182            // set default job name to current version number
183            context.job(Long.toString(version.longValue()));
184            String mergeDigest = source.fetch();
185            // this end is required before house keeping starts
186            target.flush();
187            // save state to _custom
188            XContentBuilder builder = jsonBuilder();
189            builder.startObject().startObject("jdbc");
190            if (startDate != null) {
191                builder.field("created", startDate);
192            }
193            builder.field("version", version.longValue());
194            builder.field("digest", mergeDigest);
195            builder.endObject().endObject();
196            if (logger.isDebugEnabled()) {
197                logger.debug(builder.string());
198            }
199            client.prepareBulk().add(indexRequest(context.riverIndexName())
200                    .type(context.riverName())
201                    .id(ID_INFO_RIVER_INDEX)
202                    .source(builder))
203                    .execute().actionGet();
204            // house keeping if data has changed
205            if (digest != null && mergeDigest != null && !mergeDigest.equals(digest)) {
206                versionHouseKeeping(version.longValue());
207                // perform outstanding versionHouseKeeping bulk requests
208                target.flush();
209            }
210        } catch (Exception e) {
211            logger.error(e.getMessage(), e);
212            abort = true;
213        }
214    }
215
216    /**
217     * Do the house keeping for a specific version in the index
218     *
219     * @param version the version
220     * @throws IOException
221     */
222    protected void versionHouseKeeping(long version) throws IOException {
223        logger.info("housekeeping for version " + version);
224        Client client = context.riverMouth().client();
225        String indexName = context.riverMouth().index();
226        String typeName = context.riverMouth().type();
227        int bulkSize = context.riverMouth().maxBulkActions();
228        client.admin().indices().prepareRefresh(indexName).execute().actionGet();
229        SearchResponse response = client.prepareSearch().setIndices(indexName).setTypes(typeName).setSearchType(SearchType.SCAN)
230                .setScroll(TimeValue.timeValueMinutes(10)).setSize(bulkSize).setVersion(true).setQuery(matchAllQuery()).execute().actionGet();
231        if (response.isTimedOut()) {
232            logger.error("housekeeper scan query timeout");
233            return;
234        }
235        if (response.getFailedShards() > 0) {
236            logger.error("housekeeper failed shards in scan response: {}", response.getFailedShards());
237            return;
238        }
239        String scrollId = response.getScrollId();
240        if (scrollId == null) {
241            logger.error("housekeeper failed, no scroll ID");
242            return;
243        }
244        boolean done = false;
245        // scroll
246        long deleted = 0L;
247        long t0 = System.currentTimeMillis();
248        do {
249            response = client.prepareSearchScroll(response.getScrollId()).setScroll(TimeValue.timeValueMinutes(10)).execute().actionGet();
250            if (response.isTimedOut()) {
251                logger.error("housekeeper scroll query timeout");
252                done = true;
253            } else if (response.getFailedShards() > 0) {
254                logger.error("housekeeper failed shards in scroll response: {}", response.getFailedShards());
255                done = true;
256            } else {
257                // terminate scrolling?
258                if (response.getHits() == null) {
259                    done = true;
260                } else {
261                    for (SearchHit hit : response.getHits().getHits()) {
262                        // delete all documents with lower version
263                        if (hit.getVersion() < version) {
264                            context.riverMouth().delete(new PlainStructuredObject()
265                                    .index(hit.getIndex())
266                                    .type(hit.getType())
267                                    .id(hit.getId()));
268                            deleted++;
269                        }
270                    }
271                    scrollId = response.getScrollId();
272                }
273            }
274            if (scrollId != null) {
275                done = true;
276            }
277        } while (!done);
278        long t1 = System.currentTimeMillis();
279        logger.info("housekeeping done, {} documents deleted, took {} ms", deleted, t1 - t0);
280    }
