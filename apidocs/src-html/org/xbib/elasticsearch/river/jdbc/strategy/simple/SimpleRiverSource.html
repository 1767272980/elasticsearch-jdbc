118        /**
119         * Get JDBC connection for reading
120         *
121         * @return the connection
122         * @throws java.sql.SQLException
123         */
124        @Override
125        public Connection connectionForReading() throws SQLException {
126            boolean cond = readConnection == null || readConnection.isClosed();
127            try {
128                cond = cond || !readConnection.isValid(5);
129            } catch (AbstractMethodError e) {
130                // old/buggy JDBC driver
131                logger().debug(e.getMessage());
132            } catch (SQLFeatureNotSupportedException e) {
133                // postgresql does not support isValid()
134                logger().debug(e.getMessage());
135            }
136            if (cond) {
137                int retries = context != null ? context.getRetries() : 1;
138                while (retries > 0) {
139                    retries--;
140                    try {
141                        readConnection = DriverManager.getConnection(url, user, password);
142                        // required by MySQL for large result streaming
143                        readConnection.setReadOnly(true);
144                        // Postgresql cursor mode condition:
145                        // fetchsize > 0, no scrollable result set, no auto commit, no holdable cursors over commit
146                        // https://github.com/pgjdbc/pgjdbc/blob/master/org/postgresql/jdbc2/AbstractJdbc2Statement.java#L514
147                        //readConnection.setHoldability(ResultSet.HOLD_CURSORS_OVER_COMMIT);
148                        if (context != null) {
149                            // many drivers don't like autocommit=true
150                            readConnection.setAutoCommit(context.getAutoCommit());
151                        }
152                        return readConnection;
153                    } catch (SQLException e) {
154                        logger().error("while opening read connection: " + url + " " + e.getMessage(), e);
155                        try {
156                            Thread.sleep(context != null ? context.getMaxRetryWait().millis() : 1000L);
157                        } catch (InterruptedException ex) {
158                            // do nothing
159                        }
160                    }
161                }
162            }
163            return readConnection;
164        }

169         * @return the connection
170         * @throws SQLException

172        @Override
173        public Connection connectionForWriting() throws SQLException {
174            boolean cond = writeConnection == null || writeConnection.isClosed();
175            try {
176                cond = cond || !writeConnection.isValid(5);
177            } catch (AbstractMethodError e) {
178                // old/buggy JDBC driver
179            } catch (SQLFeatureNotSupportedException e) {
180                // postgresql does not support isValid()
181            }
182            if (cond) {
183                int retries = context != null ? context.getRetries() : 1;
184                while (retries > 0) {
185                    retries--;
186                    try {
187                        writeConnection = DriverManager.getConnection(url, user, password);
188                        if (context != null) {
189                            // many drivers don't like autocommit=true
190                            writeConnection.setAutoCommit(context.getAutoCommit());
191                        }
192                        return writeConnection;
193                    } catch (SQLNonTransientConnectionException e) {
194                        // ignore derby drop=true silently
195                    } catch (SQLException e) {
196                        logger().error("while opening write connection: " + url + " " + e.getMessage(), e);
197                        try {
198                            Thread.sleep(context != null ? context.getMaxRetryWait().millis() : 1000L);
199                        } catch (InterruptedException ex) {
200                            // do nothing
201                        }
202                    }
203                }
204            }
205            return writeConnection;
206        }

208        /**
209         * River cycle fetch. Issue a series of SQL statements.
210         *
211         * @throws SQLException
212         * @throws IOException
213         */
214        @Override
215        public void fetch() throws SQLException, IOException {
216            if (logger().isDebugEnabled()) {
217                logger().debug("fetching, {} SQL commands", context.getStatements().size());
218            }
219            try {
220                for (SQLCommand command : context.getStatements()) {
221                    if (command.isCallable()) {
222                        if (logger().isDebugEnabled()) {
223                            logger().debug("executing callable SQL: {}", command);
224                        }
225                        executeCallable(command);
226                    } else if (!command.getParameters().isEmpty()) {
227                        if (logger().isDebugEnabled()) {
228                            logger().debug("executing SQL with params: {}", command);
229                        }
230                        executeWithParameter(command);
231                    } else {
232                        if (logger().isDebugEnabled()) {
233                            logger().debug("executing SQL without params: {}", command);
234                        }
235                        execute(command);
236                    }
237                }
238            } catch (Exception e) {
239                throw new IOException(e);
240            } finally {
241                closeReading();
242                closeWriting();
243            }
244        }

246        /**
247         * Execute SQL query command without parameter binding.
248         * @param command the SQL command
249         * @throws SQLException
250         * @throws IOException
251         */
252        private void execute(SQLCommand command) throws Exception {
253            Statement statement = null;
254            ResultSet results = null;
255            try {
256                if (command.isQuery()) {
257                    // use read connection
258                    // we must not use prepareStatement for Postgresql!
259                    // Postgresql requires direct use of executeQuery(sql) for cursor with fetchsize set.
260                    statement = connectionForReading().createStatement();
261                    results = executeQuery(statement, expandSQL(command.getSQL()));
262                    KeyValueStreamListener listener = new RiverKeyValueStreamListener()
263                            .output(context.riverMouth());
264                    merge(results, listener);
265                } else {
266                    // use write connection
267                    statement = connectionForWriting().createStatement();
268                    executeUpdate(statement, expandSQL(command.getSQL()));
269                }
270            } finally {
271                close(results);
272                close(statement);
273            }
274        }
275    
276        /**
277         * Execute SQL query command with parameter binding.
278         * @param command the SQL command
279         * @throws SQLException
280         * @throws IOException
281         */
282        private void executeWithParameter(SQLCommand command) throws Exception {
283            PreparedStatement statement = null;
284            ResultSet results = null;
285            try {
286                if (command.isQuery()) {
287                    statement = prepareQuery(expandSQL(command.getSQL()));
288                    bind(statement, command.getParameters());
289                    results = executeQuery(statement);
290                    KeyValueStreamListener listener = new RiverKeyValueStreamListener()
291                            .output(context.riverMouth());
292                    merge(results, listener);
293                } else {
294                    statement = prepareUpdate(expandSQL(command.getSQL()));
295                    bind(statement, command.getParameters());
296                    executeUpdate(statement);
297                }
298            } finally {
299                close(results);
300                close(statement);
301            }
302        }

304        /**
305         * Execute callable SQL command
306         * @param command the SQL command
307         * @throws SQLException
308         * @throws IOException
309         */
310        private void executeCallable(SQLCommand command) throws Exception {
311            // call stored procedure
312            CallableStatement statement = null;
313            try {
314                // we do not make a difference betwwen read/write and we assume
315                // it is safe to use the read connection and query the DB
316                statement = connectionForWriting().prepareCall(expandSQL(command.getSQL()));
317                if (!command.getParameters().isEmpty()) {
318                    bind(statement, command.getParameters());
319                }
320                if (!command.getResults().isEmpty()) {
321                    register(statement, command.getResults());
322                }
323                boolean hasRows = statement.execute();
324                KeyValueStreamListener listener = new RiverKeyValueStreamListener()
325                        .output(context.riverMouth());
326                if (!hasRows) {
327                    // merge from registered params
328                    merge(statement, command, listener);
329                } else while (hasRows) {
330                    // merge result set
331                    merge(statement.getResultSet(), listener);
332                    hasRows = statement.getMoreResults();
333                }
334            } finally {
335                close(statement);
336            }
337        }
338    
339        private String expandSQL(String sql) throws IOException {
340            if (sql.endsWith(".sql")) {
341                Reader r = new InputStreamReader(new FileInputStream(sql), "UTF-8");
342                sql = Streams.copyToString(r);
343                r.close();
344            }
345            return sql;
346        }

348        /**
349         * Merge key/values from JDBC result set
350         *
351         * @param results result set
352         * @param listener the value listener
353         * @throws IOException
354         * @throws SQLException
355         */
356        public void merge(ResultSet results, KeyValueStreamListener listener)
357                throws SQLException, IOException, ParseException {
358            if (listener == null) {
359                return;
360            }
361            beforeRows(results, listener);
362            long rows = 0L;
363            while (nextRow(results, listener)) {
364                rows++;
365            }
366            this.lastRowCount = rows;
367            if (logger().isDebugEnabled()) {
368                if (rows > 0) {
369                    logger().debug("merged {} rows", rows);
370                } else {
371                    logger().debug("no rows merged ");
372                }
373            }
374            afterRows(results, listener);
375        }

377        /**
378         * Merge key/values from registered params of a callable statement
379         *
380         * @param statement callable statement
381         * @param listener the value listener
382         * @throws SQLException
383         * @throws IOException
384         */
385        public void merge(CallableStatement statement, SQLCommand command, KeyValueStreamListener listener)
386            throws SQLException, IOException {
387            Map<String,Object> map = command.getResults();
388            if (map.isEmpty()) {
389                return;
390            }
391            List<String> keys = newLinkedList();
392            List<Object> values = newLinkedList();
393            for (String key : map.keySet()) {
394                keys.add(key);
395                Map<String,Object> m = (Map<String,Object>)map.get(key);
396                values.add(statement.getObject((Integer) m.get("pos")));
397            }
398            listener.keys(keys);
399            listener.values(values);
400            listener.end();
401        }

405         *
406         * @param sql the SQL statement
407         * @return a prepared statement
408         * @throws SQLException
409         */
410        @Override
411        public PreparedStatement prepareQuery(String sql) throws SQLException {
412            Connection connection = connectionForReading();
413            if (connection == null) {
414                throw new SQLException("can't connect to source " + url);
415            }
416            logger().debug("preparing statement with SQL {}", sql);
417            return connection.prepareStatement(sql,
418                    ResultSet.TYPE_FORWARD_ONLY,
419                    ResultSet.CONCUR_READ_ONLY);
420        }

443         * Bind values to prepared statement
444         *
445         * @param statement the prepared statement
446         * @param values the values to bind
447         * @throws SQLException
448         */
449        @Override
450        public SimpleRiverSource bind(PreparedStatement statement, List<? extends Object> values) throws SQLException {
451            if (values == null) {
452                logger().warn("no values given for bind");
453                return this;
454            }
455            for (int i = 1; i <= values.size(); i++) {
456                bind(statement, i, values.get(i - 1));
457            }
458            return this;
459        }

461        /**
462         * Register variables in callable statement
463         *
464         * @param statement callable statement
465         * @param values values
466         * @return this river source
467         * @throws SQLException
468         */
469        @Override
470        public SimpleRiverSource register(CallableStatement statement, Map<String,Object> values) throws SQLException {
471            if (values == null) {
472                return this;
473            }
474            for (Map.Entry<String,Object> me : values.entrySet()) {
475                // { "fieldname" : { "pos": n, "type" : "VARCHAR" }, ... }
476                Map<String,Object> m = (Map<String,Object>)me.getValue();
477                Integer n = (Integer)m.get("pos");
478                String type = (String)m.get("type");
479                register(statement, n, type);
480            }
481            return this;
482        }

545        /**
546         * Before rows are read, let the KeyValueStreamListener know about the keys.
547         *
548         * @param results the result set
549         * @param listener the key/value stream listener
550         * @throws SQLException
551         * @throws IOException
552         */
553        @Override
554        public void beforeRows(ResultSet results, KeyValueStreamListener listener)
555                throws SQLException, IOException {
556            ResultSetMetaData metadata = results.getMetaData();
557            int columns = metadata.getColumnCount();
558            List<String> keys = newLinkedList();
559            for (int i = 1; i <= columns; i++) {
560                keys.add(metadata.getColumnLabel(i));
561            }
562            listener.begin();
563            listener.keys(keys);
564        }

566        /**
567         * Get next row and prepare the values for processing. The labels of each
568         * columns are used for the ValueListener as paths for JSON object merging.
569         *
570         * @param results   the result set
571         * @param listener the listener
572         * @return true if row exists and was processed, false otherwise
573         * @throws SQLException
574         * @throws IOException
575         */
576        @Override
577        public boolean nextRow(ResultSet results, KeyValueStreamListener listener)
578                throws SQLException, IOException, ParseException {
579            if (results.next()) {
580                processRow(results, listener);
581                return true;
582            }
583            return false;
584        }

587         * After the rows keys and values, let the listener know about the end of
588         * the result set.
589         * @param results the result set
590         * @param listener the key/value stream listener
591         * @throws SQLException
592         * @throws IOException
593         */
594        @Override
595        public void afterRows(ResultSet results, KeyValueStreamListener listener)
596                throws SQLException, IOException {
597            listener.end();
598        }
599    
600        private void processRow(ResultSet results, KeyValueStreamListener listener)
601                throws SQLException, IOException, ParseException {
602            Locale locale = context != null ? context.locale() != null ? context.locale() : Locale.getDefault() : Locale.getDefault();
603            List<Object> values = newLinkedList();
604            ResultSetMetaData metadata = results.getMetaData();
605            int columns = metadata.getColumnCount();
606            lastRow.clear();
607            for (int i = 1; i <= columns; i++) {
608                Object value = parseType(results, i, metadata.getColumnType(i), locale);
609                if (logger().isTraceEnabled()) {
610                    logger().trace("value={} class={}", value, value != null ? value.getClass().getName() : "");
611                }
612                values.add(value);
613                lastRow.put("$row." + metadata.getColumnLabel(i), results.getObject(i));
614            }
615            if (listener != null) {
616                listener.values(values);
617            }
618        }

650         */
651        @Override
652        public SimpleRiverSource closeReading() {
653            try {
654                if (readConnection != null) {
655                    // always commit before close to finish cursors/transactions
656                    if (!readConnection.getAutoCommit()) {
657                        readConnection.commit();
658                    }
659                    if (!readConnection.isClosed()) {
660                        readConnection.close();
661                    }
662                }
663            } catch (SQLException e) {
664                logger().warn("while closing read connection: " + e.getMessage());
665            }
666            return this;

690        @Override
691        public SimpleRiverSource rounding(String rounding) {
692            if ("ceiling".equalsIgnoreCase(rounding)) {
693                this.rounding = BigDecimal.ROUND_CEILING;
694            } else if ("down".equalsIgnoreCase(rounding)) {
695                this.rounding = BigDecimal.ROUND_DOWN;
696            } else if ("floor".equalsIgnoreCase(rounding)) {
697                this.rounding = BigDecimal.ROUND_FLOOR;
698            } else if ("halfdown".equalsIgnoreCase(rounding)) {
699                this.rounding = BigDecimal.ROUND_HALF_DOWN;
700            } else if ("halfeven".equalsIgnoreCase(rounding)) {
701                this.rounding = BigDecimal.ROUND_HALF_EVEN;
702            } else if ("halfup".equalsIgnoreCase(rounding)) {
703                this.rounding = BigDecimal.ROUND_HALF_UP;
704            } else if ("unnecessary".equalsIgnoreCase(rounding)) {
705                this.rounding = BigDecimal.ROUND_UNNECESSARY;
706            } else if ("up".equalsIgnoreCase(rounding)) {
707                this.rounding = BigDecimal.ROUND_UP;
708            }
709            return this;
710        }

724        private void bind(PreparedStatement statement, int i, Object value) throws SQLException {
725            if (value == null) {
726                statement.setNull(i, Types.VARCHAR);
727            } else if (value instanceof String) {
728                String s = (String) value;
729                if (logger().isDebugEnabled()) {
730                    logger().debug("bind: value = {}", s);
731                }
732                if ("$now".equals(s)) {
733                    Timestamp t = new Timestamp(new java.util.Date().getTime());
734                    if (logger().isDebugEnabled()) {
735                        logger().debug("setting $now to {}", t);
736                    }
737                    statement.setTimestamp(i, t);
738                } else if ("$job".equals(s)) {
739                    if (logger().isDebugEnabled()) {
740                        logger().debug("setting $job to {}", context.job());
741                    }
742                    statement.setString(i, context.job());
743                } else if ("$count".equals(s)) {
744                    if (logger().isDebugEnabled()) {
745                        logger().debug("setting $count to {}", lastRowCount);
746                    }
747                    statement.setLong(i, lastRowCount);
748                } else if ("$river.name".equals(s)) {
749                    String name = context.riverName();
750                    statement.setString(i, name);
751                } else if ("$river.state.timestamp".equals(s)) {
752                    Long timestamp = context.riverFlow().riverState().timestamp();
753                    statement.setTimestamp(i, new Timestamp(timestamp));
754                } else if ("$river.state.started".equals(s)) {
755                    Long started = context.riverFlow().riverState().started();
756                    statement.setTimestamp(i, new Timestamp(started));
757                } else if ("$river.state.counter".equals(s)) {
758                    Long counter = context.riverFlow().riverState().counter();
759                    statement.setLong(i, counter);
760                } else {
761                    Object rowValue = lastRow.get(s);
762                    if (rowValue != null) {
763                        if (logger().isDebugEnabled()) {
764                            logger().debug("setting ${} to {}", s, rowValue);
765                        }
766                        statement.setObject(i, rowValue);
767                    } else {
768                        statement.setString(i, (String) value);
769                    }
770                }
771            } else if (value instanceof Integer) {
772                statement.setInt(i, (Integer) value);
773            } else if (value instanceof Long) {
774                statement.setLong(i, (Long) value);
775            } else if (value instanceof BigDecimal) {
776                statement.setBigDecimal(i, (BigDecimal) value);
777            } else if (value instanceof Date) {
778                statement.setDate(i, (Date) value);
779            } else if (value instanceof Timestamp) {
780                statement.setTimestamp(i, (Timestamp) value);
781            } else if (value instanceof Float) {
782                statement.setFloat(i, (Float) value);
783            } else if (value instanceof Double) {
784                statement.setDouble(i, (Double) value);
785            } else {
786                statement.setObject(i, value);
787            }
788        }

794        /**
795         * Parse of value of result set
796         *
797         * @param result the result set
798         * @param i the offset in the result set
799         * @param type the JDBC type
800         * @param locale the locale to use for parsing
801         * @return The parse value
802         * @throws SQLException
803         * @throws IOException
804         */
805        @Override
806        public Object parseType(ResultSet result, Integer i, int type, Locale locale)
807                throws SQLException, IOException, ParseException {
808            if (logger().isTraceEnabled()) {
809                logger().trace("{} {} {}", i, type, result.getString(i));
810            }
811            switch (type) {
812                /**
813                 * The JDBC types CHAR, VARCHAR, and LONGVARCHAR are closely
814                 * related. CHAR represents a small, fixed-length character string,
815                 * VARCHAR represents a small, variable-length character string, and
816                 * LONGVARCHAR represents a large, variable-length character string.
817                 */
818                case Types.CHAR:
819                case Types.VARCHAR:
820                case Types.LONGVARCHAR: {
821                    return result.getString(i);
822                }
823                case Types.NCHAR:
824                case Types.NVARCHAR:
825                case Types.LONGNVARCHAR: {
826                    return result.getNString(i);
827                }
828                /**
829                 * The JDBC types BINARY, VARBINARY, and LONGVARBINARY are closely
830                 * related. BINARY represents a small, fixed-length binary value,
831                 * VARBINARY represents a small, variable-length binary value, and
832                 * LONGVARBINARY represents a large, variable-length binary value
833                 */
834                case Types.BINARY:
835                case Types.VARBINARY:
836                case Types.LONGVARBINARY: {
837                    return result.getBytes(i);
838                }
839                /**
840                 * The JDBC type ARRAY represents the SQL3 type ARRAY.
841                 *
842                 * An ARRAY value is mapped to an instance of the Array interface in
843                 * the Java programming language. If a driver follows the standard
844                 * implementation, an Array object logically points to an ARRAY
845                 * value on the server rather than containing the elements of the
846                 * ARRAY object, which can greatly increase efficiency. The Array
847                 * interface contains methods for materializing the elements of the
848                 * ARRAY object on the client in the form of either an array or a
849                 * ResultSet object.
850                 */
851                case Types.ARRAY: {
852                    return result.getArray(i).getArray();
853                }
854                /**
855                 * The JDBC type BIGINT represents a 64-bit signed integer value
856                 * between -9223372036854775808 and 9223372036854775807.
857                 *
858                 * The corresponding SQL type BIGINT is a nonstandard extension to
859                 * SQL. In practice the SQL BIGINT type is not yet currently
860                 * implemented by any of the major databases, and we recommend that
861                 * its use be avoided in code that is intended to be portable.
862                 *
863                 * The recommended Java mapping for the BIGINT type is as a Java
864                 * long.
865                 */
866                case Types.BIGINT: {
867                    Object o = result.getLong(i);
868                    return result.wasNull() ? null : o;
869                }
870                /**
871                 * The JDBC type BIT represents a single bit value that can be zero
872                 * or one.
873                 *
874                 * SQL-92 defines an SQL BIT type. However, unlike the JDBC BIT
875                 * type, this SQL-92 BIT type can be used as a parameterized type to
876                 * define a fixed-length binary string. Fortunately, SQL-92 also
877                 * permits the use of the simple non-parameterized BIT type to
878                 * represent a single binary digit, and this usage corresponds to
879                 * the JDBC BIT type. Unfortunately, the SQL-92 BIT type is only
880                 * required in "full" SQL-92 and is currently supported by only a
881                 * subset of the major databases. Portable code may therefore prefer
882                 * to use the JDBC SMALLINT type, which is widely supported.
883                 */
884                case Types.BIT: {
885                    try {
886                        Object o = result.getInt(i);
887                        return result.wasNull() ? null : o;
888                    } catch (Exception e) {
889                        // PSQLException: Bad value for type int : t
890                        if (e.getMessage().startsWith("Bad value for type int")) {
891                            return "t".equals(result.getString(i));
892                        }
893                        throw new IOException(e);
894                    }
895                }
896                /**
897                 * The JDBC type BOOLEAN, which is new in the JDBC 3.0 API, maps to
898                 * a boolean in the Java programming language. It provides a
899                 * representation of true and false, and therefore is a better match
900                 * than the JDBC type BIT, which is either 1 or 0.
901                 */
902                case Types.BOOLEAN: {
903                    return result.getBoolean(i);
904                }
905                /**
906                 * The JDBC type BLOB represents an SQL3 BLOB (Binary Large Object).
907                 *
908                 * A JDBC BLOB value is mapped to an instance of the Blob interface
909                 * in the Java programming language. If a driver follows the
910                 * standard implementation, a Blob object logically points to the
911                 * BLOB value on the server rather than containing its binary data,
912                 * greatly improving efficiency. The Blob interface provides methods
913                 * for materializing the BLOB data on the client when that is
914                 * desired.
915                 */
916                case Types.BLOB: {
917                    Blob blob = result.getBlob(i);
918                    if (blob != null) {
919                        long n = blob.length();
920                        if (n > Integer.MAX_VALUE) {
921                            throw new IOException("can't process blob larger than Integer.MAX_VALUE");
922                        }
923                        byte[] tab = blob.getBytes(1, (int) n);
924                        blob.free();
925                        return tab;
926                    }
927                    break;
928                }
929                /**
930                 * The JDBC type CLOB represents the SQL3 type CLOB (Character Large
931                 * Object).
932                 *
933                 * A JDBC CLOB value is mapped to an instance of the Clob interface
934                 * in the Java programming language. If a driver follows the
935                 * standard implementation, a Clob object logically points to the
936                 * CLOB value on the server rather than containing its character
937                 * data, greatly improving efficiency. Two of the methods on the
938                 * Clob interface materialize the data of a CLOB object on the
939                 * client.
940                 */
941                case Types.CLOB: {
942                    Clob clob = result.getClob(i);
943                    if (clob != null) {
944                        long n = clob.length();
945                        if (n > Integer.MAX_VALUE) {
946                            throw new IOException("can't process clob larger than Integer.MAX_VALUE");
947                        }
948                        String str = clob.getSubString(1, (int) n);
949                        clob.free();
950                        return str;
951                    }
952                    break;
953                }
954                case Types.NCLOB: {
955                    NClob nclob = result.getNClob(i);
956                    if (nclob != null) {
957                        long n = nclob.length();
958                        if (n > Integer.MAX_VALUE) {
959                            throw new IOException("can't process nclob larger than Integer.MAX_VALUE");
960                        }
961                        String str = nclob.getSubString(1, (int) n);
962                        nclob.free();
963                        return str;
964                    }
965                    break;
966                }
967                /**
968                 * The JDBC type DATALINK, new in the JDBC 3.0 API, is a column
969                 * value that references a file that is outside of a data source but
970                 * is managed by the data source. It maps to the Java type
971                 * java.net.URL and provides a way to manage external files. For
972                 * instance, if the data source is a DBMS, the concurrency controls
973                 * it enforces on its own data can be applied to the external file
974                 * as well.
975                 *
976                 * A DATALINK value is retrieved from a ResultSet object with the
977                 * ResultSet methods getURL or getObject. If the Java platform does
978                 * not support the type of URL returned by getURL or getObject, a
979                 * DATALINK value can be retrieved as a String object with the
980                 * method getString.
981                 *
982                 * java.net.URL values are stored in a database using the method
983                 * setURL. If the Java platform does not support the type of URL
984                 * being set, the method setString can be used instead.
985                 *
986                 *
987                 */
988                case Types.DATALINK: {
989                    return result.getURL(i);
990                }
991                /**
992                 * The JDBC DATE type represents a date consisting of day, month,
993                 * and year. The corresponding SQL DATE type is defined in SQL-92,
994                 * but it is implemented by only a subset of the major databases.
995                 * Some databases offer alternative SQL types that support similar
996                 * semantics.
997                 */
998                case Types.DATE: {
999                    try {
1000                        Date d = result.getDate(i);
1001                        return d != null ? formatDateISO(d.getTime()) : null;
1002                    } catch (SQLException e) {
1003                        return null;
1004                    }
1005                }
1006                case Types.TIME: {
1007                    try {
1008                        Time t = result.getTime(i);
1009                        return t != null ? formatDateISO(t.getTime()) : null;
1010                    } catch (SQLException e) {
1011                        return null;
1012                    }
1013                }
1014                case Types.TIMESTAMP: {
1015                    try {
1016                        Timestamp t = result.getTimestamp(i);
1017                        return t != null ? formatDateISO(t.getTime()) : null;
1018                    } catch (SQLException e) {
1019                        // java.sql.SQLException: Cannot convert value '0000-00-00 00:00:00' from column ... to TIMESTAMP.
1020                        return null;
1021                    }
1022                }
1023                /**
1024                 * The JDBC types DECIMAL and NUMERIC are very similar. They both
1025                 * represent fixed-precision decimal values.
1026                 *
1027                 * The corresponding SQL types DECIMAL and NUMERIC are defined in
1028                 * SQL-92 and are very widely implemented. These SQL types take
1029                 * precision and scale parameters. The precision is the total number
1030                 * of decimal digits supported, and the scale is the number of
1031                 * decimal digits after the decimal point. For most DBMSs, the scale
1032                 * is less than or equal to the precision. So for example, the value
1033                 * "12.345" has a precision of 5 and a scale of 3, and the value
1034                 * ".11" has a precision of 2 and a scale of 2. JDBC requires that
1035                 * all DECIMAL and NUMERIC types support both a precision and a
1036                 * scale of at least 15.
1037                 *
1038                 * The sole distinction between DECIMAL and NUMERIC is that the
1039                 * SQL-92 specification requires that NUMERIC types be represented
1040                 * with exactly the specified precision, whereas for DECIMAL types,
1041                 * it allows an implementation to add additional precision beyond
1042                 * that specified when the type was created. Thus a column created
1043                 * with type NUMERIC(12,4) will always be represented with exactly
1044                 * 12 digits, whereas a column created with type DECIMAL(12,4) might
1045                 * be represented by some larger number of digits.
1046                 *
1047                 * The recommended Java mapping for the DECIMAL and NUMERIC types is
1048                 * java.math.BigDecimal. The java.math.BigDecimal type provides math
1049                 * operations to allow BigDecimal types to be added, subtracted,
1050                 * multiplied, and divided with other BigDecimal types, with integer
1051                 * types, and with floating point types.
1052                 *
1053                 * The method recommended for retrieving DECIMAL and NUMERIC values
1054                 * is ResultSet.getBigDecimal. JDBC also allows access to these SQL
1055                 * types as simple Strings or arrays of char. Thus, Java programmers
1056                 * can use getString to receive a DECIMAL or NUMERIC result.
1057                 * However, this makes the common case where DECIMAL or NUMERIC are
1058                 * used for currency values rather awkward, since it means that
1059                 * application writers have to perform math on strings. It is also
1060                 * possible to retrieve these SQL types as any of the Java numeric
1061                 * types.
1062                 */
1063                case Types.DECIMAL:
1064                case Types.NUMERIC: {
1065                    BigDecimal bd = null;
1066                    try {
1067                        bd = result.getBigDecimal(i);
1068                    } catch (NullPointerException e) {
1069                        // getBigDecimal() should get obsolete. Most seem to use getString/getObject anyway...
1070                        // But is it true? JDBC NPE exists since 13 years?
1071                        // http://forums.codeguru.com/archive/index.php/t-32443.html
1072                        // Null values are driving us nuts in JDBC:
1073                        // http://stackoverflow.com/questions/2777214/when-accessing-resultsets-in-jdbc-is-there-an-elegant-way-to-distinguish-betwee
1074                    }
1075                    if (bd == null || result.wasNull()) {
1076                        return null;
1077                    }
1078                    if (scale >= 0) {
1079                        bd = bd.setScale(scale, rounding);
1080                        try {
1081                            long l = bd.longValueExact();
1082                            if (Long.toString(l).equals(result.getString(i))) {
1083                                // convert to long if possible
1084                                return l;
1085                            } else {
1086                                // convert to double (with precision loss)
1087                                return bd.doubleValue();
1088                            }
1089                        } catch (ArithmeticException e) {
1090                            return bd.doubleValue();
1091                        }
1092                    } else {
1093                        return bd.toPlainString();
1094                    }
1095                }

1098                 * point number that supports 15 digits of mantissa.
1099                 *
1100                 * The corresponding SQL type is DOUBLE PRECISION, which is defined
1101                 * in SQL-92 and is widely supported by the major databases. The
1102                 * SQL-92 standard leaves the precision of DOUBLE PRECISION up to
1103                 * the implementation, but in practice all the major databases
1104                 * supporting DOUBLE PRECISION support a mantissa precision of at
1105                 * least 15 digits.

1107                 * The recommended Java mapping for the DOUBLE type is as a Java

1109                 */

1117                    return number.doubleValue();

1125                 *
1126                 * The corresponding SQL type FLOAT is defined in SQL-92. The SQL-92

1128                 * but in practice all the major databases supporting FLOAT support
1129                 * a mantissa precision of at least 15 digits.
1130                 *
1131                 * The recommended Java mapping for the FLOAT type is as a Java
1132                 * double. However, because of the potential confusion between the
1133                 * double precision SQL FLOAT and the single precision Java float,
1134                 * we recommend that JDBC programmers should normally use the JDBC
1135                 * DOUBLE type in preference to FLOAT.
1136                 */
1137                case Types.FLOAT: {

1140                        return null;
1141                    }
1142                    NumberFormat format = NumberFormat.getInstance(locale);
1143                    Number number = format.parse(s);
1144                    return number.doubleValue();

1150                 * instance of a class defined in the Java programming language that
1151                 * is stored as a database object. The type JAVA_OBJECT is used by a
1152                 * database whose type system has been extended so that it can store
