173    /**
174     * Get JDBC connection for reading
175     *
176     * @return the connection
177     * @throws SQLException when SQL execution gives an error
178     */
179    @Override
180    public Connection getConnectionForReading() throws SQLException {
181        boolean cond = readConnection == null || readConnection.isClosed();
182        try {
183            cond = cond || !readConnection.isValid(5);
184        } catch (AbstractMethodError e) {
185            // old/buggy JDBC driver
186            logger().debug(e.getMessage());
187        } catch (SQLFeatureNotSupportedException e) {
188            // postgresql does not support isValid()
189            logger().debug(e.getMessage());
190        }
191        if (cond) {
192            int retries = context != null ? context.getRetries() : 1;
193            while (retries > 0) {
194                retries--;
195                try {
196                    readConnection = DriverManager.getConnection(url, user, password);
197                    if (context.shouldPrepareDatabaseMetadata()) {
198                        prepare(readConnection.getMetaData());
199                    }
200                    // required by MySQL for large result streaming
201                    readConnection.setReadOnly(true);
202                    // Postgresql cursor mode condition:
203                    // fetchsize > 0, no scrollable result set, no auto commit, no holdable cursors over commit
204                    // https://github.com/pgjdbc/pgjdbc/blob/master/org/postgresql/jdbc2/AbstractJdbc2Statement.java#L514
205                    //readConnection.setHoldability(ResultSet.HOLD_CURSORS_OVER_COMMIT);
206                    if (context != null) {
207                        // many drivers don't like autocommit=true
208                        readConnection.setAutoCommit(context.getAutoCommit());
209                    }
210                    return readConnection;
211                } catch (SQLException e) {
212                    logger().error("while opening read connection: " + url + " " + e.getMessage(), e);
213                    try {
214                        Thread.sleep(context != null ? context.getMaxRetryWait().millis() : 1000L);
215                    } catch (InterruptedException ex) {
216                        // do nothing
217                    }
218                }
219            }
220        }
221        return readConnection;
222    }

229     */
230    @Override
231    public Connection getConnectionForWriting() throws SQLException {
232        boolean cond = writeConnection == null || writeConnection.isClosed();
233        try {
234            cond = cond || !writeConnection.isValid(5);
235        } catch (AbstractMethodError e) {
236            // old/buggy JDBC driver
237        } catch (SQLFeatureNotSupportedException e) {
238            // postgresql does not support isValid()
239        }
240        if (cond) {
241            int retries = context != null ? context.getRetries() : 1;
242            while (retries > 0) {
243                retries--;
244                try {
245                    writeConnection = DriverManager.getConnection(url, user, password);
246                    if (context != null) {
247                        // many drivers don't like autocommit=true
248                        writeConnection.setAutoCommit(context.getAutoCommit());
249                    }

251                } catch (SQLNonTransientConnectionException e) {
252                    // ignore derby drop=true silently
253                } catch (SQLException e) {
254                    logger().error("while opening write connection: " + url + " " + e.getMessage(), e);
255                    try {
256                        Thread.sleep(context != null ? context.getMaxRetryWait().millis() : 1000L);
257                    } catch (InterruptedException ex) {
258                        // do nothing
259                    }
260                }
261            }
262        }
263        return writeConnection;
264    }

266    /**
267     * River cycle fetch. Issue a series of SQL statements.
268     *
269     * @throws SQLException when SQL execution gives an error
270     * @throws IOException  when input/output error occurs
271     */
272    @Override
273    public void fetch() throws SQLException, IOException {
274       logger().debug("fetching, {} SQL commands", context.getStatements().size());
275        context.setLastStartDate(new java.util.Date().getTime());
276        try {
277            for (SQLCommand command : context.getStatements()) {
278                context.setLastExecutionStartDate(new java.util.Date().getTime());
279                if (command.isCallable()) {
280                    logger().debug("executing callable SQL: {}", command);
281                    executeCallable(command);
282                } else if (!command.getParameters().isEmpty()) {
283                    logger().debug("executing SQL with params: {}", command);
284                    executeWithParameter(command);
285                } else {
286                    logger().debug("executing SQL without params: {}", command);
287                    execute(command);
288                }
289                context.setLastExecutionEndDate(new java.util.Date().getTime());
290            }
291        } catch (Exception e) {
292            throw new IOException(e);
293        } finally {
294            context.setLastEndDate(new java.util.Date().getTime());
295            closeReading();
296            closeWriting();
297        }
298    }

300    /**
301     * Execute SQL query command without parameter binding.
302     *
303     * @param command the SQL command
304     * @throws SQLException when SQL execution gives an error
305     * @throws IOException  when input/output error occurs
306     */
307    private void execute(SQLCommand command) throws Exception {
308        Statement statement = null;
309        ResultSet results = null;
310        try {
311            if (command.isQuery()) {
312                // use read connection
313                // we must not use prepareStatement for Postgresql!
314                // Postgresql requires direct use of executeQuery(sql) for cursor with fetchsize set.
315                Connection connection = getConnectionForReading();
316                if (connection != null) {
317                    statement = connection.createStatement();
318                    results = executeQuery(statement, command.getSQL());
319                    if (context.shouldPrepareResultSetMetadata()) {
320                        prepare(results.getMetaData());
321                    }
322                    RiverMouthKeyValueStreamListener<Object, Object> listener = new RiverMouthKeyValueStreamListener<Object, Object>()
323                            .output(context.getRiverMouth())
324                            .shouldIgnoreNull(context.shouldIgnoreNull());
325                    merge(command, results, listener);
326                }
327            } else {
328                // use write connection
329                Connection connection = getConnectionForWriting();
330                if (connection != null) {
331                    statement = connection.createStatement();
332                    executeUpdate(statement, command.getSQL());
333                }
334            }
335        } finally {
336            close(results);
337            close(statement);
338        }
339    }

341    /**
342     * Execute SQL query command with parameter binding.
343     *
344     * @param command the SQL command
345     * @throws SQLException when SQL execution gives an error
346     * @throws IOException  when input/output error occurs
347     */
348    private void executeWithParameter(SQLCommand command) throws Exception {
349        PreparedStatement statement = null;
350        ResultSet results = null;
351        try {
352            if (command.isQuery()) {
353                statement = prepareQuery(command.getSQL());
354                bind(statement, command.getParameters());
355                results = executeQuery(statement);
356                RiverMouthKeyValueStreamListener<Object, Object> listener = new RiverMouthKeyValueStreamListener<Object, Object>()
357                        .output(context.getRiverMouth())
358                        .shouldIgnoreNull(context.shouldIgnoreNull());
359                merge(command, results, listener);
360            } else {
361                statement = prepareUpdate(command.getSQL());
362                bind(statement, command.getParameters());
363                executeUpdate(statement);
364            }
365        } finally {
366            close(results);
367            close(statement);
368        }
369    }

371    /**
372     * Execute callable SQL command
373     *
374     * @param command the SQL command
375     * @throws SQLException when SQL execution gives an error
376     * @throws IOException  when input/output error occurs
377     */
378    private void executeCallable(SQLCommand command) throws Exception {
379        // call stored procedure
380        CallableStatement statement = null;
381        try {
382            // we do not make a difference betwwen read/write and we assume
383            // it is safe to use the read connection and query the DB
384            statement = getConnectionForWriting().prepareCall(command.getSQL());
385            if (!command.getParameters().isEmpty()) {
386                bind(statement, command.getParameters());
387            }
388            if (!command.getRegister().isEmpty()) {
389                register(statement, command.getRegister());
390            }
391            boolean hasRows = statement.execute();
392            RiverMouthKeyValueStreamListener<Object, Object> listener = new RiverMouthKeyValueStreamListener<Object, Object>()
393                    .output(context.getRiverMouth());
394            if (hasRows) {
395                logger.debug("callable execution created result set");
396                while (hasRows) {
397                    // merge result set, but use register
398                    merge(command, statement.getResultSet(), listener);
399                    hasRows = statement.getMoreResults();
400                }
401            } else {
402                // no result set, merge from registered params only
403                merge(command, statement, listener);
404            }
405        } finally {
406            close(statement);
407        }
408    }

410    /**
411     * Merge key/values from JDBC result set
412     *
413     * @param results  result set
414     * @param listener the value listener
415     * @throws SQLException when SQL execution gives an error
416     * @throws IOException  when input/output error occurs
417     */
418    public void merge(SQLCommand command, ResultSet results, KeyValueStreamListener listener)
419            throws SQLException, IOException, ParseException {
420        if (listener == null) {
421            return;
422        }
423        beforeRows(command, results, listener);
424        long rows = 0L;
425        while (nextRow(command, results, listener)) {
426            rows++;
427        }
428        context.setLastRowCount(rows);
429        if (rows > 0) {
430            logger().debug("merged {} rows", rows);
431        } else {
432            logger().debug("no rows merged ");
433        }
434        afterRows(command, results, listener);
435    }

440     * Prepare a query statement
441     *
442     * @param sql the SQL statement
443     * @return a prepared statement
444     * @throws SQLException when SQL execution gives an error
445     */
446    @Override
447    public PreparedStatement prepareQuery(String sql) throws SQLException {
448        Connection connection = getConnectionForReading();
449        if (connection == null) {
450            throw new SQLException("can't connect to source " + url);
451        }
452        logger().debug("preparing statement with SQL {}", sql);
453        int type =
454                "TYPE_FORWARD_ONLY".equals(context.getResultSetType()) ? ResultSet.TYPE_FORWARD_ONLY :
455                        "TYPE_SCROLL_SENSITIVE".equals(context.getResultSetType()) ? ResultSet.TYPE_SCROLL_SENSITIVE :
456                                "TYPE_SCROLL_INSENSITIVE".equals(context.getResultSetType()) ? ResultSet.TYPE_SCROLL_INSENSITIVE :
457                                        ResultSet.TYPE_FORWARD_ONLY;
458        int concurrency = "CONCUR_READ_ONLY".equals(context.getResultSetConcurrency()) ?
459                ResultSet.CONCUR_READ_ONLY : ResultSet.CONCUR_UPDATABLE;
460        return connection.prepareStatement(sql, type, concurrency);
461    }

498    /**
499     * Merge key/values from registered params of a callable statement
500     *
501     * @param statement callable statement
502     * @param listener  the value listener
503     * @throws SQLException when SQL execution gives an error
504     * @throws IOException  when input/output error occurs
505     */
506    @SuppressWarnings({"unchecked"})
507    public void merge(SQLCommand command, CallableStatement statement, KeyValueStreamListener listener)
508            throws SQLException, IOException {
509        Map<String, Object> map = command.getRegister();
510        if (map.isEmpty()) {
511            // no register given, return without doing anything
512            return;
513        }
514        List<String> keys = newLinkedList();
515        List<Object> values = newLinkedList();
516        for (Map.Entry<String, Object> entry : map.entrySet()) {
517            String k = entry.getKey();
518            Map<String, Object> v = (Map<String, Object>) entry.getValue();
519            Integer pos = (Integer) v.get("pos"); // the parameter position of the value
520            String field = (String) v.get("field"); // the field for indexing the value (if not key name)
521            keys.add(field != null ? field : k);
522            values.add(statement.getObject(pos));
523        }
524        logger.trace("merge callable statement result: keys={} values={}", keys, values);
525        listener.keys(keys);
526        listener.values(values);
527        listener.end();
528    }

530    /**
531     * Register variables in callable statement
532     *
533     * @param statement callable statement
534     * @param values    values
535     * @return this river source
536     * @throws SQLException when SQL execution gives an error
537     */
538    @Override
539    @SuppressWarnings({"unchecked"})
540    public SimpleRiverSource register(CallableStatement statement, Map<String, Object> values) throws SQLException {
541        if (values == null) {
542            return this;
543        }
544        for (Map.Entry<String, Object> me : values.entrySet()) {
545            // { "key" : { "pos": n, "type" : "VARCHAR", "field" : "fieldname" }, ... }
546            Map<String, Object> m = (Map<String, Object>) me.getValue();
547            Integer n = (Integer) m.get("pos");
548            String type = (String) m.get("type");
549            if (n != null && type != null) {
550                logger.debug("n={} type={}", n, toJDBCType(type));
551                try {
552                    statement.registerOutParameter(n, toJDBCType(type));
553                } catch (Throwable t) {
554                    logger.warn("can't register out parameter " + n + " of type " + type);
555                }
556            }
557        }
558        return this;
559    }

628    /**
629     * Before rows are read, let the KeyValueStreamListener know about the keys.
630     * If the SQL command was a callable statement and a register is there, look into the register map
631     * for the key names, not in the result set metadata.
632     *
633     * @param command the SQL command that created this result set
634     * @param results  the result set
635     * @param listener the key/value stream listener
636     * @throws SQLException when SQL execution gives an error
637     * @throws IOException  when input/output error occurs
638     */
639    @Override
640    @SuppressWarnings({"unchecked"})
641    public void beforeRows(SQLCommand command, ResultSet results, KeyValueStreamListener listener)
642            throws SQLException, IOException {
643        List<String> keys = new LinkedList();
644        if (command != null && command.isCallable() && !command.getRegister().isEmpty()) {
645            for (Map.Entry<String,Object> me : command.getRegister().entrySet()) {
646                keys.add(me.getKey());
647            }
648        } else {
649            ResultSetMetaData metadata = results.getMetaData();
650            int columns = metadata.getColumnCount();
651            for (int i = 1; i <= columns; i++) {
652                keys.add(metadata.getColumnLabel(i));
653            }
654        }
655        listener.begin();
656        listener.keys(keys);
657    }

664    /**
665     * Get next row and prepare the values for processing. The labels of each
666     * columns are used for the ValueListener as paths for JSON object merging.
667     *
668     * @param command the SQL command that created this result set
669     * @param results  the result set
670     * @param listener the listener
671     * @return true if row exists and was processed, false otherwise
672     * @throws SQLException when SQL execution gives an error
673     * @throws IOException  when input/output error occurs
674     */
675    @Override
676    public boolean nextRow(SQLCommand command, ResultSet results, KeyValueStreamListener listener)
677            throws SQLException, IOException {
678        if (results.next() && (context.getRiverFlow() == null || !context.getRiverFlow().getFeeder().isInterrupted())) {
679            processRow(command, results, listener);
680            return true;
681        }
682        return false;
683    }

706    @SuppressWarnings({"unchecked"})
707    private void processRow(SQLCommand command, ResultSet results, KeyValueStreamListener listener)
708            throws SQLException, IOException {
709        Locale locale = context != null ? context.getLocale() != null ? context.getLocale() : Locale.getDefault() : Locale.getDefault();
710        List<Object> values = new LinkedList<Object>();
711        ResultSetMetaData metadata = results.getMetaData();
712        int columns = metadata.getColumnCount();
713        context.getLastRow().clear();
714        for (int i = 1; i <= columns; i++) {
715            try {
716                Object value = parseType(results, i, metadata.getColumnType(i), locale);
717                logger().trace("value={} class={}", value, value != null ? value.getClass().getName() : "");
718                values.add(value);
719                context.getLastRow().put("$row." + metadata.getColumnLabel(i), value);
720            } catch (ParseException e) {
721                logger().warn("parse error for value {}, using null instead", results.getObject(i));
722                values.add(null);
723            }
724        }
725        if (listener != null) {
726            listener.values(values);
727        }
728    }

730    /**
731     * Close result set
732     *
733     * @param result the result set to be closed or null
734     * @throws SQLException when SQL execution gives an error
735     */
736    @Override
737    public SimpleRiverSource close(ResultSet result) throws SQLException {

885    private void bind(PreparedStatement statement, int i, Object value) throws SQLException {
886        logger().debug("bind: value = {}", value);
887        if (value == null) {
888            statement.setNull(i, Types.VARCHAR);
889        } else if (value instanceof String) {
890            String s = (String) value;
891            if ("$now".equals(s)) {
892                Timestamp t = new Timestamp(new java.util.Date().getTime());
893                logger().debug("setting $now to {}", t);
894                statement.setTimestamp(i, t, calendar);
895            } else if ("$job".equals(s)) {
896                logger().debug("setting $job to {}", context.job());
897                statement.setString(i, context.job());
898            } else if ("$count".equals(s)) {
899                logger().debug("setting $count to {}", context.getLastRowCount());
900                statement.setLong(i, context.getLastRowCount());
901            } else if ("$last.sql.start".equals(s)) {
902                if (context.getLastExecutionStartDate() == 0L) {
903                    Timestamp riverStarted = context.getRiverFlow() != null ?
904                            new Timestamp(context.getRiverFlow().getFeeder().getRiverState().getStarted().getTime()) : null;
905                    context.setLastExecutionStartDate(riverStarted != null ? riverStarted.getTime() : new java.util.Date().getTime());
906                }
907                statement.setTimestamp(i, new Timestamp(context.getLastExecutionStartDate()), calendar);
908            } else if ("$last.sql.end".equals(s)) {
909                if (context.getLastExecutionEndDate() == 0L) {
910                    Timestamp riverStarted = context.getRiverFlow() != null ?
911                            new Timestamp(context.getRiverFlow().getFeeder().getRiverState().getStarted().getTime()) : null;
912                    context.setLastExecutionEndDate(riverStarted != null ? riverStarted.getTime() : new java.util.Date().getTime());
913                }
914                statement.setTimestamp(i, new Timestamp(context.getLastExecutionEndDate()), calendar);
915            } else if ("$last.sql.sequence.start".equals(s)) {
916                if (context.getLastStartDate() == 0L) {
917                    Timestamp riverStarted = context.getRiverFlow() != null ?
918                            new Timestamp(context.getRiverFlow().getFeeder().getRiverState().getStarted().getTime()) : null;
919                    context.setLastStartDate(riverStarted != null ? riverStarted.getTime() : new java.util.Date().getTime());
920                }
921                statement.setTimestamp(i, new Timestamp(context.getLastStartDate()), calendar);
922            } else if ("$last.sql.sequence.end".equals(s)) {
923                if (context.getLastEndDate() == 0L) {
924                    Timestamp riverStarted = context.getRiverFlow() != null ?
925                            new Timestamp(context.getRiverFlow().getFeeder().getRiverState().getStarted().getTime()) : null;
926                    context.setLastEndDate(riverStarted != null ? riverStarted.getTime() : new java.util.Date().getTime());
927                }
928                statement.setTimestamp(i, new Timestamp(context.getLastEndDate()), calendar);
929            } else if ("$river.name".equals(s)) {
930                String name = context.getRiverName();
931                statement.setString(i, name);
932            } else if ("$river.state.timestamp".equals(s)) {
933                Timestamp timestamp = context.getRiverFlow() != null ?
934                        new Timestamp(context.getRiverFlow().getFeeder().getRiverState().getTimestamp().getTime()) : null;
935                statement.setTimestamp(i, timestamp, calendar);
936            } else if ("$river.state.started".equals(s)) {
937                Timestamp started = context.getRiverFlow() != null ?
938                        new Timestamp(context.getRiverFlow().getFeeder().getRiverState().getStarted().getTime()) : null;
939                statement.setTimestamp(i, started, calendar);
940            } else if ("$river.state.counter".equals(s)) {
941                Long counter = context.getRiverFlow() != null ?
942                        context.getRiverFlow().getFeeder().getRiverState().getCounter() : null;
943                if (counter != null) {
944                    statement.setLong(i, counter);
945                }
946            } else if (context.shouldPrepareDatabaseMetadata()) {
947                for (String k : context.getLastDatabaseMetadata().keySet()) {
948                    if (k.equals(s)) {
949                        statement.setObject(i, context.getLastDatabaseMetadata().get(k));
950                    }
951                }
952            } else if (context.shouldPrepareResultSetMetadata()) {
953                for (String k : context.getLastResultSetMetadata().keySet()) {
954                    if (k.equals(s)) {
955                        statement.setObject(i, context.getLastResultSetMetadata().get(k));
956                    }
957                }
958            } else {
959                Object rowValue = context.getLastRow().get(s);
960                if (rowValue != null) {
961                    statement.setObject(i, rowValue);
962                } else {
963                    statement.setString(i, (String) value);
964                }
965            }
966        } else if (value instanceof Integer) {
967            statement.setInt(i, (Integer) value);
968        } else if (value instanceof Long) {
969            statement.setLong(i, (Long) value);
970        } else if (value instanceof BigDecimal) {
971            statement.setBigDecimal(i, (BigDecimal) value);
972        } else if (value instanceof Date) {
973            statement.setDate(i, (Date) value);
974        } else if (value instanceof Timestamp) {
975            statement.setTimestamp(i, (Timestamp) value, calendar);
976        } else if (value instanceof Float) {
977            statement.setFloat(i, (Float) value);
978        } else if (value instanceof Double) {
979            statement.setDouble(i, (Double) value);
980        } else {
981            statement.setObject(i, value);
982        }
983    }

985    /**
986     * Parse of value of result set
987     *
988     * @param result the result set
989     * @param i      the offset in the result set
990     * @param type   the JDBC type
991     * @param locale the locale to use for parsing
992     * @return The parse value
993     * @throws SQLException when SQL execution gives an error
994     * @throws IOException  when input/output error occurs
995     */
996    @Override
997    public Object parseType(ResultSet result, Integer i, int type, Locale locale)
998            throws SQLException, IOException, ParseException {
999        if (logger().isTraceEnabled()) {
1000            logger().trace("{} {} {}", i, type, result.getString(i));
1001        }
1002        switch (type) {
1003            /**
1004             * The JDBC types CHAR, VARCHAR, and LONGVARCHAR are closely
1005             * related. CHAR represents a small, fixed-length character string,
1006             * VARCHAR represents a small, variable-length character string, and
1007             * LONGVARCHAR represents a large, variable-length character string.
1008             */
1009            case Types.CHAR:
1010            case Types.VARCHAR:
1011            case Types.LONGVARCHAR: {
1012                return result.getString(i);
1013            }
1014            case Types.NCHAR:
1015            case Types.NVARCHAR:
1016            case Types.LONGNVARCHAR: {
1017                return result.getNString(i);
1018            }
1019            /**
1020             * The JDBC types BINARY, VARBINARY, and LONGVARBINARY are closely
1021             * related. BINARY represents a small, fixed-length binary value,
1022             * VARBINARY represents a small, variable-length binary value, and
1023             * LONGVARBINARY represents a large, variable-length binary value
1024             */
1025            case Types.BINARY:
1026            case Types.VARBINARY:
1027            case Types.LONGVARBINARY: {
1028                return result.getBytes(i);
1029            }
1030            /**
1031             * The JDBC type ARRAY represents the SQL3 type ARRAY.
1032             *
1033             * An ARRAY value is mapped to an instance of the Array interface in
1034             * the Java programming language. If a driver follows the standard
1035             * implementation, an Array object logically points to an ARRAY
1036             * value on the server rather than containing the elements of the
1037             * ARRAY object, which can greatly increase efficiency. The Array
1038             * interface contains methods for materializing the elements of the
1039             * ARRAY object on the client in the form of either an array or a
1040             * ResultSet object.
1041             */
1042            case Types.ARRAY: {
1043                Array arr = result.getArray(i);
1044                return arr == null ? null : arr.getArray();
1045            }
1046            /**
1047             * The JDBC type BIGINT represents a 64-bit signed integer value
1048             * between -9223372036854775808 and 9223372036854775807.
1049             *
1050             * The corresponding SQL type BIGINT is a nonstandard extension to
1051             * SQL. In practice the SQL BIGINT type is not yet currently
1052             * implemented by any of the major databases, and we recommend that
1053             * its use be avoided in code that is intended to be portable.
1054             *
1055             * The recommended Java mapping for the BIGINT type is as a Java
1056             * long.
1057             */
1058            case Types.BIGINT: {
1059                Object o = result.getLong(i);
1060                return result.wasNull() ? null : o;
1061            }
1062            /**
1063             * The JDBC type BIT represents a single bit value that can be zero
1064             * or one.
1065             *
1066             * SQL-92 defines an SQL BIT type. However, unlike the JDBC BIT
1067             * type, this SQL-92 BIT type can be used as a parameterized type to
1068             * define a fixed-length binary string. Fortunately, SQL-92 also
1069             * permits the use of the simple non-parameterized BIT type to
1070             * represent a single binary digit, and this usage corresponds to
1071             * the JDBC BIT type. Unfortunately, the SQL-92 BIT type is only
1072             * required in "full" SQL-92 and is currently supported by only a
1073             * subset of the major databases. Portable code may therefore prefer
1074             * to use the JDBC SMALLINT type, which is widely supported.
1075             */
1076            case Types.BIT: {
1077                try {
1078                    Object o = result.getInt(i);
1079                    return result.wasNull() ? null : o;
1080                } catch (Exception e) {
1081                    // PSQLException: Bad value for type int : t
1082                    if (e.getMessage().startsWith("Bad value for type int")) {
1083                        return "t".equals(result.getString(i));
1084                    }
1085                    throw new IOException(e);
1086                }
1087            }
1088            /**
1089             * The JDBC type BOOLEAN, which is new in the JDBC 3.0 API, maps to
1090             * a boolean in the Java programming language. It provides a
1091             * representation of true and false, and therefore is a better match
1092             * than the JDBC type BIT, which is either 1 or 0.
1093             */
1094            case Types.BOOLEAN: {
1095                return result.getBoolean(i);
1096            }
1097            /**
1098             * The JDBC type BLOB represents an SQL3 BLOB (Binary Large Object).
1099             *
1100             * A JDBC BLOB value is mapped to an instance of the Blob interface
1101             * in the Java programming language. If a driver follows the
1102             * standard implementation, a Blob object logically points to the
1103             * BLOB value on the server rather than containing its binary data,
1104             * greatly improving efficiency. The Blob interface provides methods
1105             * for materializing the BLOB data on the client when that is
1106             * desired.
1107             */
1108            case Types.BLOB: {
1109                Blob blob = result.getBlob(i);
1110                if (blob != null) {
1111                    long n = blob.length();
1112                    if (n > Integer.MAX_VALUE) {
1113                        throw new IOException("can't process blob larger than Integer.MAX_VALUE");
1114                    }
1115                    byte[] tab = blob.getBytes(1, (int) n);
1116                    blob.free();
1117                    return tab;
1118                }
1119                break;
1120            }
1121            /**
1122             * The JDBC type CLOB represents the SQL3 type CLOB (Character Large
1123             * Object).
1124             *
1125             * A JDBC CLOB value is mapped to an instance of the Clob interface
1126             * in the Java programming language. If a driver follows the
1127             * standard implementation, a Clob object logically points to the
1128             * CLOB value on the server rather than containing its character
1129             * data, greatly improving efficiency. Two of the methods on the
1130             * Clob interface materialize the data of a CLOB object on the
1131             * client.
1132             */
1133            case Types.CLOB: {
1134                Clob clob = result.getClob(i);
1135                if (clob != null) {
1136                    long n = clob.length();
1137                    if (n > Integer.MAX_VALUE) {
1138                        throw new IOException("can't process clob larger than Integer.MAX_VALUE");
1139                    }
1140                    String str = clob.getSubString(1, (int) n);
1141                    clob.free();
1142                    return str;
1143                }
1144                break;
1145            }
1146            case Types.NCLOB: {
1147                NClob nclob = result.getNClob(i);
1148                if (nclob != null) {
1149                    long n = nclob.length();
1150                    if (n > Integer.MAX_VALUE) {
1151                        throw new IOException("can't process nclob larger than Integer.MAX_VALUE");
1152                    }
1153                    String str = nclob.getSubString(1, (int) n);
1154                    nclob.free();
1155                    return str;
1156                }
1157                break;
1158            }
1159            /**
1160             * The JDBC type DATALINK, new in the JDBC 3.0 API, is a column
1161             * value that references a file that is outside of a data source but
1162             * is managed by the data source. It maps to the Java type
1163             * java.net.URL and provides a way to manage external files. For
1164             * instance, if the data source is a DBMS, the concurrency controls
1165             * it enforces on its own data can be applied to the external file
1166             * as well.
1167             *
1168             * A DATALINK value is retrieved from a ResultSet object with the
1169             * ResultSet methods getURL or getObject. If the Java platform does
1170             * not support the type of URL returned by getURL or getObject, a
1171             * DATALINK value can be retrieved as a String object with the
1172             * method getString.
1173             *
1174             * java.net.URL values are stored in a database using the method
1175             * setURL. If the Java platform does not support the type of URL
1176             * being set, the method setString can be used instead.
1177             *
1178             *
1179             */
1180            case Types.DATALINK: {
1181                return result.getURL(i);
1182            }
1183            /**
1184             * The JDBC DATE type represents a date consisting of day, month,
1185             * and year. The corresponding SQL DATE type is defined in SQL-92,
1186             * but it is implemented by only a subset of the major databases.
1187             * Some databases offer alternative SQL types that support similar
1188             * semantics.
1189             */
1190            case Types.DATE: {
1191                try {
1192                    Date d = result.getDate(i, calendar);
1193                    return d != null ? formatDate(d.getTime()) : null;
1194                } catch (SQLException e) {
1195                    return null;
1196                }
1197            }
1198            case Types.TIME: {
1199                try {
1200                    Time t = result.getTime(i, calendar);
1201                    return t != null ? formatDate(t.getTime()) : null;
1202                } catch (SQLException e) {
1203                    return null;
1204                }
1205            }
1206            case Types.TIMESTAMP: {
1207                try {
1208                    Timestamp t = result.getTimestamp(i, calendar);
1209                    return t != null ? formatDate(t.getTime()) : null;
1210                } catch (SQLException e) {
1211                    // java.sql.SQLException: Cannot convert value '0000-00-00 00:00:00' from column ... to TIMESTAMP.
1212                    return null;
1213                }
1214            }
1215            /**
1216             * The JDBC types DECIMAL and NUMERIC are very similar. They both
1217             * represent fixed-precision decimal values.
1218             *
1219             * The corresponding SQL types DECIMAL and NUMERIC are defined in
1220             * SQL-92 and are very widely implemented. These SQL types take
1221             * precision and scale parameters. The precision is the total number
1222             * of decimal digits supported, and the scale is the number of
1223             * decimal digits after the decimal point. For most DBMSs, the scale
1224             * is less than or equal to the precision. So for example, the value
1225             * "12.345" has a precision of 5 and a scale of 3, and the value
1226             * ".11" has a precision of 2 and a scale of 2. JDBC requires that
1227             * all DECIMAL and NUMERIC types support both a precision and a
1228             * scale of at least 15.
1229             *
1230             * The sole distinction between DECIMAL and NUMERIC is that the
1231             * SQL-92 specification requires that NUMERIC types be represented
1232             * with exactly the specified precision, whereas for DECIMAL types,
1233             * it allows an implementation to add additional precision beyond
1234             * that specified when the type was created. Thus a column created
1235             * with type NUMERIC(12,4) will always be represented with exactly
1236             * 12 digits, whereas a column created with type DECIMAL(12,4) might
1237             * be represented by some larger number of digits.
1238             *
1239             * The recommended Java mapping for the DECIMAL and NUMERIC types is
1240             * java.math.BigDecimal. The java.math.BigDecimal type provides math
1241             * operations to allow BigDecimal types to be added, subtracted,
1242             * multiplied, and divided with other BigDecimal types, with integer
1243             * types, and with floating point types.
1244             *
1245             * The method recommended for retrieving DECIMAL and NUMERIC values
1246             * is ResultSet.getBigDecimal. JDBC also allows access to these SQL
1247             * types as simple Strings or arrays of char. Thus, Java programmers
1248             * can use getString to receive a DECIMAL or NUMERIC result.
1249             * However, this makes the common case where DECIMAL or NUMERIC are
1250             * used for currency values rather awkward, since it means that
1251             * application writers have to perform math on strings. It is also
1252             * possible to retrieve these SQL types as any of the Java numeric
1253             * types.
1254             */
1255            case Types.DECIMAL:
1256            case Types.NUMERIC: {
1257                BigDecimal bd = null;
1258                try {
1259                    // getBigDecimal() should get obsolete. Most seem to use getString/getObject anyway...
1260                    bd = result.getBigDecimal(i);
1261                } catch (NullPointerException e) {
1262                    // But is it true? JDBC NPE exists since 13 years?
1263                    // http://forums.codeguru.com/archive/index.php/t-32443.html
1264                    // Null values are driving us nuts in JDBC:
1265                    // http://stackoverflow.com/questions/2777214/when-accessing-resultsets-in-jdbc-is-there-an-elegant-way-to-distinguish-betwee
1266                }
1267                if (bd == null || result.wasNull()) {
1268                    return null;
1269                }
1270                if (context.getScale() >= 0) {
1271                    bd = bd.setScale(context.getScale(), context.getRounding());
1272                    try {
1273                        long l = bd.longValueExact();
1274                        if (Long.toString(l).equals(result.getString(i))) {
1275                            // convert to long if possible
1276                            return l;
1277                        } else {
1278                            // convert to double (with precision loss)
1279                            return bd.doubleValue();
1280                        }
1281                    } catch (ArithmeticException e) {
1282                        return bd.doubleValue();
1283                    }
1284                } else {
1285                    return bd.toPlainString();
1286                }
1287            }

1290             * point number that supports 15 digits of mantissa.
