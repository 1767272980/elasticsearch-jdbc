002    package org.xbib.elasticsearch.river.jdbc.strategy.column;
003    
004    import java.io.IOException;
005    import java.util.HashMap;
006    import java.util.Map;
007    
008    import org.elasticsearch.action.get.GetResponse;
009    import org.elasticsearch.client.Client;
010    import org.elasticsearch.client.Requests;
011    import org.elasticsearch.common.logging.ESLogger;
012    import org.elasticsearch.common.logging.ESLoggerFactory;
013    import org.elasticsearch.common.unit.TimeValue;
014    import org.elasticsearch.common.xcontent.XContentBuilder;
015    import org.elasticsearch.common.xcontent.XContentFactory;
016    import org.elasticsearch.indices.IndexMissingException;
017    
018    import org.xbib.elasticsearch.river.jdbc.JDBCRiver;
019    import org.xbib.elasticsearch.river.jdbc.strategy.simple.SimpleRiverFlow;
020    
021    /**
022     * River flow implementation for the 'column' strategy
023     *
024     * @author Piotr Åšliwa <piotr.sliwa@zineinc.com>
025     */
026    public class ColumnRiverFlow extends SimpleRiverFlow {
027    
028        private final ESLogger logger = ESLoggerFactory.getLogger(ColumnRiverFlow.class.getName());
029    
030        protected static final String LAST_RUN_TIME = "last_run_time";
031    
032        protected static final String CURRENT_RUN_STARTED_TIME = "current_run_started_time";
033    
034    
035        protected ESLogger logger() {
036            return logger;
037        }
038    
039        @Override
040        public String strategy() {
041            return "column";
042        }
043    
044        @Override
045        public void move() {
046            try {
047                TimeValue lastRunTime = readLastRunTimeFromCustomInfo();
048                TimeValue currentTime = new TimeValue(new java.util.Date().getTime());
049    
050                writeTimesToJdbcSettings(lastRunTime, currentTime);
051    
052                context.riverSource().fetch();
053    
054                writeCustomInfo(currentTime.millis());
055            } catch (Exception e) {
056                logger.error(e.getMessage(), e);
057                abort = true;
058            }
059        }
060    
061        private TimeValue readLastRunTimeFromCustomInfo() throws IOException {
062            try {
063                GetResponse response = client().prepareGet("_river", context.riverName(), ID_INFO_RIVER_INDEX).execute().actionGet();
064                if (response != null && response.isExists()) {
065                    Map jdbcState = (Map) response.getSourceAsMap().get("jdbc");
066    
067                    if (jdbcState != null) {
068                        Number lastRunTime = (Number) jdbcState.get(LAST_RUN_TIME);
069    
070                        if (lastRunTime != null) {
071                            return new TimeValue(lastRunTime.longValue());
072                        }
073                    } else {
074                        throw new IOException("can't retrieve previously persisted state from _river/" + context.riverName());
075                    }
076                }
077            } catch (IndexMissingException e) {
078                logger.warn("river state missing: _river/{}/{}", context.riverName(), ID_INFO_RIVER_INDEX);
079            }
080    
081            return null;
082        }
083    
084        private Client client() {
085            return context.riverMouth().client();
086        }
087    
088        private void writeCustomInfo(long lastRunAt) throws IOException {
089            XContentBuilder builder = XContentFactory.jsonBuilder();
090    
091            builder
092                    .startObject()
093                    .startObject("jdbc")
094                    .field(LAST_RUN_TIME, lastRunAt)
095                    .endObject()
096                    .endObject();
097    
098            client().prepareBulk()
099                    .add(Requests.indexRequest("_river")
100                            .type(context.riverName())
101                            .id(ID_INFO_RIVER_INDEX)
102                            .source(builder)
103                    )
104                    .execute()
105                    .actionGet();
106        }
107    
108        private void writeTimesToJdbcSettings(TimeValue lastRunTime, TimeValue currentTime) {
109            Map<String, Object> jdbcSettings = (Map<String, Object>) context.riverSettings().get(JDBCRiver.TYPE);
110    
111            if (jdbcSettings == null) {
112                jdbcSettings = new HashMap<String, Object>();
113                context.riverSettings().put(JDBCRiver.TYPE, jdbcSettings);
114            }
115    
116            jdbcSettings.put(LAST_RUN_TIME, lastRunTime);
117            jdbcSettings.put(CURRENT_RUN_STARTED_TIME, currentTime);
118        }
119    }
