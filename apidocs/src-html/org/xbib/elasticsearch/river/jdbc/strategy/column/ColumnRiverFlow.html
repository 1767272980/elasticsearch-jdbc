003package org.xbib.elasticsearch.river.jdbc.strategy.column;
004
005import org.elasticsearch.action.get.GetResponse;
006import org.elasticsearch.client.Client;
007import org.elasticsearch.client.Requests;
008import org.elasticsearch.common.logging.ESLogger;
009import org.elasticsearch.common.logging.ESLoggerFactory;
010import org.elasticsearch.common.unit.TimeValue;
011import org.elasticsearch.common.xcontent.XContentBuilder;
012import org.elasticsearch.common.xcontent.XContentFactory;
013import org.elasticsearch.indices.IndexMissingException;
014import org.xbib.elasticsearch.river.jdbc.JDBCRiver;
015import org.xbib.elasticsearch.river.jdbc.strategy.simple.SimpleRiverFlow;
016
017import java.io.IOException;
018import java.util.HashMap;
019import java.util.Map;
020
021/**
022 * River flow implementation for the 'column' strategy
023 *
024 * @author Piotr ?liwa <piotr.sliwa@zineinc.com>
025 */
026public class ColumnRiverFlow extends SimpleRiverFlow {
027
028    private final ESLogger logger = ESLoggerFactory.getLogger(ColumnRiverFlow.class.getSimpleName());
029
030    protected ESLogger logger() {
031        return logger;
032    }
033
034    protected static final String LAST_RUN_TIME = "last_run_time";
035
036    protected static final String CURRENT_RUN_STARTED_TIME = "current_run_started_time";
037
038    @Override
039    public String strategy() {
040        return "column";
041    }
042
043    @Override
044    public void move() {
045        try {
046            TimeValue lastRunTime = readLastRunTimeFromCustomInfo();
047            TimeValue currentTime = new TimeValue(new java.util.Date().getTime());
048
049            writeTimesToJdbcSettings(lastRunTime, currentTime);
050
051            context.riverSource().fetch();
052
053            writeCustomInfo(currentTime.millis());
054        } catch (Exception e) {
055            logger.error(e.getMessage(), e);
056            abort = true;
057        }
058    }
059
060    private TimeValue readLastRunTimeFromCustomInfo() throws IOException {
061        try {
062            GetResponse response = client().prepareGet(context.riverIndexName(), context.riverName(), ID_INFO_RIVER_INDEX).execute().actionGet();
063
064            if (response != null && response.isExists()) {
065                Map jdbcState = (Map) response.getSourceAsMap().get("jdbc");
066
067                if (jdbcState != null) {
068                    Number lastRunTime = (Number) jdbcState.get(LAST_RUN_TIME);
069
070                    if (lastRunTime != null) {
071                        return new TimeValue(lastRunTime.longValue());
072                    }
073                } else {
074                    throw new IOException("can't retrieve previously persisted state from " + context.riverIndexName() + "/" + context.riverName());
075                }
076            }
077        } catch (IndexMissingException e) {
078            logger.warn("river state missing: {}/{}/{}", context.riverIndexName(), context.riverName(), ID_INFO_RIVER_INDEX);
079        }
080
081        return null;
082    }
083
084    private Client client() {
085        return context.riverMouth().client();
086    }
087
088    private void writeCustomInfo(long lastRunAt) throws IOException {
089        XContentBuilder builder = XContentFactory.jsonBuilder();
090
091        builder
092                .startObject()
093                .startObject("jdbc")
094                .field(LAST_RUN_TIME, lastRunAt)
095                .endObject()
096                .endObject();
097
098        client().prepareBulk()
099                .add(Requests.indexRequest(context.riverIndexName())
100                        .type(context.riverName())
101                        .id(ID_INFO_RIVER_INDEX)
102                        .source(builder)
103                )
104                .execute()
105                .actionGet();
106    }
107
108    private void writeTimesToJdbcSettings(TimeValue lastRunTime, TimeValue currentTime) {
109        Map<String, Object> jdbcSettings = (Map<String, Object>) context.riverSettings().get(JDBCRiver.TYPE);
110
111        if (jdbcSettings == null) {
112            jdbcSettings = new HashMap<String, Object>();
113            context.riverSettings().put(JDBCRiver.TYPE, jdbcSettings);
114        }
115
116        jdbcSettings.put(LAST_RUN_TIME, lastRunTime);
117        jdbcSettings.put(CURRENT_RUN_STARTED_TIME, currentTime);
118    }
119}
