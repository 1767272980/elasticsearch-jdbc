069    @Override
070    public String strategy() {
071        return "simple";

074    private final BulkProcessor.Listener listener = new BulkProcessor.Listener() {
075
076        @Override
077        public void beforeBulk(long executionId, BulkRequest request) {
078            long l = outstandingBulkRequests.incrementAndGet();
079            logger().info("new bulk [{}] of [{} items], {} outstanding bulk requests",
080                    executionId, request.numberOfActions(), l);
081        }
082
083        @Override
084        public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
085            outstandingBulkRequests.decrementAndGet();
086            logger().info("bulk [{}] success [{} items] [{}ms]",
087                    executionId, response.getItems().length, response.getTookInMillis());
088
089        }
090
091        @Override
092        public void afterBulk(long executionId, BulkRequest request, Throwable failure) {
093            outstandingBulkRequests.decrementAndGet();
094            logger().error("bulk [" + executionId + "] error", failure);
095            error = true;
096        }
097    };

105    @Override
106    public SimpleRiverMouth client(Client client) {
107        this.client = client;
108        this.bulk = BulkProcessor.builder(client, listener)
109                .setBulkActions(maxBulkActions - 1) // yes, offset by one!
110                .setConcurrentRequests(maxConcurrentBulkRequests)
111                .setFlushInterval(flushInterval)
112                .build();
113        return this;
114    }

184    @Override
185    public void index(StructuredObject object, boolean create) throws IOException {
186        if (error) {
187            logger().error("error, not indexing");
188            return;
189        }
190        if (!started) {
191            started = true;
192            startup();
193        }
194        if (Strings.hasLength(object.index())) {
195            setIndex(object.index());
196        }
197        if (Strings.hasLength(object.type())) {
198            setType(object.type());
199        }
200        if (Strings.hasLength(object.id())) {
201            setId(object.id());
202        }
203        IndexRequest request = Requests.indexRequest(getIndex())
204                .type(getType())
205                .id(getId())
206                .source(object.build());
207        if (object.meta(StructuredObject.VERSION) != null) {
208            request.versionType(VersionType.EXTERNAL)
209                    .version(Long.parseLong(object.meta(StructuredObject.VERSION)));
210        }
211        if (object.meta(StructuredObject.ROUTING) != null) {
212            request.routing(object.meta(StructuredObject.ROUTING));
213        }
214        if (object.meta(StructuredObject.PARENT) != null) {
215            request.parent(object.meta(StructuredObject.PARENT));
216        }
217        if (object.meta(StructuredObject.TIMESTAMP) != null) {
218            request.timestamp(object.meta(StructuredObject.TIMESTAMP));
219        }
220        if (object.meta(StructuredObject.TTL) != null) {
221            request.ttl(Long.parseLong(object.meta(StructuredObject.TTL)));
222        }
223        bulk.add(request);
224    }
225
226    @Override
227    public void delete(StructuredObject object) {
228        if (error) {
229            logger().error("error, not indexing");
230            return;
231        }
232        if (!started) {
233            started = true;
234            startup();
235        }
236        if (Strings.hasLength(object.index())) {
237            setIndex(object.index());
238        }
239        if (Strings.hasLength(object.type())) {
240            setType(object.type());
241        }
242        if (Strings.hasLength(object.id())) {
243            setId(object.id());
244        }
245        if (getId() == null) {
246            return; // skip if no doc is specified to delete
247        }
248        DeleteRequest request = Requests.deleteRequest(getIndex()).type(getType()).id(getId());
249        if (object.meta(StructuredObject.ROUTING) != null) {
250            request.routing(object.meta(StructuredObject.ROUTING));
251        }
252        if (object.meta(StructuredObject.PARENT) != null) {
253            request.parent(object.meta(StructuredObject.PARENT));
254        }
255        if (object.meta(StructuredObject.VERSION) != null) {
256            request.versionType(VersionType.EXTERNAL)
257                    .version(Long.parseLong(object.meta(StructuredObject.VERSION)));
258        }
259        bulk.add(request);
260    }
261
262    @Override
263    public void flush() throws IOException {
264        try {
265            // we must wait for flush interval... not really cool
266            Thread.sleep(flushInterval.millis());
267        } catch (InterruptedException e) {
268            Thread.currentThread().interrupt();
269            throw new IOException("interrupted (maybe harmless)");
270        }
271    }
272
273    @Override
274    public void close() {
275        bulk.close();
276    }

278    private RiverMouth startup() {
279        try {
280            createIndexIfNotExists();
281        } catch (Exception e) {
282            if (ExceptionsHelper.unwrapCause(e) instanceof IndexAlreadyExistsException) {
283                logger().warn("index already exists");
284            } else {
285                logger().warn("failed to create index", e);
286                error = true;
287            }
288        }
289        return this;
290    }
291
292    private void createIndexIfNotExists() {
293        if (error) {
294            logger().error("error, not create index");
295            return;
296        }
297        if (client.admin().indices().prepareExists(index).execute().actionGet().isExists()) {
298            if (Strings.hasLength(settings)) {
299                client.admin().indices().prepareUpdateSettings(index).setSettings(settings).execute().actionGet();
300            }
301            if (Strings.hasLength(mapping)) {
302                client.admin().indices().preparePutMapping(index).setType(type).setSource(mapping).execute().actionGet();
303            }
304            return;
305        }
306        CreateIndexRequestBuilder builder = client.admin().indices().prepareCreate(index);
307        if (Strings.hasLength(settings)) {
308            builder.setSettings(settings);
309        }
310        builder.execute().actionGet();
311        if (Strings.hasLength(mapping)) {
312            client.admin().indices().preparePutMapping(index).setType(type).setSource(mapping).execute().actionGet();
313        }
314    }
315
316    @Override
317    public void waitForCluster() throws IOException {
318        waitForCluster(ClusterHealthStatus.YELLOW, TimeValue.timeValueSeconds(30));
319    }
320
321    public void waitForCluster(ClusterHealthStatus status, TimeValue timeout) throws IOException {
322        try {
323            logger().info("waiting for cluster state {}", status.name());
324            ClusterHealthResponse healthResponse =
325                    client.admin().cluster().prepareHealth().setWaitForStatus(status).setTimeout(timeout).execute().actionGet();
326            if (healthResponse.isTimedOut()) {
327                throw new IOException("cluster state is " + healthResponse.getStatus().name()
328                        + " and not " + status.name()
329                        + ", cowardly refusing to continue with operations");
330            } else {
331                logger().info("... cluster state ok");
332            }
333        } catch (ElasticsearchTimeoutException e) {
334            throw new IOException("timeout, cluster does not respond to health request, cowardly refusing to continue with operations");
335        }
336    }
