027/**
028 * A river mouth implementation for the 'simple' strategy.
029 * <p/>
030 * This mouth receives StructuredObjects in the
031 * create(), index(), or delete() methods and passes them to the bulk indexing client.
032 * <p/>
033 * Bulk indexing is implemented concurrently. Therefore, many JDBC rivers can pass their
034 * data through this river target to ElasticSearch, without having to take precaution
035 * of overwhelming the index.
036 * <p/>
037 * The default size of a bulk request is 100 documents, the maximum number of concurrent requests is 30.
038 *
039 * @author J?rg Prante <joergprante@gmail.com>
040 */
041public class SimpleRiverMouth implements RiverMouth {
042
043    private static final ESLogger logger = ESLoggerFactory.getLogger(SimpleRiverMouth.class.getSimpleName());
044
045    private static final AtomicInteger outstandingBulkRequests = new AtomicInteger(0);
046
047    protected RiverContext context;
048
049    protected String index;
050
051    protected String type;
052
053    protected String id;
054
055    protected Client client;
056
057    private int maxBulkActions = 100;
058
059    private int maxConcurrentBulkRequests = 30;
060
061    private boolean acknowledge = false;
062
063    private volatile boolean error;
064
065    private BulkProcessor bulk;
066
067    protected ESLogger logger() {
068        return logger;
069    }
070
071    private final BulkProcessor.Listener listener = new BulkProcessor.Listener() {
072
073        @Override
074        public void beforeBulk(long executionId, BulkRequest request) {
075            long l = outstandingBulkRequests.incrementAndGet();
076            logger().info("new bulk [{}] of [{} items], {} outstanding bulk requests",
077                    executionId, request.numberOfActions(), l);
078        }
079
080        @Override
081        public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
082            if (acknowledge()) {
083                try {
084                    context.riverSource().acknowledge(response);
085                } catch (IOException e) {
086                    logger().error("bulk [" + executionId + "] acknowledge error", e);
087                }
088            }
089            outstandingBulkRequests.decrementAndGet();
090            logger().info("bulk [{}] success [{} items] [{}ms]",
091                    executionId, response.getItems().length, response.getTookInMillis());
092
093        }
094
095        @Override
096        public void afterBulk(long executionId, BulkRequest request, Throwable failure) {
097            outstandingBulkRequests.decrementAndGet();
098            logger().error("bulk [" + executionId + "] error", failure);
099            error = true;
100        }
101    };

114    @Override
115    public SimpleRiverMouth client(Client client) {
116        this.client = client;
117        this.bulk = BulkProcessor.builder(client, listener)
118                .setBulkActions(maxBulkActions - 1)
119                .setConcurrentRequests(maxConcurrentBulkRequests)
120                .setFlushInterval(TimeValue.timeValueSeconds(1))
121                .build();
122
123        return this;
124    }

196    @Override
197    public void create(StructuredObject object) throws IOException {
198        index(object, true);
199    }

201    @Override
202    public void index(StructuredObject object) throws IOException {
203        index(object, false);
204    }
205
206    public void index(StructuredObject object, boolean create) throws IOException {
207        if (Strings.hasLength(object.index())) {
208            index(object.index());
209        }
210        if (Strings.hasLength(object.type())) {
211            type(object.type());
212        }
213        if (Strings.hasLength(object.id())) {
214            id(object.id());
215        }
216        IndexRequest request = Requests.indexRequest(index())
217                .type(type())
218                .id(id())
219                .source(object.build());
220        if (create) {
221            request.create(create);
222        }
223        if (object.meta(StructuredObject.VERSION) != null) {
224            request.versionType(VersionType.EXTERNAL)
225                    .version(Long.parseLong(object.meta(StructuredObject.VERSION)));
226        }
227        if (object.meta(StructuredObject.ROUTING) != null) {
228            request.routing(object.meta(StructuredObject.ROUTING));
229        }
230        if (object.meta(StructuredObject.PERCOLATE) != null) {
231            request.percolate(object.meta(StructuredObject.PERCOLATE));
232        }
233        if (object.meta(StructuredObject.PARENT) != null) {
234            request.parent(object.meta(StructuredObject.PARENT));
235        }
236        if (object.meta(StructuredObject.TIMESTAMP) != null) {
237            request.timestamp(object.meta(StructuredObject.TIMESTAMP));
238        }
239        if (object.meta(StructuredObject.TTL) != null) {
240            request.ttl(Long.parseLong(object.meta(StructuredObject.TTL)));
241        }
242        bulk.add(request);
243    }
244
245
246    @Override
247    public void delete(StructuredObject object) {
248        if (Strings.hasLength(object.index())) {
249            index(object.index());
250        }
251        if (Strings.hasLength(object.type())) {
252            type(object.type());
253        }
254        if (Strings.hasLength(object.id())) {
255            id(object.id());
256        }
257        if (id == null) {
258            return; // skip if no doc is specified to delete
259        }
260        DeleteRequest request = Requests.deleteRequest(index()).type(type()).id(id());
261        if (object.meta(StructuredObject.ROUTING) != null) {
262            request.routing(object.meta(StructuredObject.ROUTING));
263        }
264        if (object.meta(StructuredObject.PARENT) != null) {
265            request.parent(object.meta(StructuredObject.PARENT));
266        }
267        if (object.meta(StructuredObject.VERSION) != null) {
268            request.versionType(VersionType.EXTERNAL)
269                    .version(Long.parseLong(object.meta(StructuredObject.VERSION)));
270        }
271        bulk.add(request);
272    }

287    @Override
288    public void createIndexIfNotExists(String settings, String mapping) {
289
290        if (client.admin().indices().prepareExists(index).execute().actionGet().isExists()) {
291            if (Strings.hasLength(settings)) {
292                client.admin().indices().prepareUpdateSettings(index).setSettings(settings).execute().actionGet();
293            }
294            if (Strings.hasLength(mapping)) {
295                client.admin().indices().preparePutMapping(index).setType(type).setSource(mapping).execute().actionGet();
296            }
297            return;
298        }
299        CreateIndexRequestBuilder builder = client.admin().indices().prepareCreate(index);
300        if (Strings.hasLength(settings)) {
301            builder.setSettings(settings);
302        }
303        builder.execute().actionGet();
304        if (Strings.hasLength(mapping)) {
305            client.admin().indices().preparePutMapping(index).setType(type).setSource(mapping).execute().actionGet();
306        }
307    }

309    @Override
310    public void waitForCluster() throws IOException {
311        waitForCluster(ClusterHealthStatus.YELLOW, TimeValue.timeValueSeconds(30));
312    }
313
314    public void waitForCluster(ClusterHealthStatus status, TimeValue timeout) throws IOException {
315        try {
316            logger().info("waiting for cluster state {}", status.name());
317            ClusterHealthResponse healthResponse =
318                    client.admin().cluster().prepareHealth().setWaitForStatus(status).setTimeout(timeout).execute().actionGet();
319            if (healthResponse.isTimedOut()) {
320                throw new IOException("cluster state is " + healthResponse.getStatus().name()
321                        + " and not " + status.name()
322                        + ", cowardly refusing to continue with operations");
323            } else {
324                logger().info("... cluster state ok");
325            }
326        } catch (ElasticSearchTimeoutException e) {
327            throw new IOException("timeout, cluster does not respond to health request, cowardly refusing to continue with operations");
328        }
329    }
