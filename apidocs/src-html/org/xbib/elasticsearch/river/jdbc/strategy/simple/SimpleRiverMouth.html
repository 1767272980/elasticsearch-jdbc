046/**
047 * A river mouth implementation for the 'simple' strategy.
048 * <p/>
049 * This mouth receives StructuredObjects in the
050 * create(), index(), or delete() methods and passes them to the bulk indexing client.
051 * <p/>
052 * Bulk indexing is implemented concurrently. Therefore, many JDBC rivers can pass their
053 * data through this river target to ElasticSearch, without having to take precaution
054 * of overwhelming the index.
055 * <p/>
056 * The default size of a bulk request is 100 documents, the maximum number of concurrent requests is 30.
057 *
058 * @author JÃ¶rg Prante <joergprante@gmail.com>
059 */
060public class SimpleRiverMouth implements RiverMouth {
061
062    private static final ESLogger logger = ESLoggerFactory.getLogger(SimpleRiverMouth.class.getName());
063
064    private static final AtomicInteger outstandingBulkRequests = new AtomicInteger(0);
065    private static final ThreadFactory THREAD_FACTORY_HEALTH = EsExecutors.daemonThreadFactory("jdbc-river-health");
066
067    protected RiverContext context;
068    protected String index;
069    protected String type;
070    protected String id;
071    protected Client client;
072    private int maxBulkActions = 100;
073    private int maxConcurrentBulkRequests = 30;
074    private boolean versioning = false;
075    private boolean acknowledge = false;
076    private volatile boolean error;
077    private BulkProcessor bulk;
078    private HealthMonitorThread healthThread;
079
080    private final BulkProcessor.Listener listener = new BulkProcessor.Listener() {
081
082        @Override
083        public void beforeBulk(long executionId, BulkRequest request) {
084            long l = outstandingBulkRequests.incrementAndGet();
085            logger.info("new bulk [{}] of [{} items], {} outstanding bulk requests",
086                    executionId, request.numberOfActions(), l);
087        }
088
089        @Override
090        public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
091            if(acknowledge()){
092                        try {
093                                        context.riverSource().acknowledge(response);
094                                } catch (IOException e) {
095                                        logger.error("bulk ["+executionId+"] acknowledge error", e);
096                                }
097            }
098                outstandingBulkRequests.decrementAndGet();
099            logger.info("bulk [{}] success [{} items] [{}ms]",
100                    executionId, response.getItems().length, response.getTookInMillis());
101            
102        }
103
104        @Override
105        public void afterBulk(long executionId, BulkRequest request, Throwable failure) {
106            outstandingBulkRequests.decrementAndGet();
107            logger.error("bulk [" + executionId + "] error", failure);
108            error = true;
109        }
110    };

123    @Override
124    public SimpleRiverMouth client(Client client) {
125        this.client = client;
126        this.bulk = BulkProcessor.builder(client, listener)
127                .setBulkActions(maxBulkActions-1)
128                .setConcurrentRequests(maxConcurrentBulkRequests)
129                .setFlushInterval(TimeValue.timeValueSeconds(1))
130                .build();
131
132        //Monitor cluster health in separate thread
133        healthThread = new HealthMonitorThread(client);
134        THREAD_FACTORY_HEALTH.newThread(healthThread).start();
135
136        return this;
137    }

230    public void index(StructuredObject object, boolean create) throws IOException {
231        if (!checkStatus()) {
232            return;
233        }
234
235        if (Strings.hasLength(object.index())) {
236            index(object.index());
237        }
238        if (Strings.hasLength(object.type())) {
239            type(object.type());
240        }
241        if (Strings.hasLength(object.id())) {
242            id(object.id());
243        }
244        IndexRequest request = Requests.indexRequest(index())
245                .type(type())
246                .id(id())
247                .source(object.build());
248        if (create) {
249            request.create(create);
250        }
251        if (object.meta(StructuredObject.VERSION) != null && versioning) {
252            request.versionType(VersionType.EXTERNAL)
253                    .version(Long.parseLong(object.meta(StructuredObject.VERSION)));
254        }
255        if (object.meta(StructuredObject.ROUTING) != null) {
256            request.routing(object.meta(StructuredObject.ROUTING));
257        }
258        if (object.meta(StructuredObject.PERCOLATE) != null) {
259            request.percolate(object.meta(StructuredObject.PERCOLATE));
260        }
261        if (object.meta(StructuredObject.PARENT) != null) {
262            request.parent(object.meta(StructuredObject.PARENT));
263        }
264        if (object.meta(StructuredObject.TIMESTAMP) != null) {
265            request.timestamp(object.meta(StructuredObject.TIMESTAMP));
266        }
267        if (object.meta(StructuredObject.TTL) != null) {
268            request.ttl(Long.parseLong(object.meta(StructuredObject.TTL)));
269        }
270        bulk.add(request);
271    }
272
273
274    @Override
275    public void delete(StructuredObject object) {
276        if (!checkStatus()) {
277            return;
278        }
279
280        if (Strings.hasLength(object.index())) {
281            index(object.index());
282        }
283        if (Strings.hasLength(object.type())) {
284            type(object.type());
285        }
286        if (Strings.hasLength(object.id())) {
287            id(object.id());
288        }
289        if (id == null) {
290            return; // skip if no doc is specified to delete
291        }
292        DeleteRequest request = Requests.deleteRequest(index()).type(type()).id(id());
293        if (object.meta(StructuredObject.ROUTING) != null) {
294            request.routing(object.meta(StructuredObject.ROUTING));
295        }
296        if (object.meta(StructuredObject.PARENT) != null) {
297            request.parent(object.meta(StructuredObject.PARENT));
298        }
299        if (object.meta(StructuredObject.VERSION) != null && versioning) {
300            request.versionType(VersionType.EXTERNAL)
301                    .version(Long.parseLong(object.meta(StructuredObject.VERSION)));
302        }
303        bulk.add(request);
304    }

320    @Override
321    public void createIndexIfNotExists(String settings, String mapping) {
322        if (!checkStatus()) {
323            return;
324        }
325
326        if (client.admin().indices().prepareExists(index).execute().actionGet().isExists()) {
327            if (Strings.hasLength(settings)) {
328                client.admin().indices().prepareUpdateSettings(index).setSettings(settings).execute().actionGet();
329            }
330            if (Strings.hasLength(mapping)) {
331                client.admin().indices().preparePutMapping(index).setType(type).setSource(mapping).execute().actionGet();
332            }
333            return;
334        }
335        CreateIndexRequestBuilder builder = client.admin().indices().prepareCreate(index);
336        if (Strings.hasLength(settings)) {
337            builder.setSettings(settings);
338        }
339        builder.execute().actionGet();
340        if (Strings.hasLength(mapping)) {
341            client.admin().indices().preparePutMapping(index).setType(type).setSource(mapping).execute().actionGet();
342        }
343    }
344
345    /**
346     * Checks the cluster health and error flag.
347     * @return True if status is ok, false if action should be aborted.
348     */
349    private boolean checkStatus() {
350        while (!healthThread.isHealthy()) {
351            try {
352                Thread.sleep(5000);
353            } catch (InterruptedException e) {
354                logger.warn("Thread interrupted while waiting to load", e);
355                Thread.interrupted();
356                return false;
357            }
358        }
359        return !error;
360    }
