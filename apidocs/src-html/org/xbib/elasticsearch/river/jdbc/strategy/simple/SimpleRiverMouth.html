079    private final BulkProcessor.Listener listener = new BulkProcessor.Listener() {
080
081        @Override
082        public void beforeBulk(long executionId, BulkRequest request) {
083            long l = outstandingBulkRequests.incrementAndGet();
084            logger().info("new bulk [{}] of [{} items], {} outstanding bulk requests",
085                    executionId, request.numberOfActions(), l);
086        }
087
088        @Override
089        public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
090            outstandingBulkRequests.decrementAndGet();
091            logger().info("bulk [{}] success [{} items] [{}ms]",
092                    executionId, response.getItems().length, response.getTookInMillis());
093        }
094
095        @Override
096        public void afterBulk(long executionId, BulkRequest request, Throwable failure) {
097            outstandingBulkRequests.decrementAndGet();
098            logger().error("bulk [" + executionId + "] error", failure);
099            error = true;
100        }
101    };

195    @Override
196    public void index(IndexableObject object, boolean create) throws IOException {
197        if (error) {
198            logger().error("error, not indexing");
199            return;
200        }
201        if (!started) {
202            started = true;
203            startup();
204        }
205        if (Strings.hasLength(object.index())) {
206            setIndex(object.index());
207        }
208        if (Strings.hasLength(object.type())) {
209            setType(object.type());
210        }
211        if (Strings.hasLength(object.id())) {
212            setId(object.id());
213        }
214        IndexRequest request = Requests.indexRequest(getIndex())
215                .type(getType())
216                .id(getId())
217                .source(object.build());
218        if (object.meta(ControlKeys._version.name()) != null) {
219            request.versionType(VersionType.EXTERNAL)
220                    .version(Long.parseLong(object.meta(ControlKeys._version.name())));
221        }
222        if (object.meta(ControlKeys._routing.name()) != null) {
223            request.routing(object.meta(ControlKeys._routing.name()));
224        }
225        if (object.meta(ControlKeys._parent.name()) != null) {
226            request.parent(object.meta(ControlKeys._parent.name()));
227        }
228        if (object.meta(ControlKeys._timestamp.name()) != null) {
229            request.timestamp(object.meta(ControlKeys._timestamp.name()));
230        }
231        if (object.meta(ControlKeys._ttl.name()) != null) {
232            request.ttl(Long.parseLong(object.meta(ControlKeys._ttl.name())));
233        }
234        bulk.add(request);
235    }
236
237    @Override
238    public void delete(IndexableObject object) {
239        if (error) {
240            logger().error("error, not indexing");
241            return;
242        }
243        if (!started) {
244            started = true;
245            startup();
246        }
247        if (Strings.hasLength(object.index())) {
248            setIndex(object.index());
249        }
250        if (Strings.hasLength(object.type())) {
251            setType(object.type());
252        }
253        if (Strings.hasLength(object.id())) {
254            setId(object.id());
255        }
256        if (getId() == null) {
257            return; // skip if no doc is specified to delete
258        }
259        DeleteRequest request = Requests.deleteRequest(getIndex()).type(getType()).id(getId());
260        if (object.meta(ControlKeys._version.name()) != null) {
261            request.versionType(VersionType.EXTERNAL)
262                    .version(Long.parseLong(object.meta(ControlKeys._version.name())));
263        }
264        if (object.meta(ControlKeys._routing.name()) != null) {
265            request.routing(object.meta(ControlKeys._routing.name()));
266        }
267        if (object.meta(ControlKeys._parent.name()) != null) {
268            request.parent(object.meta(ControlKeys._parent.name()));
269        }
270        bulk.add(request);
271    }

273    @Override
274    public void flush() throws IOException {
275        try {
276            // we must wait for flush interval... not really cool
277            Thread.sleep(flushInterval.millis());
278        } catch (InterruptedException e) {
279            Thread.currentThread().interrupt();
280            throw new IOException("interrupted (maybe harmless)");
281        }
282    }

289    private RiverMouth startup() {
290        try {
291            //updateSettings();
292            //updateMapping();
293            createIndexIfNotExists();
294        } catch (Exception e) {
295            if (ExceptionsHelper.unwrapCause(e) instanceof IndexAlreadyExistsException) {
296                logger().warn("index {} already exists", index);
297            } else {
298                logger().warn("failed to create index", e);
299                error = true;
300            }
301        }
302        return this;
303    }
304
305    private void updateSettings() {
306        if (error) {
307            logger().error("error, not updating settings");
308            return;
309        }
310        if (settings != null) {
311            client.admin().indices().prepareUpdateSettings(index)
312                    .setSettings(settings)
313                    .execute().actionGet();
314        }
315    }
316
317    private void updateMapping() {
318        if (error) {
319            logger().error("error, not updating mapping");
320            return;
321        }
322        if (mapping != null) {
323            client.admin().indices().preparePutMapping(index)
324                    .setType(type)
325                    .setSource(mapping)
326                    .execute().actionGet();
327        }
328    }
329
330    private void createIndexIfNotExists() {
331        if (error) {
332            logger().error("error, not creating index");
333            return;
334        }
335        if (client.admin().indices().prepareExists(index).execute().actionGet().isExists()) {
336            logger().warn("index {} already exists, ignoring index creation", index);
337            return;
338        }
339        CreateIndexRequestBuilder createIndexRequestBuilder =
340                client.admin().indices().prepareCreate(index);
341        if (settings != null) {
342            createIndexRequestBuilder.setSettings(settings);
343        }
344        if (mapping != null) {
345            createIndexRequestBuilder.addMapping(type, mapping);
346        }
347        createIndexRequestBuilder.execute().actionGet();
348    }

350    @Override
351    public void waitForCluster() throws IOException {
352        waitForCluster(ClusterHealthStatus.YELLOW, TimeValue.timeValueSeconds(30));
353    }
354
355    public void waitForCluster(ClusterHealthStatus status, TimeValue timeout) throws IOException {
356        try {
357            logger().info("waiting for cluster state {}", status.name());
358            ClusterHealthResponse healthResponse =
359                    client.admin().cluster().prepareHealth().setWaitForStatus(status).setTimeout(timeout).execute().actionGet();
360            if (healthResponse.isTimedOut()) {
361                throw new IOException("cluster state is " + healthResponse.getStatus().name()
362                        + " and not " + status.name()
363                        + ", cowardly refusing to continue with operations");
364            } else {
365                logger().info("... cluster state ok");
366            }
367        } catch (ElasticsearchTimeoutException e) {
368            throw new IOException("timeout, cluster does not respond to health request, cowardly refusing to continue with operations");
369        }
370    }
