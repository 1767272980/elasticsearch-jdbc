002package org.xbib.elasticsearch.river.jdbc;
003
004import java.io.IOException;
005import java.util.Date;
006import java.util.Map;
007import java.util.concurrent.TimeUnit;
008
009import org.elasticsearch.action.get.GetResponse;
010import org.elasticsearch.client.Client;
011import org.elasticsearch.common.joda.DateMathParser;
012import org.elasticsearch.common.joda.Joda;
013import org.elasticsearch.common.logging.ESLogger;
014import org.elasticsearch.common.logging.ESLoggerFactory;
015import org.elasticsearch.common.xcontent.ToXContent;
016import org.elasticsearch.common.xcontent.XContentBuilder;
017import org.elasticsearch.common.xcontent.XContentFactory;
018import org.elasticsearch.common.xcontent.XContentParser;
019import org.elasticsearch.common.xcontent.XContentType;
020import org.elasticsearch.indices.IndexMissingException;
021
022import static org.elasticsearch.client.Requests.indexRequest;
023import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
024import static org.elasticsearch.common.xcontent.XContentParser.Token.END_OBJECT;
025import static org.elasticsearch.common.xcontent.XContentParser.Token.FIELD_NAME;
026import static org.elasticsearch.common.xcontent.XContentParser.Token.START_OBJECT;
027import static org.elasticsearch.common.xcontent.XContentParser.Token.VALUE_BOOLEAN;
028import static org.elasticsearch.common.xcontent.XContentParser.Token.VALUE_NULL;
029import static org.elasticsearch.common.xcontent.XContentParser.Token.VALUE_NUMBER;
030import static org.elasticsearch.common.xcontent.XContentParser.Token.VALUE_STRING;
031
032public class RiverState implements ToXContent {
033
034    private final ESLogger logger = ESLoggerFactory.getLogger(RiverState.class.getName());
035
036    private String name;
037
038    private Long started;
039
040    private Long timestamp;
041
042    private Long counter;
043
044    private boolean active;
045
046    private Map<String,Object> info;
047
048    private String index;
049
050    private String type;
051
052    private String id;
053
054    public RiverState setIndex(String index) {
055        this.index = index;
056        return this;
057    }
058
059    public RiverState setType(String type) {
060        this.type = type;
061        return this;
062    }
063
064    public RiverState setId(String id) {
065        this.id = id;
066        return this;
067    }
068
069    public RiverState name(String name) {
070        this.name = name;
071        return this;
072    }
073
074    public RiverState counter(Long counter) {
075        this.counter = counter;
076        return this;
077    }
078
079    public Long counter() {
080        return counter;
081    }
082
083    public RiverState started(Long started) {
084        this.started = started;
085        return this;
086    }
087
088    public RiverState timestamp(Long timestamp) {
089        this.timestamp = timestamp;
090        return this;
091    }
092
093    public RiverState active(boolean active) {
094        this.active = active;
095        return this;
096    }
097
098    public RiverState info(Map<String,Object> info) {
099        this.info = info;
100        return this;
101    }
102
103    public void save(Client client) throws IOException {
104        XContentBuilder builder = jsonBuilder();
105        builder = toXContent(builder, ToXContent.EMPTY_PARAMS);
106        logger.debug("save state={}",builder.string());
107        client.index(indexRequest().index(index).type(type).id(id)
108                .source(builder.string()))
109                .actionGet();
110    }
111
112    public void load(Client client) throws IOException {
113        GetResponse get = null;
114        try {
115            client.admin().indices().prepareRefresh(index).execute().actionGet();
116            get = client.prepareGet(index, type, id).execute().actionGet();
117        } catch (IndexMissingException e) {
118            logger.warn("river state missing: {}/{}/{}", index, type, id);
119        }
120        if (get != null && get.isExists()) {
121            logger.debug("load state={}", get.getSourceAsString());
122            XContentParser parser = XContentFactory.xContent(XContentType.JSON)
123                    .createParser(get.getSourceAsBytes());
124            fromXContent(parser);
125        } else {
126            counter = 0L;
127        }
128    }
129
130    public void fromXContent(XContentParser parser) throws IOException {
131        DateMathParser dateParser = new DateMathParser(Joda.forPattern("dateOptionalTime"), TimeUnit.MILLISECONDS);
132        String currentFieldName = null;
133        XContentParser.Token token;
134        while ((token = parser.nextToken()) != END_OBJECT) {
135            if (token == null) {
136                break;
137            } else if (token == FIELD_NAME) {
138                currentFieldName = parser.currentName();
139            } else if (token == VALUE_NULL || token == VALUE_STRING
140                    || token == VALUE_BOOLEAN || token == VALUE_NUMBER) {
141                if ("name".equals(currentFieldName)) {
142                    name(parser.text());
143                } else if ("started".equals(currentFieldName)) {
144                    started(dateParser.parse(parser.text(), 0));
145                } else if ("timestamp".equals(currentFieldName)) {
146                    timestamp(dateParser.parse(parser.text(), 0));
147                } else if ("counter".equals(currentFieldName)) {
148                    counter(parser.longValue());
149                } else if ("active".equals(currentFieldName)) {
150                    active(parser.booleanValue());
151                }
152            } else if (token == START_OBJECT) {
153                if ("info".equals(currentFieldName)) {
154                    info(parser.map());
155                } // else skip
156            }
157        }
158    }
159
160    @Override
161    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
162        builder.startObject()
163                .field("name", name)
164                .field("started", new Date(started))
165                .field("timestamp", new Date(timestamp))
166                .field("counter", counter)
167                .field("active", active)
168                .field("info", info)
169            .endObject();
170        return builder;
171    }
172}
