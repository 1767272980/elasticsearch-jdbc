005import org.xbib.elasticsearch.river.jdbc.RiverMouth;

007import java.io.IOException;

016/**
017 * The SimpleValueListener class consumes values from a database and transports
018 * them to the river target.
019 *
020 * @author J?rg Prante <joergprante@gmail.com>
021 */
022public class SimpleValueListener<O extends Object> implements ValueListener {
023
024    /**
025     * The current structured object
026     */
027    private StructuredObject current;
028    /**
029     * The object before the current object
030     */
031    private StructuredObject prev;
032    /**
033     * The river target where the structured objects will be moved
034     */
035    private RiverMouth target;
036    /**
037     * The keys of the values. They are examined for the Elasticsearch index
038     * attributes.
039     */
040    private List<String> keys;
041    /**
042     * The delimiter between the key names in the path, used for string split.
043     * Should not be modified.
044     */
045    private char delimiter = '.';
046    /**
047     * If digesting is enabled. This is the default.
048     */
049    private boolean digesting = true;
050
051    private MessageDigest digest;
052
053    public SimpleValueListener target(RiverMouth target) {
054        this.target = target;
055        return this;
056    }
057
058    public SimpleValueListener delimiter(char delimiter) {
059        this.delimiter = delimiter;
060        return this;
061    }
062
063    public SimpleValueListener digest(boolean digesting) {
064        this.digesting = digesting;
065        if (digesting) {
066            try {
067                digest = MessageDigest.getInstance("SHA-256");
068            } catch (Exception e) {
069                e.printStackTrace();
070            }
071        }
072        return this;
073    }
074
075    @Override
076    public MessageDigest digest() {
077        return digest;
078    }
079
080    /**
081     * Just some syntactic sugar.
082     *
083     * @return this value listener
084     * @throws IOException
085     */
086    public SimpleValueListener begin() throws IOException {
087        return this;
088    }
089
090    /**
091     * Set the keys.
092     *
093     * @param keys
094     * @return this value listener
095     */
096    @Override
097    public SimpleValueListener keys(List<String> keys) {
098        this.keys = keys;
099        return this;
100    }
101
102    /**
103     * Receive values.
104     *
105     * @param values
106     * @return this value listener
107     * @throws IOException
108     */
109    @Override
110    public SimpleValueListener values(List<? extends Object> values) throws IOException {
111        boolean hasSource = false;
112        if (current == null) {
113            current = newObject();
114        }
115        if (prev == null) {
116            prev = newObject();
117        }
118        // execute meta operations on pseudo columns
119        for (int i = 0; i < values.size(); i++) {
120            // v may be null
121            Object o = values.get(i);
122            if (o == null) {
123                continue;
124            }
125            String v = o.toString();
126            // JAVA7: string switch
127            String k = keys.get(i);
128            map(k, v, current);
129            if (StructuredObject.SOURCE.equals(k)) {
130                hasSource = true;
131            }
132        }
133        if (hasSource) {
134            end(current);
135            current = newObject();
136            return this;
137        }
138        // switch to next structured object if current is not equal to previous
139        if (!current.equals(prev) || current.isEmpty()) {
140            prev.source(current.source()); // "steal" source 
141            end(prev); // here, the element is being prepared for bulk indexing
142            prev = current;
143            current = newObject();
144        }
145        // create current object from values by sequentially merging the values
146        for (int i = 0; i < keys.size(); i++) {
147            Map map = null;
148            try {
149                map = JsonXContent.jsonXContent.createParser(values.get(i).toString()).mapAndClose();
150            } catch (Exception e) {
151            }
152
153            Map m = merge(current.source(), keys.get(i), map != null && map.size() > 0 ? map : values.get(i));
154            current.source(m);
155        }
156        return this;
157    }
158
159    protected void map(String k, String v, StructuredObject current) throws IOException {
160        if (StructuredObject.JOB.equals(k)) {
161            current.meta(StructuredObject.JOB, v);
162        } else if (StructuredObject.OPTYPE.equals(k)) {
163            current.optype(v);
164        } else if (StructuredObject.INDEX.equals(k)) {
165            current.index(v);
166        } else if (StructuredObject.TYPE.equals(k)) {
167            current.type(v);
168        } else if (StructuredObject.ID.equals(k)) {
169            current.id(v);
170        } else if (StructuredObject.VERSION.equals(k)) {
171            current.meta(StructuredObject.VERSION, v);
172        } else if (StructuredObject.ROUTING.equals(k)) {
173            current.meta(StructuredObject.ROUTING, v);
174        } else if (StructuredObject.PERCOLATE.equals(k)) {
175            current.meta(StructuredObject.PERCOLATE, v);
176        } else if (StructuredObject.PARENT.equals(k)) {
177            current.meta(StructuredObject.PARENT, v);
178        } else if (StructuredObject.TIMESTAMP.equals(k)) {
179            current.meta(StructuredObject.TIMESTAMP, v);
180        } else if (StructuredObject.TTL.equals(k)) {
181            current.meta(StructuredObject.TTL, v);
182        } else if (StructuredObject.SOURCE.equals(k)) {
183            current.source(JsonXContent.jsonXContent.createParser(v).mapAndClose());
184        }
185    }
186
187    /**
188     * End of values.
189     *
190     * @return this value listener
191     * @throws IOException
192     */
193    public SimpleValueListener end() throws IOException {
194        if (prev != null) {
195            prev.source(current.source());
196            end(prev);
197        }
198        prev = newObject();
199        current = newObject();
200        return this;
201    }
202
203    /**
204     * The object is complete. Push it to the river target.
205     *
206     * @param object the object
207     * @return this value listener
208     * @throws IOException
209     */
210    public SimpleValueListener end(StructuredObject object) throws IOException {
211        if (object.source().isEmpty()) {
212            return this;
213        }
214        if (target != null) {
215            if (object.optype() == null) {
216                target.index(object);
217            } else if (Operations.OP_INDEX.equals(object.optype())) {
218                target.index(object);
219            } else if (Operations.OP_CREATE.equals(object.optype())) {
220                target.create(object);
221            } else if (Operations.OP_DELETE.equals(object.optype())) {
222                target.delete(object);
223            } else {
224                throw new IllegalArgumentException("unknown optype: " + object.optype());
225            }
226        }
227        return this;
228    }
229
230    /**
231     * Reset this listener
232     *
233     * @throws IOException
234     */
235    @Override
236    public SimpleValueListener reset() throws IOException {
237        end();
238        return this;
239    }

241    /**
242     * Merge key/value pair to a map holding a JSON object. The key consists of
243     * a path pointing to the value position in the JSON object. The key,
244     * representing a path, is divided into head/tail. The recursion terminates
245     * if there is only a head and no tail. In this case, the value is added as
246     * a tuple to the map. If the head key exists, the merge process is
247     * continued by following the path represented by the key. If the path does
248     * not exist, a new map is created. A conflict arises if there is no map at
249     * a head key position. Then, the prefix given in the path is considered
250     * illegal.
251     *
252     * @param map   the map for the JSON object
253     * @param key   the key
254     * @param value the value
255     */
256    protected Map<String, Values<O>> merge(Map map, String key, Object value) {
257        if (PseudoColumnNames.INDEX.equals(key)
258                || PseudoColumnNames.ID.equals(key)
259                || PseudoColumnNames.TYPE.equals(key)
260                || PseudoColumnNames.PARENT.equals(key)) {
261            return map;
262        }
263        int i = key.indexOf(delimiter);
264        String index = null;
265        if (i <= 0) {
266            Matcher matcher = p.matcher(key);
267            boolean isSequence = matcher.matches();
268            String head = key;
269            if (isSequence) {
270                head = matcher.group(1);
271                index = matcher.group(2);
272            }
273            if (index == null || index.isEmpty()) {
274                map.put(head, new Values(map.get(head), value, isSequence));
275            } else {
276                if (!map.containsKey(head)) {
277                    map.put(head, new ArrayList());
278                }
279                Object o = map.get(head);
280                if (o instanceof List) {
281                    List l = (List) o;
282                    int j = l.isEmpty() ? -1 : l.size() - 1;
283                    if (j >= 0) {
284                        Map<String, Values<O>> m = (Map<String, Values<O>>) l.get(j);
285                        if (!m.containsKey(index)) {
286                            l.set(j, merge(m, index, value)); // append
287                        } else {
288                            l.add(merge(new HashMap(), index, value));
289                        }
290                    } else {
291                        l.add(merge(new HashMap(), index, value));
292                    }
293                }
294            }
295        } else {
296            String head = key.substring(0, i);
297            Matcher matcher = p.matcher(head);
298            boolean isSequence = matcher.matches();
299            if (isSequence) {
300                head = matcher.group(1);
301            }
302            String tail = key.substring(i + 1);
303            if (map.containsKey(head)) {
304                Object o = map.get(head);
305                if (o instanceof Map) {
306                    merge((Map<String, Values<O>>) o, tail, value);
307                } else {
308                    throw new IllegalArgumentException("illegal head: " + head);
309                }
310            } else {
311                Map<String, Values<O>> m = new HashMap<String, Values<O>>();
312                map.put(head, m);
313                merge(m, tail, value);
314            }
315        }
316        return map;
317    }
318
319    private final static Pattern p = Pattern.compile("^(.*)\\[(.*?)\\]$");
320
321    /**
322     * Create a new structured object
323     *
324     * @return a new structured object
325     */
326    private StructuredObject newObject() {
327        return digesting ?
328                new DigestStructuredObject().digest(digest) :
329                new PlainStructuredObject();
330    }
