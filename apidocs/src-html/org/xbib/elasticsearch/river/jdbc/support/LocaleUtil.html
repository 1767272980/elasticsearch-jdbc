002    package org.xbib.elasticsearch.river.jdbc.support;
003    
004    import org.elasticsearch.common.base.Strings;
005    
006    import java.util.Locale;
007    
008    public class LocaleUtil {
009    
010        public static Locale toLocale(String localeString) {
011            if (Strings.isNullOrEmpty(localeString)) {
012                return Locale.getDefault();
013            }
014            int separatorCountry = localeString.indexOf('_');
015            char separator;
016            if (separatorCountry >= 0) {
017                separator = '_';
018            } else {
019                separatorCountry = localeString.indexOf('-');
020                separator = '-';
021            }
022            String language;
023            String country;
024            String variant;
025            if (separatorCountry < 0) {
026                language = localeString;
027                country = "";
028                variant = "";
029            } else {
030                language = localeString.substring(0, separatorCountry);
031                int separatorVariant = localeString.indexOf(separator, separatorCountry + 1);
032                if (separatorVariant < 0) {
033                    country = localeString.substring(separatorCountry + 1);
034                    variant = "";
035                } else {
036                    country = localeString.substring(separatorCountry + 1, separatorVariant);
037                    variant = localeString.substring(separatorVariant + 1);
038                }
039            }
040            return new Locale(language, country, variant);
041        }
042    
043        public static String fromLocale(Locale locale) {
044            return locale.getLanguage() +
045                    (Strings.isNullOrEmpty(locale.getCountry()) ? "_" + locale.getCountry() : "");
046        }
047    }
