002package org.xbib.elasticsearch.river.jdbc.support;
003
004import org.elasticsearch.common.base.Objects;
005import org.elasticsearch.common.xcontent.XContentBuilder;
006
007import java.io.IOException;
008import java.security.MessageDigest;
009import java.util.HashMap;
010import java.util.List;
011import java.util.Map;
012
013import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
014
015public class PlainStructuredObject implements StructuredObject {
016
017    private Map<String, String> meta;
018
019    private Map<String, ? super Object> source;
020
021    public PlainStructuredObject() {
022        this.meta = new HashMap();
023        this.source = new HashMap();
024    }
025
026    public StructuredObject optype(String optype) {
027        meta.put(OPTYPE, optype);
028        return this;
029    }
030
031    public String optype() {
032        return meta.get(OPTYPE);
033    }
034
035    public StructuredObject index(String index) {
036        meta.put(INDEX, index);
037        return this;
038    }
039
040    public String index() {
041        return meta.get(INDEX);
042    }
043
044    public StructuredObject type(String type) {
045        meta.put(TYPE, type);
046        return this;
047    }
048
049    public String type() {
050        return meta.get(TYPE);
051    }
052
053    public StructuredObject id(String id) {
054        meta.put(ID, id);
055        return this;
056    }
057
058    public String id() {
059        return meta.get(ID);
060    }
061
062    public StructuredObject meta(String key, String value) {
063        meta.put(key, value);
064        return this;
065    }
066
067    public String meta(String key) {
068        return meta.get(key);
069    }
070
071    public StructuredObject source(Map<String, ? super Object> source) {
072        this.source = source;
073        return this;
074    }
075
076    public Map source() {
077        return source;
078    }
079
080    @Override
081    public boolean equals(Object o) {
082        if (o == null) {
083            return false;
084        }
085        if (!(o instanceof StructuredObject)) {
086            return false;
087        }
088        StructuredObject c = (StructuredObject) o;
089        return Objects.equal(optype(), c.optype()) &&
090                Objects.equal(index(), c.index()) &&
091                Objects.equal(type(), c.type()) &&
092                id() != null && id().equals(c.id());
093    }
094
095    @Override
096    public int hashCode() {
097        int hash = 3;
098        hash = 37 * hash + (optype() != null ? optype().hashCode() : 0);
099        hash = 37 * hash + (index() != null ? index().hashCode() : 0);
100        hash = 37 * hash + (type() != null ? type().hashCode() : 0);
101        hash = 37 * hash + (id() != null ? id().hashCode() : 0);
102        return hash;
103    }
104
105    @Override
106    public int compareTo(StructuredObject o) {
107        int i = 0;
108        if (o == null) {
109            return -1;
110        }
111        if (optype() != null && o.optype() != null) {
112            i = optype().compareTo(o.optype());
113        }
114        if (i != 0) {
115            return i;
116        }
117        if (index() != null && o.index() != null) {
118            i = index().compareTo(o.index());
119        }
120        if (i != 0) {
121            return i;
122        }
123        if (type() != null && o.type() != null) {
124            i = type().compareTo(o.type());
125        }
126        if (i != 0) {
127            return i;
128        }
129        if (id() != null && o.id() != null) {
130            i = id().compareTo(o.id());
131        }
132        return i;
133    }
134
135    /**
136     * Build JSON with the help of XContentBuilder.
137     *
138     * @throws java.io.IOException
139     */
140    public String build() throws IOException {
141        if (index() != null) {
142            checksum(index());
143        }
144        if (type() != null) {
145            checksum(type());
146        }
147        if (id() != null) {
148            checksum(id());
149        }
150        XContentBuilder builder = jsonBuilder();
151        build(builder, source);
152        return builder.string();
153    }
154
155    /**
156     * Recursive method to build XContent from a map of ValueSets
157     *
158     * @param builder the builder
159     * @param map     the map
160     * @throws IOException
161     */
162    protected void build(XContentBuilder builder, Map<String, ? super Object> map) throws IOException {
163        builder.startObject();
164        for (String k : map.keySet()) {
165            builder.field(k);
166            checksum(k);
167            Object o = map.get(k);
168            if (o instanceof Values) {
169                Values v = (Values) o;
170                v.build(builder);
171                for (Object vv : v.getValues()) {
172                    if (vv != null) {
173                        checksum(vv.toString());
174                    }
175                }
176            } else if (o instanceof Map) {
177                build(builder, (Map<String, ? super Object>) o);
178            } else if (o instanceof List) {
179                build(builder, (List) o);
180            } else {
181                try {
182                    builder.value(o);
183                } catch (Exception e) {
184                    throw new IOException("unknown object class:" + o.getClass().getName());
185                }
186            }
187        }
188        builder.endObject();
189    }
190
191    protected void build(XContentBuilder builder, List list) throws IOException {
192        builder.startArray();
193        for (Object o : list) {
194            if (o instanceof Values) {
195                Values v = (Values) o;
196                v.build(builder);
197                for (Object vv : v.getValues()) {
198                    if (vv != null) {
199                        checksum(vv.toString());
200                    }
201                }
202            } else if (o instanceof Map) {
203                build(builder, (Map<String, ? super Object>) o);
204            } else if (o instanceof List) {
205                build(builder, (List) o);
206            } else {
207                try {
208                    builder.value(o);
209                } catch (Exception e) {
210                    throw new IOException("unknown object class:" + o.getClass().getName());
211                }
212            }
213        }
214        builder.endArray();
215    }
216
217    public boolean isEmpty() {
218        return index() == null && type() == null && id() == null && source.isEmpty();
219    }
220
221    public void clear() {
222        this.meta = null;
223        this.source = null;
224    }
225
226    public StructuredObject digest(MessageDigest digest) {
227        return this;
228    }
229
230    public MessageDigest digest() {
231        return null;
232    }
233
234    public void checksum(String data) throws IOException {
235        // skip checksum
236    }
237
238    public String checksum() {
239        return null;
240    }
241
242    @Override
243    public String toString() {
244        return optype() + "/" + index() + "/" + type() + "/" + id() + " " + source;
245    }
