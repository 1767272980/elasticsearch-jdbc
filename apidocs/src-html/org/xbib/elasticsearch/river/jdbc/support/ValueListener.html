002package org.xbib.elasticsearch.river.jdbc.support;
003
004import java.io.IOException;
005import java.security.MessageDigest;
006import java.util.List;
007
008/**
009 * Value listener interface for receiving a stream of key/value attributes,
010 * like a number of rows of columns.
011 *
012 * @author JÃ¶rg Prante <joergprante@gmail.com>
013 */
014public interface ValueListener {
015
016    /**
017     * Set the key names for the values
018     *
019     * @param keys the keys
020     * @return this ValueListener
021     */
022    ValueListener keys(List<String> keys);
023
024    /**
025     * Receive values for the keys
026     *
027     * @param values the values
028     * @return this ValueListener
029     * @throws IOException
030     */
031    ValueListener values(List<? extends Object> values) throws IOException;
032
033    /**
034     * Reset the key/value configuration
035     *
036     * @return this value listener
037     * @throws IOException
038     */
039    ValueListener reset() throws IOException;
040
041    /**
042     * Compute digest value for these values.
043     *
044     * @return digest
045     */
046    MessageDigest digest();
047}
