002    package org.xbib.elasticsearch.river.jdbc.support;
003    
004    import java.util.ServiceLoader;
005    
006    import org.xbib.elasticsearch.river.jdbc.RiverFlow;
007    import org.xbib.elasticsearch.river.jdbc.RiverMouth;
008    import org.xbib.elasticsearch.river.jdbc.RiverSource;
009    import org.xbib.elasticsearch.river.jdbc.strategy.simple.SimpleRiverFlow;
010    import org.xbib.elasticsearch.river.jdbc.strategy.simple.SimpleRiverMouth;
011    import org.xbib.elasticsearch.river.jdbc.strategy.simple.SimpleRiverSource;
012    
013    /**
014     * The river service loader looks up sources
015     */
016    public class RiverServiceLoader {
017    
018        /**
019         * A river source is the origin, the data producing side
020         *
021         * @param strategy the strategy
022         * @return a river source, or the SimpleRiverSource
023         */
024        public static RiverSource findRiverSource(String strategy) {
025            ServiceLoader<RiverSource> sourceLoader = ServiceLoader.load(RiverSource.class);
026            for (RiverSource rs : sourceLoader) {
027                if (strategy.equals(rs.strategy())) {
028                    return rs;
029                }
030            }
031            return new SimpleRiverSource();
032        }
033    
034        /**
035         * A river mouth is the Elasticsearch side of the river, where the bulk processor lives
036         *
037         * @param strategy the strategy
038         * @return a river mouth, or the SimpleRiverMouth
039         */
040        public static RiverMouth findRiverMouth(String strategy) {
041            ServiceLoader<RiverMouth> riverMouthLoader = ServiceLoader.load(RiverMouth.class);
042            for (RiverMouth rt : riverMouthLoader) {
043                if (strategy.equals(rt.strategy())) {
044                    return rt;
045                }
046            }
047            return new SimpleRiverMouth();
048        }
049    
050        /**
051         * A river flow encapsulates the thread that moves the data from source to mouth
052         *
053         * @param strategy the strategy
054         * @return a river flow, or the SimpleRiverFlow
055         */
056        public static RiverFlow findRiverFlow(String strategy) {
057            ServiceLoader<RiverFlow> riverFlowServiceLoader = ServiceLoader.load(RiverFlow.class);
058            for (RiverFlow rc : riverFlowServiceLoader) {
059                if (strategy.equals(rc.strategy())) {
060                    return rc;
061                }
062            }
063            return new SimpleRiverFlow();
064        }
065    
066    }
