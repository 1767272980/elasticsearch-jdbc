002package org.xbib.elasticsearch.river.jdbc.support;
003
004import java.io.IOException;
005import java.util.ArrayList;
006import java.util.Collection;
007import java.util.HashMap;
008import java.util.LinkedList;
009import java.util.List;
010import java.util.Map;
011import java.util.regex.Matcher;
012import java.util.regex.Pattern;
013
014import org.elasticsearch.common.xcontent.json.JsonXContent;
015
016import org.xbib.elasticsearch.river.jdbc.RiverMouth;

018/**
019 * The SimpleValueListener class consumes values from a tabular source and transports
020 * them to the river indexer.
021 */
022public class StructuredObjectKeyValueStreamListener<O extends Object> implements KeyValueStreamListener {
023
024    /**
025     * The current structured object
026     */
027    private StructuredObject current;
028    /**
029     * The object before the current object
030     */
031    private StructuredObject prev;
032    /**
033     * The river target where the structured objects will be moved
034     */
035    private RiverMouth output;
036    /**
037     * The keys of the values. They are examined for the Elasticsearch index
038     * attributes.
039     */
040    private List<String> keys;
041
042    public StructuredObjectKeyValueStreamListener output(RiverMouth output) {
043        this.output = output;
044        return this;
045    }
046
047    /**
048     * Just some syntactic sugar.
049     *
050     * @return this value listener
051     * @throws IOException
052     */
053    @Override
054    public StructuredObjectKeyValueStreamListener begin() throws IOException {
055        return this;
056    }
057
058    /**
059     * Set the keys.
060     *
061     * @param keys the keys
062     * @return this value listener
063     */
064    @Override
065    public StructuredObjectKeyValueStreamListener keys(Collection<String> keys) throws IOException {
066        this.keys = new LinkedList<String>(keys);
067        return this;
068    }
069
070    /**
071     * Receive values.
072     *
073     * @param values the values
074     * @return this value listener
075     * @throws IOException
076     */
077    @Override
078    public StructuredObjectKeyValueStreamListener values(Collection<? extends Object> values) throws IOException {
079        boolean hasSource = false;
080        if (current == null) {
081            current = newObject();
082        }
083        if (prev == null) {
084            prev = newObject();
085        }
086        // execute meta operations on pseudo columns
087        int i = 0;
088        for (Object o : values) {
089            if (o == null) {
090                continue;
091            }
092            String v = o.toString();
093            String k = keys.get(i);
094            map(k, v, current);
095            if (StructuredObject.SOURCE.equals(k)) {
096                hasSource = true;
097            }
098            i++;
099        }
100        if (hasSource) {
101            end(current);
102            current = newObject();
103            return this;
104        }
105        // switch to next structured object if current is not equal to previous
106        if (!current.equals(prev) || current.isEmpty()) {
107            prev.source(current.source()); // "steal" source 
108            end(prev); // here, the element is being prepared for bulk indexing
109            prev = current;
110            current = newObject();
111        }
112        // create current object from values by sequentially merging the values
113        i = 0;
114        for (Object o : values) {
115            Map map = null;
116            try {
117                map = JsonXContent.jsonXContent.createParser(o.toString()).mapAndClose();
118            } catch (Exception e) {
119                // ignore
120            }
121            Map m = merge(current.source(),
122                    keys.get(i),
123                    map != null && map.size() > 0 ? map : o);
124            current.source(m);
125            i++;
126        }
127        return this;
128    }
129
130    /**
131     * End of value sequence
132     *
133     * @return this value listener
134     * @throws IOException
135     */
136    @Override
137    public StructuredObjectKeyValueStreamListener end() throws IOException {
138        if (prev != null) {
139            prev.source(current.source());
140            end(prev);
141        }
142        prev = newObject();
143        current = newObject();
144        return this;
145    }
146
147    protected void map(String k, String v, StructuredObject current) throws IOException {
148        if (StructuredObject.JOB.equals(k)) {
149            current.meta(StructuredObject.JOB, v);
150        } else if (StructuredObject.OPTYPE.equals(k)) {
151            current.optype(v);
152        } else if (StructuredObject.INDEX.equals(k)) {
153            current.index(v);
154        } else if (StructuredObject.TYPE.equals(k)) {
155            current.type(v);
156        } else if (StructuredObject.ID.equals(k)) {
157            current.id(v);
158        } else if (StructuredObject.VERSION.equals(k)) {
159            current.meta(StructuredObject.VERSION, v);
160        } else if (StructuredObject.ROUTING.equals(k)) {
161            current.meta(StructuredObject.ROUTING, v);
162        } else if (StructuredObject.PERCOLATE.equals(k)) {
163            current.meta(StructuredObject.PERCOLATE, v);
164        } else if (StructuredObject.PARENT.equals(k)) {
165            current.meta(StructuredObject.PARENT, v);
166        } else if (StructuredObject.TIMESTAMP.equals(k)) {
167            current.meta(StructuredObject.TIMESTAMP, v);
168        } else if (StructuredObject.TTL.equals(k)) {
169            current.meta(StructuredObject.TTL, v);
170        } else if (StructuredObject.SOURCE.equals(k)) {
171            current.source(JsonXContent.jsonXContent.createParser(v).mapAndClose());
172        }
173    }
174
175
176    /**
177     * The object is complete. Push it to the river target.
178     *
179     * @param object the object
180     * @return this value listener
181     * @throws IOException
182     */
183    public StructuredObjectKeyValueStreamListener end(StructuredObject object) throws IOException {
184        if (object.source().isEmpty()) {
185            return this;
186        }
187        if (output != null) {
188            if (object.optype() == null) {
189                output.index(object, false);
190            } else if (Operations.OP_INDEX.equals(object.optype())) {
191                output.index(object, false);
192            } else if (Operations.OP_CREATE.equals(object.optype())) {
193                output.index(object, true);
194            } else if (Operations.OP_DELETE.equals(object.optype())) {
195                output.delete(object);
196            } else {
197                throw new IllegalArgumentException("unknown optype: " + object.optype());
198            }
199        }
200        return this;
201    }
202
203    /**
204     * Merge key/value pair to a map holding a JSON object. The key consists of
205     * a path pointing to the value position in the JSON object. The key,
206     * representing a path, is divided into head/tail. The recursion terminates
207     * if there is only a head and no tail. In this case, the value is added as
208     * a tuple to the map. If the head key exists, the merge process is
209     * continued by following the path represented by the key. If the path does
210     * not exist, a new map is created. A conflict arises if there is no map at
211     * a head key position. Then, the prefix given in the path is considered
212     * illegal.
213     *
214     * @param map   the map for the JSON object
215     * @param key   the key
216     * @param value the value
217     */
218    protected Map<String, Values<O>> merge(Map map, String key, Object value) {
219        if (PseudoColumnNames.INDEX.equals(key)
220                || PseudoColumnNames.ID.equals(key)
221                || PseudoColumnNames.TYPE.equals(key)
222                || PseudoColumnNames.PARENT.equals(key)) {
223            return map;
224        }
225        int i = key.indexOf('.');
226        String index = null;
227        if (i <= 0) {
228            Matcher matcher = p.matcher(key);
229            boolean isSequence = matcher.matches();
230            String head = key;
231            if (isSequence) {
232                head = matcher.group(1);
233                index = matcher.group(2);
234            }
235            if (index == null || index.isEmpty()) {
236                map.put(head, new Values(map.get(head), value, isSequence));
237            } else {
238                if (!map.containsKey(head)) {
239                    map.put(head, new ArrayList());
240                }
241                Object o = map.get(head);
242                if (o instanceof List) {
243                    List l = (List) o;
244                    int j = l.isEmpty() ? -1 : l.size() - 1;
245                    if (j >= 0) {
246                        Map<String, Values<O>> m = (Map<String, Values<O>>) l.get(j);
247                        if (!m.containsKey(index)) {
248                            l.set(j, merge(m, index, value)); // append
249                        } else {
250                            l.add(merge(new HashMap(), index, value));
251                        }
252                    } else {
253                        l.add(merge(new HashMap(), index, value));
254                    }
255                }
256            }
257        } else {
258            String head = key.substring(0, i);
259            Matcher matcher = p.matcher(head);
260            boolean isSequence = matcher.matches();
261            if (isSequence) {
262                head = matcher.group(1);
263            }
264            String tail = key.substring(i + 1);
265            if (map.containsKey(head)) {
266                Object o = map.get(head);
267                if (o instanceof Map) {
268                    merge((Map<String, Values<O>>) o, tail, value);
269                } else {
270                    throw new IllegalArgumentException("illegal head: " + head);
271                }
272            } else {
273                Map<String, Values<O>> m = new HashMap<String, Values<O>>();
274                map.put(head, m);
275                merge(m, tail, value);
276            }
277        }
278        return map;
279    }
280
281    private final static Pattern p = Pattern.compile("^(.*)\\[(.*?)\\]$");
282
283    /**
284     * Create a new structured object
285     *
286     * @return a new structured object
287     */
288    protected StructuredObject newObject() {
289        return new PlainStructuredObject();
290    }
