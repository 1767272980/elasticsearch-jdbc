002    package org.xbib.elasticsearch.river.jdbc.support;
003    
004    import java.io.IOException;
005    
006    import org.xbib.elasticsearch.river.jdbc.RiverMouth;
007    import org.xbib.elasticsearch.gatherer.IndexableObject;
008    import org.xbib.elasticsearch.gatherer.PlainKeyValueStreamListener;
009    
010    /**
011     * This class consumes pairs from a key/value stream
012     * and transports them to the river mouth.
013     */
014    public class RiverKeyValueStreamListener extends PlainKeyValueStreamListener<String,String> {
015    
016        private RiverMouth output;
017    
018        public RiverKeyValueStreamListener output(RiverMouth output) {
019            this.output = output;
020            return this;
021        }
022    
023        /**
024         * The object is complete. Push it to the river mouth.
025         *
026         * @param object the object
027         * @return this value listener
028         * @throws IOException
029         */
030        public RiverKeyValueStreamListener end(IndexableObject object) throws IOException {
031            if (object.isEmpty()) {
032                return this;
033            }
034            if (output != null) {
035                if (object.optype() == null) {
036                    output.index(object, false);
037                } else if ("index".equals(object.optype())) {
038                    output.index(object, false);
039                } else if ("create".equals(object.optype())) {
040                    output.index(object, true);
041                } else if ("delete".equals(object.optype())) {
042                    output.delete(object);
043                } else {
044                    throw new IllegalArgumentException("unknown optype: " + object.optype());
045                }
046            }
047            return this;
048        }
049    
050    }
