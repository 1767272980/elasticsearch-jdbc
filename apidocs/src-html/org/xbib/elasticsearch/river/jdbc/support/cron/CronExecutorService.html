002    package org.xbib.elasticsearch.river.jdbc.support.cron;
003    
004    import java.util.concurrent.ExecutorService;
005    
006    /**
007     * Executor service that schedules a task for execution via a cron expression.
008     */
009    public interface CronExecutorService extends ExecutorService {
010        /**
011         * Schedules the specified task to execute according to the specified cron expression.
012         *
013         * @param task       the Runnable task to schedule
014         * @param expression a cron expression
015         */
016        void schedule(Runnable task, CronExpression expression);
017    }
