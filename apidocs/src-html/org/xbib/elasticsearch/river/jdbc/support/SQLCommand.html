002    package org.xbib.elasticsearch.river.jdbc.support;
003    
004    import java.util.Arrays;
005    import java.util.List;
006    import java.util.Map;
007    
008    import org.elasticsearch.common.xcontent.support.XContentMapValues;
009    
010    import static org.elasticsearch.common.collect.Lists.newLinkedList;
011    import static org.elasticsearch.common.collect.Maps.newHashMap;
012    
013    public class SQLCommand {
014    
015        private String sql;
016    
017        private List<? super Object> params = newLinkedList();
018    
019        private Map<String,Object> results = newHashMap();
020    
021        private boolean callable;
022    
023        public SQLCommand setSQL(String sql) {
024            this.sql = sql;
025            return this;
026        }
027    
028        public String getSQL() {
029            return sql;
030        }
031    
032        public SQLCommand setParameters(List<? super Object> params) {
033            this.params = params;
034            return this;
035        }
036    
037        public List<? super Object> getParameters() {
038            return params;
039        }
040    
041        public SQLCommand setCallable(boolean callable) {
042            this.callable = callable;
043            return this;
044        }
045    
046        public boolean isCallable() {
047            return callable;
048        }
049    
050        public boolean isQuery() {
051            int p1 = sql.toLowerCase().indexOf("select");
052            if (p1 < 0) {
053                return false;
054            }
055            int p2 = sql.toLowerCase().indexOf("update");
056            if (p2 < 0) {
057                return true;
058            }
059            int p3 = sql.toLowerCase().indexOf("insert");
060            return p3 < 0 || p1 < p2 && p1 < p3;
061        }
062    
063        public void setResults(Map<String,Object> results) {
064            this.results = results;
065        }
066    
067        public Map<String,Object> getResults() {
068            return results;
069        }
070    
071        public static List<SQLCommand> parse(Map<String,Object> settings) {
072            List<SQLCommand> sql = newLinkedList();
073            if (!XContentMapValues.isArray(settings.get("sql"))) {
074                settings.put("sql", Arrays.asList(settings.get("sql")));
075            }
076            List<? super Object> list = (List<? super Object>) settings.get("sql");
077            for (Object entry : list) {
078                SQLCommand command = new SQLCommand();
079                if (entry instanceof Map) {
080                    Map<String,Object> m = (Map<String,Object>)entry;
081                    if (m.containsKey("statement")) {
082                        command.setSQL((String)m.get("statement"));
083                    }
084                    if (m.containsKey("parameter")) {
085                        command.setParameters(XContentMapValues.extractRawValues("parameter", m));
086                    }
087                    if (m.containsKey("callable")) {
088                        command.setCallable(XContentMapValues.nodeBooleanValue(m.get("callable")));
089                    }
090                    if (m.containsKey("register")) {
091                        command.setResults(XContentMapValues.nodeMapValue(m.get("register"), null));
092                    }
093                } else if (entry instanceof String) {
094                    command.setSQL((String)entry);
095                }
096                sql.add(command);
097            }
098            return sql;
099        }
100    
101        public String toString() {
102            StringBuilder sb = new StringBuilder();
103            sb.append("statement=")
104                    .append(sql)
105                    .append(" parameter=")
106                    .append(params)
107                    .append(" callable=")
108                    .append(callable);
109            return sb.toString();
110        }
111    
112    }
