022/**
023 * The names of the column names with a special meaning for the river execution.
024 * <p/>
025 * Mostly, they map to the Elasticsearch bulk item. The _job column denotes
026 * an ID for the event of a fetch execution.
027 *
028 * @author JÃ¶rg Prante <joergprante@gmail.com>
029 */
030public interface PseudoColumnNames {
031
032    // bulk job name
033    String JOB = "_job";
034
035    // bulk operation type
036    String OPTYPE = "_optype";
037
038    // bulk document
039    String INDEX = "_index";
040    String TYPE = "_type";
041    String ID = "_id";
042
043    // bulk parameters
044    String VERSION = "_version";
045    String ROUTING = "_routing";
046    String PERCOLATE = "_percolate";
047    String PARENT = "_parent";
048    String TIMESTAMP = "_timestamp";
049    String TTL = "_ttl";
050
051    // JSON
052    String SOURCE = "_source";
053}
