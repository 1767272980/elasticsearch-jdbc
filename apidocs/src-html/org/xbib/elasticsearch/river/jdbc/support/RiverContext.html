004import java.io.IOException;
005import java.util.List;
006import java.util.Locale;

008
009import org.elasticsearch.common.collect.Tuple;

011
012import org.elasticsearch.common.xcontent.XContentBuilder;
013import org.elasticsearch.common.xcontent.XContentHelper;

018import static org.elasticsearch.common.collect.Maps.newHashMap;

021/**
022 * The river context consists of the parameters that span source and mouth settings.
023 * It represents the river state, for supporting the river task execution, and river scripting.
024 */
025public class RiverContext {
026
027    /**
028     * The name of the river.
029     */
030    private String name;
031    /**
032     * The settings of the river
033     */
034    private Map<String, Object> settings;
035    /**
036     * The source of the river
037     */
038    private RiverSource source;
039    /**
040     * The target of the river
041     */
042    private RiverMouth mouth;
043
044    private String schedule;
045
046    private Integer poolsize;
047    /**
048     * The polling interval
049     */
050    private TimeValue interval;
051    /**
052     * The job name of the current river task
053     */
054    private String job;
055    /**
056     * The SQL commands
057     */
058    private List<SQLCommand> sql;
059
060    /**
061     * Autocomit enabled or not
062     */
063    private boolean autocommit;
064    /**
065     * The fetch size
066     */
067    private int fetchSize;
068    /**
069     * The maximum numbe rof rows per statement execution
070     */
071    private int maxRows;
072    /**
073     * The number of retries
074     */
075    private int retries;
076    /**
077     * The time to wait between retries
078     */
079    private TimeValue maxretrywait;
080
081    /**
082     * The locale for numerical format
083     */
084    private Locale locale;
085
086    /**
087     * Column name that contains creation time (for column strategy)
088     */
089    private String columnCreatedAt;
090
091    /**
092     * Column name that contains last update time (for column strategy)
093     */
094    private String columnUpdatedAt;
095
096    /**
097     * Column name that contains deletion time (for column strategy)
098     */
099    private String columnDeletedAt;
100
101    /**
102     * Columns name should be automatically escaped by proper db quote mark or not (for column strategy)
103     */
104    private boolean columnEscape;
105
106    public RiverContext riverSettings(Map<String, Object> settings) {
107        this.settings = settings;
108        return this;
109    }
110
111    public Map<String, Object> riverSettings() {
112        return settings;
113    }
114
115    public RiverContext riverName(String name) {
116        this.name = name;
117        return this;
118    }
119
120    public String riverName() {
121        return name;
122    }
123
124    public RiverContext riverSource(RiverSource source) {
125        this.source = source;
126        return this;
127    }
128
129    public RiverSource riverSource() {
130        return source;
131    }
132
133    public RiverContext riverMouth(RiverMouth mouth) {
134        this.mouth = mouth;
135        return this;
136    }
137
138    public RiverMouth riverMouth() {
139        return mouth;
140    }
141
142    public RiverContext locale(String languageTag) {
143        this.locale = LocaleUtil.toLocale(languageTag);
144        Locale.setDefault(locale); // for JDBC drivers internals
145        return this;
146    }
147
148    public Locale locale() {
149        return locale;
150    }
151
152    public RiverContext job(String job) {
153        this.job = job;
154        return this;
155    }
156
157    public String job() {
158        return job;
159    }
160
161    public RiverContext setInterval(TimeValue interval) {
162        this.interval = interval;
163        return this;
164    }
165
166    public TimeValue getInterval() {
167        return interval;
168    }
169
170    public RiverContext setSchedule(String schedule) {
171        this.schedule = schedule;
172        return this;
173    }
174
175    public String getSchedule() {
176        return schedule;
177    }
178
179    public RiverContext setPoolSize(Integer poolsize) {
180        this.poolsize = poolsize;
181        return this;
182    }
183
184    public Integer getPoolSize() {
185        return poolsize;
186    }
187
188    public RiverContext setStatements(List<SQLCommand> sql) {
189        this.sql = sql;
190        return this;
191    }
192
193    public List<SQLCommand> getStatements() {
194        return sql;
195    }
196
197    public RiverContext setAutoCommit(boolean autocommit) {
198        this.autocommit = autocommit;
199        return this;
200    }
201
202    public boolean getAutoCommit() {
203        return autocommit;
204    }
205
206    public RiverContext setFetchSize(int fetchSize) {
207        this.fetchSize = fetchSize;
208        return this;
209    }
210
211    public int getFetchSize() {
212        return fetchSize;
213    }
214
215    public RiverContext setMaxRows(int maxRows) {
216        this.maxRows = maxRows;
217        return this;
218    }
219
220    public int getMaxRows() {
221        return maxRows;
222    }
223
224    public RiverContext setRetries(int retries) {
225        this.retries = retries;
226        return this;
227    }
228
229    public int getRetries() {
230        return retries;
231    }
232
233    public RiverContext setMaxRetryWait(TimeValue maxretrywait) {
234        this.maxretrywait = maxretrywait;
235        return this;
236    }
237
238    public TimeValue getMaxRetryWait() {
239        return maxretrywait;
240    }
241
242    public RiverContext columnUpdatedAt(String updatedAt) {
243        this.columnUpdatedAt = updatedAt;
244        return this;
245    }
246
247    public String columnUpdatedAt() {
248        return columnUpdatedAt;
249    }
250
251    public RiverContext columnCreatedAt(String createdAt) {
252        this.columnCreatedAt = createdAt;
253        return this;
254    }
255
256    public String columnCreatedAt() {
257        return columnCreatedAt;
258    }
259
260    public RiverContext columnDeletedAt(String deletedAt) {
261        this.columnDeletedAt = deletedAt;
262        return this;
263    }
264
265    public String columnDeletedAt() {
266        return columnDeletedAt;
267    }
268
269    public RiverContext columnEscape(boolean escape) {
270        this.columnEscape = escape;
271        return this;
272    }
273
274    public boolean columnEscape() {
275        return this.columnEscape;
276    }
277
278    public RiverContext contextualize() {
279        if (source != null) {
280            source.riverContext(this);
281        }
282        if (mouth != null) {
283            mouth.riverContext(this);
284        }
285        return this;
286    }
287
288    public Map<String,Object> asMap() {
289        try {
290           XContentBuilder builder = jsonBuilder();
291           builder.startObject()
292                .field("riverName", name)
293                .field("settings", settings)
294                .field("locale", LocaleUtil.fromLocale(locale))
295                .field("job", job)
296                .field("schedule", schedule)
297                .field("interval", interval)
298                .field("sql", sql)
299                .field("autocommit", autocommit)
300                .field("fetchsize", fetchSize)
301                .field("maxrows", maxRows)
302                .field("retries", retries)
303                .field("maxretrywait", maxretrywait)
304                .field("columnCreatedAt", columnCreatedAt)
305                .field("columnUpdatedAt", columnUpdatedAt)
306                .field("columnDeletedAt", columnDeletedAt)
307                .field("columnEscape", columnEscape)
308                .endObject();
309            Tuple<XContentType,Map<String,Object>> tuple =XContentHelper.convertToMap(builder.bytes(), true);
310            return tuple.v2();
311        } catch (IOException e) {
312            // does not happen
313            return newHashMap();
314        }
315    }
316}
