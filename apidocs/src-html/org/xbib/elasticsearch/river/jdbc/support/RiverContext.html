002package org.xbib.elasticsearch.river.jdbc.support;
003
004import org.elasticsearch.common.unit.TimeValue;
005import org.xbib.elasticsearch.river.jdbc.RiverMouth;
006import org.xbib.elasticsearch.river.jdbc.RiverSource;
007
008import java.util.List;
009import java.util.Locale;
010import java.util.Map;
011
012/**
013 * The river context consists of the parameters that span over source and target, and the source and target.
014 * It represents the river state, for supporting the river task, and river scripting.
015 *
016 * @author J?rg Prante <joergprante@gmail.com>
017 */
018public class RiverContext {
019
020    /**
021     * The name of the river.
022     */
023    private String name;
024    /**
025     * The river index name
026     */
027    private String riverIndexName;
028    /**
029     * The settings of the river
030     */
031    private Map<String, Object> settings;
032    /**
033     * The source of the river
034     */
035    private RiverSource source;
036    /**
037     * The target of the river
038     */
039    private RiverMouth mouth;
040    /**
041     * The polling interval
042     */
043    private TimeValue poll;
044
045    /**
046     * The job name of the current river task
047     */
048    private String job;
049    /**
050     * The SQL statement
051     */
052    private String sql;
053    /**
054     * Parameters for the SQL statement
055     */
056    private List<? extends Object> sqlparams;
057
058    /**
059     * Is this SQL statemen a CallableStatement?
060     */
061    private boolean callable;
062
063    /**
064     * The acknowledge SQL statement
065     */
066    private String acksql;
067    /**
068     * Parameters for the acknowledge SQL statement
069     */
070    private List<? extends Object> acksqlparams;
071    /**
072     * Autocomit enabled or not
073     */
074    private boolean autocommit;
075    /**
076     * The fetch size
077     */
078    private int fetchSize;
079    /**
080     * The maximum numbe rof rows per statement execution
081     */
082    private int maxRows;
083    /**
084     * The number of retries
085     */
086    private int retries;
087    /**
088     * The time to wait between retries
089     */
090    private TimeValue maxretrywait;
091
092    /**
093     * The locale for numerical format
094     */
095    private Locale locale;
096
097    /**
098     * If digesting should be used or not
099     */
100    private boolean digesting;
101
102    /**
103     * Column name that contains creation time (for column strategy)
104     */
105    private String columnCreatedAt;
106
107    /**
108     * Column name that contains last update time (for column strategy)
109     */
110    private String columnUpdatedAt;
111
112    /**
113     * Column name that contains deletion time (for column strategy)
114     */
115    private String columnDeletedAt;
116
117    /**
118     * Columns name should be automatically escaped by proper db quote mark or not (for column strategy)
119     */
120    private boolean columnEscape;
121
122    public RiverContext riverSettings(Map<String, Object> settings) {
123        this.settings = settings;
124        return this;
125    }
126
127    public Map<String, Object> riverSettings() {
128        return settings;
129    }
130
131    public RiverContext riverIndexName(String name) {
132        this.riverIndexName = name;
133        return this;
134    }
135
136    public String riverIndexName() {
137        return riverIndexName;
138    }
139
140    public RiverContext riverName(String name) {
141        this.name = name;
142        return this;
143    }
144
145    public String riverName() {
146        return name;
147    }
148
149    public RiverContext riverSource(RiverSource source) {
150        this.source = source;
151        return this;
152    }
153
154    public RiverSource riverSource() {
155        return source;
156    }
157
158    public RiverContext riverMouth(RiverMouth mouth) {
159        this.mouth = mouth;
160        return this;
161    }
162
163    public RiverMouth riverMouth() {
164        return mouth;
165    }
166
167    public RiverContext job(String job) {
168        this.job = job;
169        return this;
170    }
171
172    public String job() {
173        return job;
174    }
175
176    public RiverContext pollInterval(TimeValue poll) {
177        this.poll = poll;
178        return this;
179    }
180
181    public TimeValue pollingInterval() {
182        return poll;
183    }
184
185    public RiverContext pollStatement(String sql) {
186        this.sql = sql;
187        return this;
188    }
189
190    public String pollStatement() {
191        return sql;
192    }
193
194    public RiverContext pollStatementParams(List<? extends Object> params) {
195        this.sqlparams = params;
196        return this;
197    }
198
199    public List<? extends Object> pollStatementParams() {
200        return sqlparams;
201    }
202
203    public RiverContext callable(boolean enabled) {
204        this.callable = enabled;
205        return this;
206    }
207
208    public boolean callable() {
209        return callable;
210    }
211
212    public RiverContext pollAckStatement(String acksql) {
213        this.acksql = acksql;
214        return this;
215    }
216
217    public String pollAckStatement() {
218        return acksql;
219    }
220
221    public RiverContext pollAckStatementParams(List<? extends Object> params) {
222        this.acksqlparams = params;
223        return this;
224    }
225
226    public List<? extends Object> pollAckStatementParams() {
227        return acksqlparams;
228    }
229
230    public RiverContext autocommit(boolean enabled) {
231        this.autocommit = enabled;
232        return this;
233    }
234
235    public boolean autocommit() {
236        return autocommit;
237    }
238
239    public RiverContext fetchSize(int fetchSize) {
240        this.fetchSize = fetchSize;
241        return this;
242    }
243
244    public int fetchSize() {
245        return fetchSize;
246    }
247
248    public RiverContext maxRows(int maxRows) {
249        this.maxRows = maxRows;
250        return this;
251    }
252
253    public int maxRows() {
254        return maxRows;
255    }
256
257    public RiverContext columnUpdatedAt(String updatedAt) {
258        this.columnUpdatedAt = updatedAt;
259        return this;
260    }
261
262    public String columnUpdatedAt() {
263        return columnUpdatedAt;
264    }
265
266    public RiverContext columnCreatedAt(String createdAt) {
267        this.columnCreatedAt = createdAt;
268        return this;
269    }
270
271    public String columnCreatedAt() {
272        return columnCreatedAt;
273    }
274
275    public RiverContext columnDeletedAt(String deletedAt) {
276        this.columnDeletedAt = deletedAt;
277        return this;
278    }
279
280    public String columnDeletedAt() {
281        return columnDeletedAt;
282    }
283
284    public RiverContext columnEscape(boolean escape) {
285        this.columnEscape = escape;
286        return this;
287    }
288
289    public boolean columnEscape() {
290        return this.columnEscape;
291    }
292
293    public RiverContext retries(int retries) {
294        this.retries = retries;
295        return this;
296    }
297
298    public int retries() {
299        return retries;
300    }
301
302    public RiverContext maxRetryWait(TimeValue maxretrywait) {
303        this.maxretrywait = maxretrywait;
304        return this;
305    }
306
307    public TimeValue maxRetryWait() {
308        return maxretrywait;
309    }
310
311    public RiverContext locale(String languageTag) {
312        this.locale = LocaleUtil.toLocale(languageTag);
313        Locale.setDefault(locale); // for JDBC drivers internals
314        return this;
315    }
316
317    public Locale locale() {
318        return locale;
319    }
320
321    public RiverContext digesting(boolean digesting) {
322        this.digesting = digesting;
323        return this;
324    }
325
326    public boolean digesting() {
327        return digesting;
328    }
329
330    public RiverContext contextualize() {
331        if (source != null) {
332            source.riverContext(this);
333        }
334        if (mouth != null) {
335            mouth.riverContext(this);
336        }
337        return this;
338    }
339}
