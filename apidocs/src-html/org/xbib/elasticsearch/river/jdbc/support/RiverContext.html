022    /**
023     * The river context consists of the parameters that span source and mouth settings.
024     * It represents the river state, for supporting the river task execution, and river scripting.
025     */
026    public class RiverContext {
027    
028        /**
029         * The name of the river.
030         */
031        private String name;
032        /**
033         * The settings of the river
034         */
035        private Map<String, Object> settings;
036    
037        /**
038         * The flow of the river
039         */
040        private RiverFlow flow;
041        /**
042         * The source of the river
043         */
044        private RiverSource source;
045        /**
046         * The target of the river
047         */
048        private RiverMouth mouth;
049    
050        private String schedule;
051    
052        private Integer poolsize;
053        /**
054         * The polling interval
055         */
056        private TimeValue interval;
057        /**
058         * The job name of the current river task
059         */
060        private String job;
061        /**
062         * The SQL commands
063         */
064        private List<SQLCommand> sql;
065    
066        /**
067         * Autocomit enabled or not
068         */
069        private boolean autocommit;
070    
071        /**
072         * The fetch size
073         */
074        private int fetchSize;
075    
076        /**
077         * The maximum numbe rof rows per statement execution
078         */
079        private int maxRows;
080    
081        /**
082         * The number of retries
083         */
084        private int retries;
085    
086        /**
087         * The time to wait between retries
088         */
089        private TimeValue maxretrywait;
090    
091        /**
092         * The locale for numerical format
093         */
094        private Locale locale;
095    
096        /**
097         * Column name that contains creation time (for column strategy)
098         */
099        private String columnCreatedAt;
100    
101        /**
102         * Column name that contains last update time (for column strategy)
103         */
104        private String columnUpdatedAt;
105    
106        /**
107         * Column name that contains deletion time (for column strategy)
108         */
109        private String columnDeletedAt;
110    
111        /**
112         * Columns name should be automatically escaped by proper db quote mark or not (for column strategy)
113         */
114        private boolean columnEscape;
115    
116        public RiverContext riverSettings(Map<String, Object> settings) {
117            this.settings = settings;
118            return this;
119        }
120    
121        public Map<String, Object> riverSettings() {
122            return settings;
123        }
124    
125        public RiverContext riverName(String name) {
126            this.name = name;
127            return this;
128        }
129    
130        public String riverName() {
131            return name;
132        }
133    
134        public RiverContext riverFlow(RiverFlow flow) {
135            this.flow = flow;
136            return this;
137        }
138    
139        public RiverFlow riverFlow() {
140            return flow;
141        }
142    
143        public RiverContext riverSource(RiverSource source) {
144            this.source = source;
145            return this;
146        }
147    
148        public RiverSource riverSource() {
149            return source;
150        }
151    
152        public RiverContext riverMouth(RiverMouth mouth) {
153            this.mouth = mouth;
154            return this;
155        }
156    
157        public RiverMouth riverMouth() {
158            return mouth;
159        }
160    
161        public RiverContext locale(String languageTag) {
162            this.locale = LocaleUtil.toLocale(languageTag);
163            Locale.setDefault(locale); // for JDBC drivers internals
164            return this;
165        }
166    
167        public Locale locale() {
168            return locale;
169        }
170    
171        public RiverContext job(String job) {
172            this.job = job;
173            return this;
174        }
175    
176        public String job() {
177            return job;
178        }
179    
180        public RiverContext setInterval(TimeValue interval) {
181            this.interval = interval;
182            return this;
183        }
184    
185        public TimeValue getInterval() {
186            return interval;
187        }
188    
189        public RiverContext setSchedule(String schedule) {
190            this.schedule = schedule;
191            return this;
192        }
193    
194        public String getSchedule() {
195            return schedule;
196        }
197    
198        public RiverContext setPoolSize(Integer poolsize) {
199            this.poolsize = poolsize;
200            return this;
201        }
202    
203        public Integer getPoolSize() {
204            return poolsize;
205        }
206    
207        public RiverContext setStatements(List<SQLCommand> sql) {
208            this.sql = sql;
209            return this;
210        }
211    
212        public List<SQLCommand> getStatements() {
213            return sql;
214        }
215    
216        public RiverContext setAutoCommit(boolean autocommit) {
217            this.autocommit = autocommit;
218            return this;
219        }
220    
221        public boolean getAutoCommit() {
222            return autocommit;
223        }
224    
225        public RiverContext setFetchSize(int fetchSize) {
226            this.fetchSize = fetchSize;
227            return this;
228        }
229    
230        public int getFetchSize() {
231            return fetchSize;
232        }
233    
234        public RiverContext setMaxRows(int maxRows) {
235            this.maxRows = maxRows;
236            return this;
237        }
238    
239        public int getMaxRows() {
240            return maxRows;
241        }
242    
243        public RiverContext setRetries(int retries) {
244            this.retries = retries;
245            return this;
246        }
247    
248        public int getRetries() {
249            return retries;
250        }
251

253            this.maxretrywait = maxretrywait;
254            return this;
255        }
256    
257        public TimeValue getMaxRetryWait() {
258            return maxretrywait;
259        }
260    
261        public RiverContext columnUpdatedAt(String updatedAt) {
262            this.columnUpdatedAt = updatedAt;
263            return this;
264        }
265    
266        public String columnUpdatedAt() {
267            return columnUpdatedAt;
268        }
269    
270        public RiverContext columnCreatedAt(String createdAt) {
271            this.columnCreatedAt = createdAt;
272            return this;
273        }
274    
275        public String columnCreatedAt() {
276            return columnCreatedAt;
277        }
278    
279        public RiverContext columnDeletedAt(String deletedAt) {
280            this.columnDeletedAt = deletedAt;
281            return this;
282        }
283    
284        public String columnDeletedAt() {
285            return columnDeletedAt;
286        }
287    
288        public RiverContext columnEscape(boolean escape) {
289            this.columnEscape = escape;
290            return this;
291        }
292    
293        public boolean columnEscape() {
294            return this.columnEscape;
295        }
296    
297        public RiverContext contextualize() {
298            if (source != null) {
299                source.riverContext(this);
300            }
301            if (mouth != null) {
302                mouth.riverContext(this);
303            }
304            if (flow != null) {
305                flow.riverContext(this);
306            }
307            return this;
308        }
309    
310        public Map<String,Object> asMap() {
311            try {
312               XContentBuilder builder = jsonBuilder();
313               builder.startObject()
314                    .field("riverName", name)
315                    .field("settings", settings)
316                    .field("locale", LocaleUtil.fromLocale(locale))
317                    .field("job", job)
318                    .field("schedule", schedule)
319                    .field("interval", interval)
320                    .field("sql", sql)
321                    .field("autocommit", autocommit)
322                    .field("fetchsize", fetchSize)
323                    .field("maxrows", maxRows)
324                    .field("retries", retries)
325                    .field("maxretrywait", maxretrywait)
326                    .field("columnCreatedAt", columnCreatedAt)
327                    .field("columnUpdatedAt", columnUpdatedAt)
328                    .field("columnDeletedAt", columnDeletedAt)
329                    .field("columnEscape", columnEscape)
330                    .endObject();
331                Tuple<XContentType,Map<String,Object>> tuple =XContentHelper.convertToMap(builder.bytes(), true);
332                return tuple.v2();
333            } catch (IOException e) {
334                // does not happen
335                return newHashMap();
336            }
337        }
338    }
