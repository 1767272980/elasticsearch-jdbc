275    public RiverContext setRounding(String rounding) {
276        if ("ceiling".equalsIgnoreCase(rounding)) {
277            this.rounding = BigDecimal.ROUND_CEILING;
278        } else if ("down".equalsIgnoreCase(rounding)) {
279            this.rounding = BigDecimal.ROUND_DOWN;
280        } else if ("floor".equalsIgnoreCase(rounding)) {
281            this.rounding = BigDecimal.ROUND_FLOOR;
282        } else if ("halfdown".equalsIgnoreCase(rounding)) {
283            this.rounding = BigDecimal.ROUND_HALF_DOWN;
284        } else if ("halfeven".equalsIgnoreCase(rounding)) {
285            this.rounding = BigDecimal.ROUND_HALF_EVEN;
286        } else if ("halfup".equalsIgnoreCase(rounding)) {
287            this.rounding = BigDecimal.ROUND_HALF_UP;
288        } else if ("unnecessary".equalsIgnoreCase(rounding)) {
289            this.rounding = BigDecimal.ROUND_UNNECESSARY;
290        } else if ("up".equalsIgnoreCase(rounding)) {
291            this.rounding = BigDecimal.ROUND_UP;
292        }
293        return this;
294    }

358    public RiverContext contextualize() {
359        if (source != null) {
360            source.riverContext(this);
361        }
362        if (mouth != null) {
363            mouth.riverContext(this);
364        }
365        if (flow != null) {
366            flow.riverContext(this);
367        }
368        return this;
369    }

371    public Map<String,Object> asMap() {
372        try {
373           XContentBuilder builder = jsonBuilder();
374           builder.startObject()
375                .field("riverName", name)
376                .field("settings", settings)
377                .field("locale", LocaleUtil.fromLocale(locale))
378                .field("job", job)
379                .field("schedule", schedule)
380                .field("interval", interval)
381                .field("sql", sql)
382                .field("autocommit", autocommit)
383                .field("fetchsize", fetchSize)
384                .field("maxrows", maxRows)
385                .field("retries", retries)
386                .field("maxretrywait", maxretrywait)
387                .field("columnCreatedAt", columnCreatedAt)
388                .field("columnUpdatedAt", columnUpdatedAt)
389                .field("columnDeletedAt", columnDeletedAt)
390                .field("columnEscape", columnEscape)
391                .endObject();
392            Tuple<XContentType,Map<String,Object>> tuple =XContentHelper.convertToMap(builder.bytes(), true);
393            return tuple.v2();
394        } catch (IOException e) {
395            // does not happen
396            return newHashMap();
397        }
398    }
