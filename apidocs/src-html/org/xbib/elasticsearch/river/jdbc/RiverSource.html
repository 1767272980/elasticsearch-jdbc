001package org.xbib.elasticsearch.river.jdbc;
002
003import org.xbib.elasticsearch.plugin.jdbc.RiverContext;
004import org.xbib.keyvalue.KeyValueStreamListener;
005
006import java.io.IOException;
007import java.sql.CallableStatement;
008import java.sql.Connection;
009import java.sql.PreparedStatement;
010import java.sql.ResultSet;
011import java.sql.SQLException;
012import java.sql.Statement;
013import java.text.ParseException;
014import java.util.Date;
015import java.util.List;
016import java.util.Locale;
017import java.util.Map;
018import java.util.TimeZone;
019
020/**
021 * The river source models the data producing side
022 */
023public interface RiverSource {
024
025    /**
026     * The strategy this river source supports.
027     *
028     * @return the strategy as a string
029     */
030    String strategy();
031
032    /**
033     * Set the river context
034     *
035     * @param context the context
036     * @return this river source
037     */
038    RiverSource setRiverContext(RiverContext context);
039
040    /**
041     * Fetch a data portion from the database and pass it to the river task
042     * for firther processing.
043     *
044     * @throws SQLException when SQL execution gives an error
045     * @throws IOException when input/output error occurs
046     */
047    void fetch() throws SQLException, IOException;
048
049    /**
050     * Set the driver URL
051     *
052     * @param url the JDBC URL
053     * @return this river source
054     */
055    RiverSource setUrl(String url);
056
057    /**
058     * Set the user authentication
059     *
060     * @param user the user
061     * @return this river source
062     */
063    RiverSource setUser(String user);
064
065    /**
066     * Set the password authentication
067     *
068     * @param password the password
069     * @return this river source
070     */
071    RiverSource setPassword(String password);
072
073
074    /**
075     * Get a connection for reading data
076     *
077     * @return connection
078     * @throws SQLException when SQL execution gives an error
079     */
080    Connection getConnectionForReading() throws SQLException;
081
082    /**
083     * Get a connection for writing data
084     *
085     * @return connection
086     * @throws SQLException when SQL execution gives an error
087     */
088    Connection getConnectionForWriting() throws SQLException;
089
090    /**
091     * Prepare query statement
092     *
093     * @param sql SQL statement
094     * @return a prepared statement
095     * @throws SQLException when SQL execution gives an error
096     */
097    PreparedStatement prepareQuery(String sql) throws SQLException;
098
099    /**
100     * Prepare insert/update statement
101     *
102     * @param sql SQL statement
103     * @return a prepared statement
104     * @throws SQLException when SQL execution gives an error
105     */
106    PreparedStatement prepareUpdate(String sql) throws SQLException;
107
108    /**
109     * Bind query variables
110     *
111     * @param statement prepared statement
112     * @param values    values
113     * @return this river source
114     * @throws SQLException when SQL execution gives an error
115     */
116    RiverSource bind(PreparedStatement statement, List<Object> values) throws SQLException;
117
118    /**
119     * Register output variables for callable statement
120     *
121     * @param statement callable statement
122     * @param values    values
123     * @return this river source
124     * @throws SQLException when SQL execution gives an error
125     */
126    RiverSource register(CallableStatement statement, Map<String, Object> values) throws SQLException;
127
128    /**
129     * Execute query
130     *
131     * @param statement prepared statement
132     * @return the result set
133     * @throws SQLException when SQL execution gives an error
134     */
135    ResultSet executeQuery(PreparedStatement statement) throws SQLException;
136
137    /**
138     * Execute query without binding parameters
139     *
140     * @param statement the SQL statement
141     * @param sql       the SQL query
142     * @return the result set
143     * @throws SQLException when SQL execution gives an error
144     */
145    ResultSet executeQuery(Statement statement, String sql) throws SQLException;
146
147    /**
148     * Execute insert/update
149     *
150     * @param statement statement
151     * @return this river source
152     * @throws SQLException when SQL execution gives an error
153     */
154    RiverSource executeUpdate(PreparedStatement statement) throws SQLException;
155
156    RiverSource executeUpdate(Statement statement, String sql) throws SQLException;
157
158    void beforeRows(ResultSet results, KeyValueStreamListener listener) throws SQLException, IOException;
159
160    /**
161     * Action for the next row of the result set to be processed
162     *
163     * @param results  result
164     * @param listener listener
165     * @return true if next row exists
166     * @throws SQLException when SQL execution gives an error
167     * @throws IOException when input/output error occurs
168     */
169    boolean nextRow(ResultSet results, KeyValueStreamListener listener) throws SQLException, IOException;
170
171    void afterRows(ResultSet results, KeyValueStreamListener listener) throws SQLException, IOException;
172
173    /**
174     * Parse a value in a row column
175     *
176     * @param results result set
177     * @param num     position
178     * @param type    type
179     * @param locale  locale
180     * @return object
181     * @throws SQLException when SQL execution gives an error
182     * @throws IOException when input/output error occurs
183     * @throws ParseException if number format could not be parsed
184     */
185    Object parseType(ResultSet results, Integer num, int type, Locale locale) throws SQLException, IOException, ParseException;
186
187    /**
188     * Close result set
189     *
190     * @param result result set
191     * @return this river source
192     * @throws SQLException when SQL execution gives an error
193     */
194    RiverSource close(ResultSet result) throws SQLException;
195
196    /**
197     * Close statement
198     *
199     * @param statement statement
200     * @return this river source
201     * @throws SQLException when SQL execution gives an error
202     */
203    RiverSource close(Statement statement) throws SQLException;
204
205    /**
206     * Close reading from this river source
207     *
208     * @return this river source
209     */
210    RiverSource closeReading();
211
212    /**
213     * Close writing to this river source
214     *
215     * @return this river source
216     */
217    RiverSource closeWriting();
218
219    /**
220     * Set the timezone for JDBC setTimestamp() calls with calendar object.
221     * @param timeZone the time zone
222     * @return this river source
223     */
224    RiverSource setTimeZone(TimeZone timeZone);
225
226    /**
227     * Get the current timezone of this river source for the JDBC setTimestamp() call
228     * @return the time zone
229     */
230    TimeZone getTimeZone();
231
232}
