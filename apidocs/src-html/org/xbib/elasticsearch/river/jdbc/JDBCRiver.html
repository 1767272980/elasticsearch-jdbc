002    package org.xbib.elasticsearch.river.jdbc;
003    
004    import java.util.List;
005    import java.util.Locale;
006    import java.util.Map;
007    
008    import org.elasticsearch.client.Client;
009    import org.elasticsearch.common.inject.Inject;
010    import org.elasticsearch.common.logging.ESLogger;
011    import org.elasticsearch.common.logging.ESLoggerFactory;
012    import org.elasticsearch.common.unit.TimeValue;
013    import org.elasticsearch.common.util.concurrent.EsExecutors;
014    import org.elasticsearch.common.xcontent.support.XContentMapValues;
015    import org.elasticsearch.river.AbstractRiverComponent;
016    import org.elasticsearch.river.River;
017    import org.elasticsearch.river.RiverName;
018    import org.elasticsearch.river.RiverSettings;
019    
020    import org.xbib.elasticsearch.river.jdbc.support.LocaleUtil;
021    import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
022    import org.xbib.elasticsearch.river.jdbc.support.RiverServiceLoader;
023    import org.xbib.elasticsearch.river.jdbc.support.SQLCommand;
024    
025    import static org.elasticsearch.common.collect.Maps.newHashMap;
026    
027    /**
028     * The JDBC river
029     */
030    public class JDBCRiver extends AbstractRiverComponent implements River {
031    
032        private final ESLogger logger = ESLoggerFactory.getLogger(JDBCRiver.class.getName());
033    
034        public final static String NAME = "river-jdbc";
035    
036        public final static String TYPE = "jdbc";
037    
038        private final String strategy;
039    
040        private final String url;
041    
042        private final String indexName;
043    
044        private final String typeName;
045    
046        private final RiverSource riverSource;
047    
048        private final RiverMouth riverMouth;
049    
050        private final RiverFlow riverFlow;
051    
052        private final RiverContext riverContext;
053    
054        private volatile boolean closed;
055    
056        @Inject
057        public JDBCRiver(RiverName riverName,
058                         RiverSettings riverSettings,
059                         Client client) {
060            super(riverName, riverSettings);
061            logger.debug("JDBC river initializing...");
062    
063            Map<String, Object> mySettings = newHashMap();
064            if (riverSettings.settings().containsKey(TYPE)) {
065                mySettings = (Map<String, Object>) riverSettings.settings().get(TYPE);
066            }
067            strategy = XContentMapValues.nodeStringValue(mySettings.get("strategy"), "simple");
068            String schedule = mySettings.containsKey("schedule") ?
069                XContentMapValues.nodeStringValue(mySettings.get("schedule"), null) :
070                null;
071            Integer poolsize = XContentMapValues.nodeIntegerValue(mySettings.get("poolsize"), 1);
072            // disable interval
073            TimeValue interval = XContentMapValues.nodeTimeValue(mySettings.get("interval"), TimeValue.timeValueMinutes(-1));
074    
075            // JDBC
076            url = XContentMapValues.nodeStringValue(mySettings.get("url"), null);
077            String user = XContentMapValues.nodeStringValue(mySettings.get("user"), null);
078            String password = XContentMapValues.nodeStringValue(mySettings.get("password"), null);
079            List<SQLCommand> sql = SQLCommand.parse(mySettings);
080            String rounding = XContentMapValues.nodeStringValue(mySettings.get("rounding"), null);
081            int scale = XContentMapValues.nodeIntegerValue(mySettings.get("scale"), 2);
082            boolean autocommit = XContentMapValues.nodeBooleanValue(mySettings.get("autocommit"), Boolean.FALSE);
083    
084            int fetchsize = url.startsWith("jdbc:mysql") ? Integer.MIN_VALUE :
085                    XContentMapValues.nodeIntegerValue(mySettings.get("fetchsize"), 10);
086            int maxrows = XContentMapValues.nodeIntegerValue(mySettings.get("max_rows"), 0);
087            int maxretries = XContentMapValues.nodeIntegerValue(mySettings.get("max_retries"), 3);
088            TimeValue maxretrywait =
089                    XContentMapValues.nodeTimeValue(mySettings.get("max_retries_wait"), TimeValue.timeValueSeconds(30));
090            String locale = XContentMapValues.nodeStringValue(mySettings.get("locale"), LocaleUtil.fromLocale(Locale.getDefault()));
091    
092            // defaults for column strategy
093            String columnCreatedAt = XContentMapValues.nodeStringValue(mySettings.get("created_at"), "created_at");
094            String columnUpdatedAt = XContentMapValues.nodeStringValue(mySettings.get("updated_at"), "updated_at");
095            String columnDeletedAt = XContentMapValues.nodeStringValue(mySettings.get("deleted_at"), null);
096            boolean columnEscape = XContentMapValues.nodeBooleanValue(mySettings.get("column_escape"), true);
097    
098            // set up bulk indexer. If no _index or _type pseudo columns are set in SQL statements, this index is used.
099            indexName = XContentMapValues.nodeStringValue(mySettings.get("index"), TYPE);
100            typeName = XContentMapValues.nodeStringValue(mySettings.get("type"), TYPE);
101            int bulkSize = XContentMapValues.nodeIntegerValue(mySettings.get("bulk_size"), 100);
102            int maxBulkRequests = XContentMapValues.nodeIntegerValue(mySettings.get("max_bulk_requests"), 30);
103            // flush interval for bulk indexer
104            TimeValue flushInterval = XContentMapValues.nodeTimeValue(mySettings.get("bulk_flush_interval"),
105                    TimeValue.timeValueSeconds(5));
106            // get two maps from the river settings to improve index creation
107            Map<String,Object> indexSettings = mySettings.containsKey("index_settings") ?
108                    XContentMapValues.nodeMapValue(mySettings.get("index_settings"), null) : null;
109            Map<String,Object> typeMapping = mySettings.containsKey("type_mapping") ?
110                    XContentMapValues.nodeMapValue(mySettings.get("type_mapping"), null) : null;
111    
112            riverSource = RiverServiceLoader.findRiverSource(strategy);
113            logger.debug("found river source class {} for strategy {}", riverSource.getClass().getName(), strategy);
114    
115            riverMouth = RiverServiceLoader.findRiverMouth(strategy);
116            logger.debug("found river mouth class {} for strategy {}", riverMouth.getClass().getName(), strategy);
117    
118            riverFlow = RiverServiceLoader.findRiverFlow(strategy);
119            logger.debug("found river flow class {} for strategy {}", riverFlow.getClass().getName(), strategy);
120    
121            riverSource.url(url)
122                    .user(user)
123                    .password(password)
124                    .rounding(rounding)
125                    .precision(scale);
126    
127            riverMouth.setSettings(indexSettings)
128                    .setMapping(typeMapping)
129                    .setIndex(indexName)
130                    .setType(typeName)
131                    .setMaxBulkActions(bulkSize)
132                    .setMaxConcurrentBulkRequests(maxBulkRequests)
133                    .setFlushInterval(flushInterval)
134                    .client(client);
135    
136            riverContext = new RiverContext()
137                    .riverName(riverName.getName())
138                    .riverSettings(riverSettings.settings())
139                    .riverSource(riverSource)
140                    .riverMouth(riverMouth)
141                    .riverFlow(riverFlow)
142                    .locale(locale)
143                    .setSchedule(schedule)
144                    .setPoolSize(poolsize)
145                    .setInterval(interval)
146                    .setStatements(sql)
147                    .setAutoCommit(autocommit)
148                    .setMaxRows(maxrows)
149                    .setFetchSize(fetchsize)
150                    .setRetries(maxretries)
151                    .setMaxRetryWait(maxretrywait)
152                    .columnCreatedAt(columnCreatedAt)
153                    .columnUpdatedAt(columnUpdatedAt)
154                    .columnDeletedAt(columnDeletedAt)
155                    .columnEscape(columnEscape)
156                    .contextualize();
157    
158            logger.debug("JDBC river initialized, ready to start");
159    
160        }
161    
162        @Override
163        public void start() {
164            logger.info("starting JDBC river: URL [{}], strategy [{}], index/type [{}/{}]",
165                    url, strategy, indexName, typeName);
166            Thread thread = EsExecutors.daemonThreadFactory(settings.globalSettings(),
167                    "jdbc-river-[" + riverName.name() + '/' + strategy + ']')
168                    .newThread(riverFlow);
169            riverFlow.schedule(thread);
170        }
171    
172        @Override
173        public void close() {
174            if (closed) {
175                return;
176            }
177            closed = true;
178            logger.info("closing JDBC river: URL [{}], strategy [{}], index/type [{}/{}]",
179                    url, strategy, indexName, typeName);
180            if (riverFlow != null) {
181                riverFlow.abort();
182            }
183            if (riverSource != null) {
184                riverSource.closeReading();
185                riverSource.closeWriting();
186            }
187            if (riverMouth != null) {
188                riverMouth.close();
189            }
190        }
191    
192        /**
193         * Induce a river run once, but in a synchronous manner. Mainly used for tests.
194         */
195        public void once() {
196            if (riverFlow != null) {
197                riverFlow.move();
198            }
199        }
200    
201        /**
202         * Induce a river run once, but in an asynchronous manner.
203         */
204        public void induce() {
205            RiverFlow riverFlow = RiverServiceLoader.findRiverFlow(strategy);
206            // prepare task for run
207            riverFlow.riverContext(riverContext);
208            Thread thread = EsExecutors.daemonThreadFactory(settings.globalSettings(),
209                    "JDBC river: [" + riverName.name() + '/' + strategy + ')')
210                    .newThread(riverFlow);
211            riverFlow.once(thread);
212        }
213    
214        public RiverFlow riverFlow() {
215            return riverFlow;
216        }
217    
218    }
