001package org.xbib.elasticsearch.action.river.jdbc.state.delete;
002
003import org.elasticsearch.ElasticsearchException;
004import org.elasticsearch.action.ActionListener;
005import org.elasticsearch.action.support.master.TransportMasterNodeOperationAction;
006import org.elasticsearch.cluster.ClusterService;
007import org.elasticsearch.cluster.ClusterState;
008import org.elasticsearch.cluster.ack.ClusterStateUpdateResponse;
009import org.elasticsearch.common.inject.Inject;
010import org.elasticsearch.common.settings.Settings;
011import org.elasticsearch.threadpool.ThreadPool;
012import org.elasticsearch.transport.TransportService;
013import org.xbib.elasticsearch.action.river.jdbc.state.RiverStateService;
014
015public class TransportDeleteRiverStateAction extends TransportMasterNodeOperationAction<DeleteRiverStateRequest, DeleteRiverStateResponse> {
016
017    private final RiverStateService riverStateService;
018
019    @Inject
020    public TransportDeleteRiverStateAction(Settings settings, ThreadPool threadPool,
021                                           ClusterService clusterService, TransportService transportService,
022                                           RiverStateService riverStateService) {
023        super(settings, DeleteRiverStateAction.NAME, transportService, clusterService, threadPool);
024        this.riverStateService = riverStateService;
025    }
026
027    @Override
028    protected String executor() {
029        return ThreadPool.Names.SAME;
030    }
031
032    @Override
033    protected DeleteRiverStateRequest newRequest() {
034        return new DeleteRiverStateRequest();
035    }
036
037    @Override
038    protected DeleteRiverStateResponse newResponse() {
039        return new DeleteRiverStateResponse();
040    }
041
042    @Override
043    protected void masterOperation(DeleteRiverStateRequest request, ClusterState state, final ActionListener<DeleteRiverStateResponse> listener) throws ElasticsearchException {
044        riverStateService.unregisterRiver(new RiverStateService.UnregisterRiverStateRequest("delete_river_state[" + request.getRiverName() + "]", request.getRiverName())
045                .masterNodeTimeout(request.masterNodeTimeout())
046                .ackTimeout(request.ackTimeout()), new ActionListener<ClusterStateUpdateResponse>() {
047            @Override
048            public void onResponse(ClusterStateUpdateResponse clusterStateUpdateResponse) {
049                listener.onResponse(new DeleteRiverStateResponse(clusterStateUpdateResponse.isAcknowledged()));
050            }
051
052            @Override
053            public void onFailure(Throwable e) {
054                listener.onFailure(e);
055            }
056        });
057    }
058
059}
