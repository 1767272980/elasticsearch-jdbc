001package org.xbib.elasticsearch.action.river.jdbc.state.delete;
002
003import org.elasticsearch.action.ActionRequestValidationException;
004import org.elasticsearch.action.support.master.AcknowledgedRequest;
005import org.elasticsearch.common.io.stream.StreamInput;
006import org.elasticsearch.common.io.stream.StreamOutput;
007
008import java.io.IOException;
009
010import static org.elasticsearch.action.ValidateActions.addValidationError;
011
012public class DeleteRiverStateRequest extends AcknowledgedRequest<DeleteRiverStateRequest> {
013
014    private String riverName;
015
016    public DeleteRiverStateRequest setRiverName(String riverName) {
017        this.riverName = riverName;
018        return this;
019    }
020
021    public String getRiverName() {
022        return riverName;
023    }
024
025    @Override
026    public ActionRequestValidationException validate() {
027        ActionRequestValidationException validationException = null;
028        if (riverName == null) {
029            validationException = addValidationError("name is missing", null);
030        }
031        return validationException;
032    }
033
034    @Override
035    public void readFrom(StreamInput in) throws IOException {
036        super.readFrom(in);
037        readTimeout(in);
038        this.riverName = in.readString();
039    }
040
041    @Override
042    public void writeTo(StreamOutput out) throws IOException {
043        super.writeTo(out);
044        writeTimeout(out);
045        out.writeString(riverName);
046    }
047}
