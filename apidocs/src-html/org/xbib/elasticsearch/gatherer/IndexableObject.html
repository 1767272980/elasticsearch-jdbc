002    package org.xbib.elasticsearch.gatherer;
003    
004    import java.io.IOException;
005    import java.util.Map;
006    
007    /**
008     * A structured object is composed by an object data source together with
009     * meta data about the object.
010     */
011    public interface IndexableObject extends Comparable<IndexableObject> {
012    
013        /**
014         * Set the operation type, either "index", "create", or "delete"
015         * @param optype the operaion type
016         * @return this indexable object
017         */
018        IndexableObject optype(String optype);
019    
020        /**
021         * Get the operation type
022         * @return the operation type
023         */
024        String optype();
025    
026        /**
027         * Set the index
028         * @param index the index
029         * @return this object
030         */
031        IndexableObject index(String index);
032    
033        /**
034         * Get the index
035         * @return the index
036         */
037        String index();
038    
039        /**
040         * Set the type
041         * @param type the type
042         * @return this object
043         */
044        IndexableObject type(String type);
045    
046        /**
047         * Get the type
048         * @return the type
049         */
050        String type();
051    
052        /**
053         * Set the ID
054         * @param id the ID
055         * @return this object
056         */
057        IndexableObject id(String id);
058    
059        /**
060         * Get the ID
061         * @return the ID
062         */
063        String id();
064    
065        /**
066         * Set meta data of this indexable object
067         * @param key the meta data key
068         * @param value the meta data value
069         * @return this object
070         */
071        IndexableObject meta(String key, String value);
072    
073        String meta(String key);
074    
075        IndexableObject source(Map<String, Object> source);
076    
077        Map source();
078    
079        String build() throws IOException;
080    
081        boolean isEmpty();
082    
083    }
