002    package org.xbib.elasticsearch.gatherer;
003    
004    import org.elasticsearch.common.base.Objects;
005    import org.elasticsearch.common.xcontent.ToXContent;
006    import org.elasticsearch.common.xcontent.XContentBuilder;
007    
008    import java.io.IOException;
009    import java.util.List;
010    import java.util.Map;
011    
012    import static org.elasticsearch.common.collect.Maps.newHashMap;
013    import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
014    
015    /**
016     * A basic indexable object. The indexable object can store meta data and core data.
017     */
018    public class PlainIndexableObject implements IndexableObject, ToXContent {
019    
020        private Map<String, String> meta;
021    
022        private Map<String, Object> core;
023    
024        public PlainIndexableObject() {
025            this.meta = newHashMap();
026            this.core = newHashMap();
027        }
028    
029        public IndexableObject optype(String optype) {
030            meta.put(ControlKeys._optype.name(), optype);
031            return this;
032        }
033    
034        public String optype() {
035            return meta.get(ControlKeys._optype.name());
036        }
037    
038        public IndexableObject index(String index) {
039            meta.put(ControlKeys._index.name(), index);
040            return this;
041        }
042    
043        public String index() {
044            return meta.get(ControlKeys._index.name());
045        }
046    
047        public IndexableObject type(String type) {
048            meta.put(ControlKeys._type.name(), type);
049            return this;
050        }
051    
052        public String type() {
053            return meta.get(ControlKeys._type.name());
054        }
055    
056        public IndexableObject id(String id) {
057            meta.put(ControlKeys._id.name(), id);
058            return this;
059        }
060    
061        public String id() {
062            return meta.get(ControlKeys._id.name());
063        }
064    
065        public IndexableObject meta(String key, String value) {
066            meta.put(key, value);
067            return this;
068        }
069    
070        public String meta(String key) {
071            return meta.get(key);
072        }
073    
074        public IndexableObject source(Map<String, Object> source) {
075            this.core = source;
076            return this;
077        }
078    
079        public Map source() {
080            return core;
081        }
082    
083        @Override
084        public boolean equals(Object o) {
085            if (o == null) {
086                return false;
087            }
088            if (!(o instanceof IndexableObject)) {
089                return false;
090            }
091            IndexableObject c = (IndexableObject) o;
092            return Objects.equal(optype(), c.optype()) &&
093                    Objects.equal(index(), c.index()) &&
094                    Objects.equal(type(), c.type()) &&
095                    id() != null && id().equals(c.id());
096        }
097    
098        @Override
099        public int hashCode() {
100            int hash = 3;
101            hash = 37 * hash + (optype() != null ? optype().hashCode() : 0);
102            hash = 37 * hash + (index() != null ? index().hashCode() : 0);
103            hash = 37 * hash + (type() != null ? type().hashCode() : 0);
104            hash = 37 * hash + (id() != null ? id().hashCode() : 0);
105            return hash;
106        }
107    
108        @Override
109        public int compareTo(IndexableObject o) {
110            int i = 0;
111            if (o == null) {
112                return -1;
113            }
114            if (optype() != null) {
115                i = optype().compareTo(o.optype());
116            }
117            if (i != 0) {
118                return i;
119            }
120            if (index() != null) {
121                i = index().compareTo(o.index());
122            }
123            if (i != 0) {
124                return i;
125            }
126            if (type() != null) {
127                i = type().compareTo(o.type());
128            }
129            if (i != 0) {
130                return i;
131            }
132            if (id() != null) {
133                i = id().compareTo(o.id());
134            }
135            return i;
136        }
137    
138        /**
139         * Build a string that can be used for indexing.
140         *
141         * @throws java.io.IOException
142         */
143        public String build() throws IOException {
144            XContentBuilder builder = jsonBuilder();
145            toXContent(builder, ToXContent.EMPTY_PARAMS);
146            return builder.string();
147        }
148    
149        @Override
150        public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
151            toXContent(builder, params, core);
152            return builder;
153        }
154    
155        /**
156         * Recursive method to build XContent from a key/value map of Values
157         *
158         * @param builder the builder
159         * @param map     the map
160         * @throws java.io.IOException
161         */
162        protected XContentBuilder toXContent(XContentBuilder builder, Params params, Map<String, Object> map) throws IOException {
163            builder.startObject();
164            for (String k : map.keySet()) {
165                builder.field(k);
166                Object o = map.get(k);
167                if (o instanceof Values) {
168                    Values v = (Values) o;
169                    v.toXContent(builder, params);
170                } else if (o instanceof Map) {
171                    toXContent(builder, params, (Map<String, Object>) o);
172                } else if (o instanceof List) {
173                    toXContent(builder, params, (List) o);
174                } else {
175                    try {
176                        builder.value(o);
177                    } catch (Exception e) {
178                        throw new IOException("unknown object class:" + o.getClass().getName());
179                    }
180                }
181            }
182            builder.endObject();
183            return builder;
184        }
185    
186        protected XContentBuilder toXContent(XContentBuilder builder, Params params, List list) throws IOException {
187            builder.startArray();
188            for (Object o : list) {
189                if (o instanceof Values) {
190                    Values v = (Values) o;
191                    v.toXContent(builder, ToXContent.EMPTY_PARAMS);
192                } else if (o instanceof Map) {
193                    toXContent(builder, params, (Map<String, Object>) o);
194                } else if (o instanceof List) {
195                    toXContent(builder, params, (List) o);
196                } else {
197                    try {
198                        builder.value(o);
199                    } catch (Exception e) {
200                        throw new IOException("unknown object class:" + o.getClass().getName());
201                    }
202                }
203            }
204            builder.endArray();
205            return builder;
206        }
207    
208        public boolean isEmpty() {
209            return optype() == null && index() == null && type() == null && id() == null && core.isEmpty();
210        }
211    
212        public void clear() {
213            this.meta = null;
214            this.core = null;
215        }
216    
217        @Override
218        public String toString() {
219            return "[" + optype() + "/" + index() + "/" + type() + "/" + id() + "]->" + core;
220        }
221    
222    }
