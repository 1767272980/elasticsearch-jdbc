002    package org.xbib.elasticsearch.gatherer;
003    
004    import java.util.EnumSet;
005    import java.util.Set;
006    
007    import static org.elasticsearch.common.collect.Sets.newHashSet;
008    
009    /**
010     * The names of keys with a special meaning for controlling Elasticsearch indexing.
011     *
012     * Mostly, they map to the Elasticsearch bulk item control keys.
013     *
014     *
015     * The _job column denotes an ID for the event of a fetch execution.
016     */
017    public enum ControlKeys {
018    
019        _optype, _index, _type, _id, _version, _timestamp, _ttl, _routing, _parent, _source, _job;
020    
021        public static Set<String> makeSet() {
022            Set<String> set = newHashSet();
023            for (ControlKeys k : EnumSet.allOf(ControlKeys.class)) {
024                set.add(k.name());
025            }
026            return set;
027        }
028    
029    }
