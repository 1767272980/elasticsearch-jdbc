001    
002    package org.xbib.elasticsearch.gatherer;
003    
004    import org.elasticsearch.common.xcontent.ToXContent;
005    import org.elasticsearch.common.xcontent.XContentBuilder;
006    
007    import java.io.IOException;
008    
009    /**
010     * This class represents one or many values. Each value is represented at most
011     * once. New values are appended.
012     */
013    public class Values<O extends Object> implements ToXContent {
014    
015        /**
016         * The values are implemented as an object array
017         */
018        private O[] values = (O[]) new Object[0];
019    
020        /**
021         * Create new Values from an existing values by appending a value.
022         *
023         * @param old existing values or null if values should be created
024         * @param value  a new value
025         * @param sequence true if value should be splitted by commas to multiple values
026         */
027        public Values(Object old, O value, boolean sequence) {
028            if (old instanceof Values) {
029                O[] vals = (O[]) ((Values) old).getValues();
030                if (vals != null) {
031                    this.values = vals;
032                }
033            }
034            O[] newValues;
035            if (sequence && value != null) {
036                newValues = (O[]) value.toString().split(",");
037            } else if (value instanceof Object[]) {
038                newValues = (O[]) value;
039            } else {
040                newValues = (O[]) new Object[]{value};
041            }
042            for (O v : newValues) {
043                addValue(v);
044            }
045        }
046    
047        /**
048         * Append value to existing values.
049         *
050         * @param v the value to add
051         */
052        public void addValue(O v) {
053            int l = this.values.length;
054            for (O aValue : this.values) {
055                if (v != null && v.equals(aValue)) {
056                    //value already found, do nothing
057                    return;
058                }
059            }
060            if (l == 0) {
061                this.values = (O[]) new Object[]{v};
062            } else {
063                // never add a null value to an existing list of values
064                if (v != null) {
065                    if (l == 1 && this.values[0] == null) {
066                        // if there's one existing value and it's null, replace it with the new one
067                        this.values = (O[]) new Object[]{v};
068                    } else {
069                        // otherwise copy the existing value(s) and add the new one
070                        O[] oldValues = this.values;
071                        this.values = (O[]) new Object[l + 1];
072                        System.arraycopy(oldValues, 0, this.values, 0, l);
073                        this.values[l] = v;
074                    }
075                }
076            }
077        }
078    
079        /**
080         * Get the values.
081         *
082         * @return the values
083         */
084        public O[] getValues() {
085            return this.values;
086        }
087    
088        public boolean isNull() {
089            return this.values.length == 0 || (this.values.length == 1 && this.values[0] == null);
090        }
091    
092        @Override
093        public String toString() {
094            StringBuilder sb = new StringBuilder();
095            if (this.values.length > 1) {
096                sb.append('[');
097            }
098            for (int i = 0; i < this.values.length; i++) {
099                if (i > 0) {
100                    sb.append(',');
101                }
102                sb.append(format(this.values[i]));
103            }
104            if (this.values.length > 1) {
105                sb.append(']');
106            }
107            return sb.toString();
108        }
109    
110        /**
111         * Format a value for JSON.
112         *
113         * @param o the value
114         * @return the formtted value
115         */
116        private String format(Object o) {
117            if (o == null) {
118                return "null";
119            }
120            if (o instanceof Integer) {
121                return Integer.toString((Integer) o);
122            }
123            if (o instanceof Long) {
124                return Long.toString((Long) o);
125            }
126            if (o instanceof Float) {
127                return Float.toString((Float) o);
128            }
129            if (o instanceof Double) {
130                return Double.toString((Double) o);
131            }
132            if (o instanceof Boolean) {
133                return Boolean.toString((Boolean) o);
134            }
135            // stringify
136            String t = o.toString();
137            t = t.replaceAll("\"", "\\\"");
138            return '"' + t + '"';
139        }
140    
141        @Override
142        public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
143            if (this.values.length > 1) {
144                builder.startArray();
145            }
146            for (O aValue : this.values) {
147                builder.value(aValue);
148            }
149            if (this.values.length > 1) {
150                builder.endArray();
151            }
152            return builder;
153        }
154    }
