092    /**
093     * Prepare a feed run. Open a client to Elasticsearch and create a queue of input specifications.
094     * This method is executed in a single thread before feed threads are created.
095     * @return this feeder
096     * @throws IOException
097     */
098    public Feeder<T, R, P> beforeRun() throws IOException {
099        if (spec == null) {
100            throw new IllegalArgumentException("no spec?");
101        }
102        if (settings == null) {
103            throw new IllegalArgumentException("no settings?");
104        }
105        if (ingest == null) {
106            Integer maxbulkactions = settings.getAsInt("maxbulkactions", 1000);
107            Integer maxconcurrentbulkrequests = settings.getAsInt("maxconcurrentbulkrequests",
108                    Runtime.getRuntime().availableProcessors());
109            ByteSizeValue maxvolume = settings.getAsBytesSize("maxbulkvolume", ByteSizeValue.parseBytesSizeValue("10m"));
110            TimeValue maxrequestwait = settings.getAsTime("maxrequestwait", TimeValue.timeValueSeconds(60));
111            ingest = "ingest".equals(settings.get("client")) ?
112                    new IngestTransportClient()
113                    : new BulkTransportClient();
114            ingest.maxActionsPerBulkRequest(maxbulkactions)
115                    .maxConcurrentBulkRequests(maxconcurrentbulkrequests)
116                    .maxVolumePerBulkRequest(maxvolume)
117                    .maxRequestWait(maxrequestwait);
118            ingest.newClient(URI.create(settings.get("elasticsearch")));
119        }
120        // create queue
121        super.beforeRun();
122        return this;
123    }
124
125    /**
126     * Each feed thread executes this method to do the main work.
127     * This method is executed in paralle by all feed threads.
128     * @param map the specification f the task to perform
129     * @throws Exception
130     */
131    @Override
132    public void executeTask(Map<String, Object> map) throws Exception {
133        if (isInterrupted()) {
134            logger.warn("interrupted");
135            return;
136        }
137        createRiverContext(getType(), getName(), map);
138        startBulk();
139        if (riverState == null) {
140            riverState = new RiverState();
141        }
142        riverState.load(ingest.client());
143        riverState.setCustom(riverContext.asMap());
144        // increment state counter
145        Long counter = riverState.getCounter() + 1;
146        this.riverState = riverState.setCounter(counter)
147                .setEnabled(true)
148                .setActive(true)
149                .setTimestamp(new Date());
150        riverState.save(ingest.client());
151        if (logger.isDebugEnabled()) {
152            logger.debug("state saved before fetch");
153        }
154        // set the job number to the state counter
155        riverContext.job(Long.toString(counter));
156        if (logger.isDebugEnabled()) {
157            logger.debug("trying to fetch ...");
158        }
159        riverContext.getRiverSource().fetch();
160        if (logger.isDebugEnabled()) {
161            logger.debug("fetched, flushing");
162        }
163        riverContext.getRiverMouth().flush();
164        if (logger.isDebugEnabled()) {
165            logger.debug("flushed");
166        }
167        // we don't know if this is the last run. Stop bulk for now, make indexed documents visible for search
168        stopBulk();
169    }

171    protected void createRiverContext(String riverType, String riverName, Map<String, Object> mySettings) {
172        String strategy = XContentMapValues.nodeStringValue(mySettings.get("strategy"), "simple");
173
174        String url = XContentMapValues.nodeStringValue(mySettings.get("url"), null);
175        String user = XContentMapValues.nodeStringValue(mySettings.get("user"), null);
176        String password = XContentMapValues.nodeStringValue(mySettings.get("password"), null);
177        List<SQLCommand> sql = SQLCommand.parse(mySettings);
178        String rounding = XContentMapValues.nodeStringValue(mySettings.get("rounding"), null);
179        int scale = XContentMapValues.nodeIntegerValue(mySettings.get("scale"), 2);
180        boolean autocommit = XContentMapValues.nodeBooleanValue(mySettings.get("autocommit"), false);
181        int fetchsize = url != null && url.startsWith("jdbc:mysql") ? Integer.MIN_VALUE :
182                XContentMapValues.nodeIntegerValue(mySettings.get("fetchsize"), 10);
183        int maxrows = XContentMapValues.nodeIntegerValue(mySettings.get("max_rows"), 0);
184        int maxretries = XContentMapValues.nodeIntegerValue(mySettings.get("max_retries"), 3);
185        TimeValue maxretrywait = XContentMapValues.nodeTimeValue(mySettings.get("max_retries_wait"),
186                        TimeValue.timeValueSeconds(30));
187        String locale = XContentMapValues.nodeStringValue(mySettings.get("locale"),
188                LocaleUtil.fromLocale(Locale.getDefault()));
189        String resultSetType = XContentMapValues.nodeStringValue(mySettings.get("resultset_type"),
190                "TYPE_FORWARD_ONLY");
191        String resultSetConcurrency = XContentMapValues.nodeStringValue(mySettings.get("resultset_concurrency"),
192                "CONCUR_UPDATABLE");
193        boolean shouldIgnoreNull = XContentMapValues.nodeBooleanValue(mySettings.get("ignore_null_values"), false);
194        String timezone = XContentMapValues.nodeStringValue(mySettings.get("timezone"), TimeZone.getDefault().getID());
195        boolean shouldPrepareDatabaseMetadata = XContentMapValues.nodeBooleanValue(mySettings.get("prepare_database_metadata"), false);
196        boolean shouldPrepareResultSetMetadata = XContentMapValues.nodeBooleanValue(mySettings.get("prepare_resultset_metadata"), false);
197
198        RiverSource riverSource = RiverServiceLoader.findRiverSource(strategy);
199        logger.debug("found river source class {} for strategy {}", riverSource.getClass().getName(), strategy);
200        RiverMouth riverMouth = RiverServiceLoader.findRiverMouth(strategy);
201        logger.debug("found river mouth class {} for strategy {}", riverMouth.getClass().getName(), strategy);
202        RiverFlow riverFlow = RiverServiceLoader.findRiverFlow(strategy);
203        logger.debug("found river flow class {} for strategy {}", riverFlow.getClass().getName(), strategy);
204
205        defaultIndex = XContentMapValues.nodeStringValue(mySettings.get("index"), "jdbc");
206        String defaultType = XContentMapValues.nodeStringValue(mySettings.get("type"), "jdbc");
207
208        logger.info("river default index/type {}/{}", defaultIndex, defaultType);
209
210        if (mySettings.containsKey("index_settings")) {
211            try {
212                ingest.setSettings(settingsBuilder().put(new JsonSettingsLoader()
213                        .load(jsonBuilder().map((Map<String, Object>) mySettings.get("index_settings")).string()))
214                        .build());
215                if (mySettings.containsKey("type_mapping")) {
216                    ingest.addMapping(defaultType,
217                        jsonBuilder().map((Map<String, Object>) mySettings.get("type_mapping")).string());
218                }
219            } catch (Exception e) {
220                logger.error(e.getMessage(),e);
221            }
222        }
223        riverSource.setUrl(url)
224                .setUser(user)
225                .setPassword(password)
226                .setTimeZone(TimeZone.getTimeZone(timezone));
227        riverMouth.setIndex(defaultIndex).setType(defaultType).setIngest(ingest);
228        riverFlow.setFeeder(this);
229        this.riverContext = new RiverContext()
230                .setRiverName(riverName)
231                .setRiverSettings(mySettings)
232                .setRiverSource(riverSource)
233                .setRiverMouth(riverMouth)
234                .setRiverFlow(riverFlow)
235                .setLocale(locale)
236                .setRounding(rounding)
237                .setScale(scale)
238                .setStatements(sql)
239                .setAutoCommit(autocommit)
240                .setMaxRows(maxrows)
241                .setFetchSize(fetchsize)
242                .setRetries(maxretries)
243                .setMaxRetryWait(maxretrywait)
244                .setResultSetType(resultSetType)
245                .setResultSetConcurrency(resultSetConcurrency)
246                .shouldIgnoreNull(shouldIgnoreNull)
247                .shouldPrepareDatabaseMetadata(shouldPrepareDatabaseMetadata)
248                .shouldPrepareResultSetMetadata(shouldPrepareResultSetMetadata)
249                .contextualize();
250        logger.trace("JDBC feeder ready to start, context is {}", riverContext);
251    }
252
253    /**
254     * Close a single cycle. First, open connections in the river source are closed,
255     * then river mouth is closed. After this, set river state to disabled and save river state.
256     *
257     * @throws IOException if close failed
258     */
259    @Override
260    public void close() throws IOException {
261        super.close();
262        if (riverContext != null) {
263            riverContext.getRiverSource().closeReading();
264            logger.info("reading connection closed");
265            riverContext.getRiverSource().closeWriting();
266            logger.info("writing connection closed");
267            try {
268                logger.info("river mouth closing");
269                riverContext.getRiverMouth().close();
270                logger.info("river mouth closed");
271            } catch (IOException e) {
272                logger.error(e.getMessage(), e);
273            }
274            this.riverState = riverState
275                    .setEnabled(false)
276                    .setActive(false)
277                    .setTimestamp(new Date());
278            riverState.save(ingest.client());
279            logger.info("river state saved");
280        }
281    }
282
283    private void startBulk() {
284        try {
285            if (!ingest.client().admin().indices().prepareExists(defaultIndex).execute().actionGet().isExists()) {
286                logger.info("creating index {} and enabling bulk mode", defaultIndex);
287                ingest.newIndex(defaultIndex);
288            }
289            ingest.startBulk(defaultIndex);
290        } catch (IOException e) {
291            logger.error(e.getMessage(),e);
292        }
293    }
294
295    private void stopBulk() {
296        State state = ingest.getState();
297        if (state.indices() != null && !state.indices().isEmpty()) {
298            logger.info("stopping bulk mode for indices {}...", state.indices());
299            for (String index : ImmutableSet.copyOf(state.indices())) {
300                try {
301                    ingest.stopBulk(index);
302                } catch (IOException e) {
303                    logger.error(e.getMessage(),e);
304                }
305            }
306        }
307    }
