002package org.xbib.elasticsearch.plugin.feeder.jdbc;
003
004import org.xbib.common.unit.TimeValue;
005import org.xbib.common.xcontent.support.XContentMapValues;
006import org.xbib.elasticsearch.plugin.feeder.AbstractFeeder;
007import org.xbib.elasticsearch.plugin.support.LocaleUtil;
008import org.xbib.elasticsearch.plugin.support.RiverContext;
009import org.xbib.elasticsearch.plugin.support.RiverServiceLoader;
010import org.xbib.elasticsearch.plugin.support.RiverState;
011import org.xbib.elasticsearch.plugin.support.SQLCommand;
012import org.xbib.elasticsearch.river.jdbc.RiverFlow;
013import org.xbib.elasticsearch.river.jdbc.RiverMouth;
014import org.xbib.elasticsearch.river.jdbc.RiverSource;
015import org.xbib.logging.Logger;
016import org.xbib.logging.LoggerFactory;
017import org.xbib.pipeline.Pipeline;
018import org.xbib.pipeline.PipelineProvider;
019import org.xbib.pipeline.PipelineRequest;
020
021import java.io.IOException;
022import java.util.Date;
023import java.util.List;
024import java.util.Locale;
025import java.util.Map;
026
027public class JDBCFeeder<T, R extends PipelineRequest, P extends Pipeline<T, R>>
028        extends AbstractFeeder<T, R, P> {
029
030    private final static Logger logger = LoggerFactory.getLogger(JDBCFeeder.class.getSimpleName());
031
032    protected RiverContext riverContext;
033
034    protected RiverState state;
035
036    public JDBCFeeder() {
037    }
038
039    public JDBCFeeder(JDBCFeeder feeder) {
040        super(feeder);
041    }
042
043    @Override
044    @SuppressWarnings({"unchecked"})
045    public PipelineProvider<P> pipelineProvider() {
046        return () -> (P) new JDBCFeeder(JDBCFeeder.this);
047    }
048
049    @Override
050    public String getType() {
051        return "jdbc";
052    }
053
054    @Override
055    public void process(Map<String, Object> map) throws Exception {
056        logger.info("processing map {}", map);
057        createRiverContext("jdbc", "feeder", map);
058        if (state == null) {
059            // create state object
060            logger.debug("state creation {}/{}/{}", "_river", riverContext.getRiverName(), "_custom");
061            this.state = new RiverState()
062                    .setStarted(new Date())
063                    .setName("jdbc")
064                    .setCoordinates("_river", riverContext.getRiverName(), "_custom")
065                    .setCustom(riverContext.asMap());
066        }
067        state.load(ingest.client());
068        // increment state counter
069        Long counter = state.getCounter() + 1;
070        this.state = state.setCounter(counter)
071                .setEnabled(true)
072                .setActive(true)
073                .setTimestamp(new Date());
074        state.save(ingest.client());
075        if (logger.isDebugEnabled()) {
076            logger.debug("state saved before fetch");
077        }
078        // set the job number to the state counter
079        riverContext.job(Long.toString(counter));
080        if (logger.isDebugEnabled()) {
081            logger.debug("trying to fetch ...");
082        }
083        riverContext.getRiverSource().fetch();
084        if (logger.isDebugEnabled()) {
085            logger.debug("fetched, flushing");
086        }
087        riverContext.getRiverMouth().flush();
088        if (logger.isDebugEnabled()) {
089            logger.debug("flushed");
090        }
091        riverContext.getRiverSource().closeReading();
092        riverContext.getRiverSource().closeWriting();
093        this.state = state
094                .setActive(false)
095                .setTimestamp(new Date());
096        state.save(ingest.client());
097    }
098
099    protected void createRiverContext(String riverType, String riverName, Map<String, Object> mySettings) {
100        String strategy = XContentMapValues.nodeStringValue(mySettings.get("strategy"), "simple");
101
102        String url = XContentMapValues.nodeStringValue(mySettings.get("url"), null);
103        String user = XContentMapValues.nodeStringValue(mySettings.get("user"), null);
104        String password = XContentMapValues.nodeStringValue(mySettings.get("password"), null);
105        List<SQLCommand> sql = SQLCommand.parse(mySettings);
106        String rounding = XContentMapValues.nodeStringValue(mySettings.get("rounding"), null);
107        int scale = XContentMapValues.nodeIntegerValue(mySettings.get("scale"), 2);
108        boolean autocommit = XContentMapValues.nodeBooleanValue(mySettings.get("autocommit"), Boolean.FALSE);
109
110        int fetchsize = url != null && url.startsWith("jdbc:mysql") ? Integer.MIN_VALUE :
111                XContentMapValues.nodeIntegerValue(mySettings.get("fetchsize"), 10);
112        int maxrows = XContentMapValues.nodeIntegerValue(mySettings.get("max_rows"), 0);
113        int maxretries = XContentMapValues.nodeIntegerValue(mySettings.get("max_retries"), 3);
114        TimeValue maxretrywait =
115                XContentMapValues.nodeTimeValue(mySettings.get("max_retries_wait"),
116                        TimeValue.timeValueSeconds(30));
117        String locale = XContentMapValues.nodeStringValue(mySettings.get("locale"),
118                LocaleUtil.fromLocale(Locale.getDefault()));
119        String resultSetType = XContentMapValues.nodeStringValue(mySettings.get("resultset_type"),
120                "TYPE_FORWARD_ONLY");
121        String resultSetConcurrency = XContentMapValues.nodeStringValue(mySettings.get("resultset_concurrency"),
122                "CONCUR_UPDATABLE");
123
124        // defaults for column strategy
125        String columnCreatedAt = XContentMapValues.nodeStringValue(mySettings.get("created_at"), "created_at");
126        String columnUpdatedAt = XContentMapValues.nodeStringValue(mySettings.get("updated_at"), "updated_at");
127        String columnDeletedAt = XContentMapValues.nodeStringValue(mySettings.get("deleted_at"), null);
128        boolean columnEscape = XContentMapValues.nodeBooleanValue(mySettings.get("column_escape"), true);
129
130        // set up bulk indexer. If no _index or _type pseudo columns are set in SQL statements, this index is used.
131        String indexName = XContentMapValues.nodeStringValue(mySettings.get("index"), "jdbc");
132        String typeName = XContentMapValues.nodeStringValue(mySettings.get("type"), "jdbc");
133        // get two maps from the river settings to improve index creation
134        Map<String, Object> indexSettings = mySettings.containsKey("index_settings") ?
135                (Map<String, Object>) mySettings.get("index_settings") : null;
136        Map<String, Object> typeMapping = mySettings.containsKey("type_mapping") ?
137                (Map<String, Object>) mySettings.get("type_mapping") : null;
138
139        RiverSource riverSource = RiverServiceLoader.findRiverSource(strategy);
140        logger.debug("found river source class {} for strategy {}", riverSource.getClass().getName(), strategy);
141        RiverMouth riverMouth = RiverServiceLoader.findRiverMouth(strategy);
142        logger.debug("found river mouth class {} for strategy {}", riverMouth.getClass().getName(), strategy);
143        RiverFlow riverFlow = RiverServiceLoader.findRiverFlow(strategy);
144        logger.debug("found river flow class {} for strategy {}", riverFlow.getClass().getName(), strategy);
145
146        riverSource.setUrl(url)
147                .setUser(user)
148                .setPassword(password);
149        riverMouth.setSettings(indexSettings)
150                .setMapping(typeMapping)
151                .setIndex(indexName)
152                .setType(typeName)
153                .setIngest(ingest);
154        riverFlow.setFeeder(this);
155        this.riverContext = new RiverContext()
156                .setRiverName(riverName)
157                .setRiverSettings(mySettings)
158                .setRiverSource(riverSource)
159                .setRiverMouth(riverMouth)
160                .setRiverFlow(riverFlow)
161                .setLocale(locale)
162                .setRounding(rounding)
163                .setScale(scale)
164                .setStatements(sql)
165                .setAutoCommit(autocommit)
166                .setMaxRows(maxrows)
167                .setFetchSize(fetchsize)
168                .setRetries(maxretries)
169                .setMaxRetryWait(maxretrywait)
170                .setResultSetType(resultSetType)
171                .setResultSetConcurrency(resultSetConcurrency)
172                .columnCreatedAt(columnCreatedAt)
173                .columnUpdatedAt(columnUpdatedAt)
174                .columnDeletedAt(columnDeletedAt)
175                .columnEscape(columnEscape)
176                .contextualize();
177        logger.info("JDBC feeder ready to start");
178    }
179
180    @Override
181    public void close() throws IOException {
182        super.close();
183        if (riverContext != null) {
184            riverContext.getRiverSource().closeReading();
185            logger.info("reading connection closed");
186            riverContext.getRiverSource().closeWriting();
187            logger.info("writing connection closed");
188            this.state = state
189                    .setEnabled(false)
190                    .setActive(false)
191                    .setTimestamp(new Date());
192            state.save(ingest.client());
193            logger.info("state saved");
194            try {
195                logger.info("river mouth closing");
196                riverContext.getRiverMouth().close();
197                logger.info("river mouth closed");
198            } catch (IOException e) {
199                logger.error(e.getMessage(), e);
200            }
201        }
202    }
