035
036public class JDBCFeeder<T, R extends PipelineRequest, P extends Pipeline<T, R>>
037        extends AbstractFeeder<T, R, P> {
038
039    private final static ESLogger logger = ESLoggerFactory.getLogger(JDBCFeeder.class.getSimpleName());
040
041    protected RiverContext riverContext;
042
043    public JDBCFeeder() {
044    }
045
046    public JDBCFeeder(JDBCFeeder feeder) {
047        super(feeder);
048    }
049
050    @Override
051    @SuppressWarnings({"unchecked"})
052    public PipelineProvider<P> pipelineProvider() {
053        return new PipelineProvider<P>() {
054            @Override
055            public P get() {
056                return  (P) new JDBCFeeder(JDBCFeeder.this);
057            }
058        };
059    }
060
061    @Override
062    public String getType() {
063        return "jdbc";
064    }
065
066    public Feeder<T, R, P> beforeRun() throws IOException {
067        if (spec == null) {
068            throw new IllegalArgumentException("no spec?");
069        }
070        if (settings == null) {
071            throw new IllegalArgumentException("no settings?");
072        }
073        if (ingest == null) {
074            Integer maxbulkactions = settings.getAsInt("maxbulkactions", 1000);
075            Integer maxconcurrentbulkrequests = settings.getAsInt("maxconcurrentbulkrequests",
076                    Runtime.getRuntime().availableProcessors());
077            ByteSizeValue maxvolume = settings.getAsBytesSize("maxbulkvolume", ByteSizeValue.parseBytesSizeValue("10m"));
078            TimeValue maxrequestwait = settings.getAsTime("maxrequestwait", TimeValue.timeValueSeconds(60));
079            ingest = "ingest".equals(settings.get("client")) ?
080                    new IngestTransportClient()
081                    : new BulkTransportClient();
082            ingest.maxActionsPerBulkRequest(maxbulkactions)
083                    .maxConcurrentBulkRequests(maxconcurrentbulkrequests)
084                    .maxVolumePerBulkRequest(maxvolume)
085                    .maxRequestWait(maxrequestwait);
086            ingest.newClient(URI.create(settings.get("elasticsearch")));
087        }
088        // create queue
089        super.beforeRun();
090        return this;
091    }
092
093    @Override
094    public void executeTask(Map<String, Object> map) throws Exception {
095        if (isInterrupted()) {
096            logger.warn("interrupted");
097            return;
098        }
099        createRiverContext("jdbc", "feeder", map);
100        if (riverState == null) {
101            riverState = new RiverState();
102        }
103        riverState.load(ingest.client());
104        riverState.setCustom(riverContext.asMap());
105        // increment state counter
106        Long counter = riverState.getCounter() + 1;
107        this.riverState = riverState.setCounter(counter)
108                .setEnabled(true)
109                .setActive(true)
110                .setTimestamp(new Date());
111        riverState.save(ingest.client());
112        if (logger.isDebugEnabled()) {
113            logger.debug("state saved before fetch");
114        }
115        // set the job number to the state counter
116        riverContext.job(Long.toString(counter));
117        if (logger.isDebugEnabled()) {
118            logger.debug("trying to fetch ...");
119        }
120        riverContext.getRiverSource().fetch();
121        if (logger.isDebugEnabled()) {
122            logger.debug("fetched, flushing");
123        }
124        riverContext.getRiverMouth().flush();
125        if (logger.isDebugEnabled()) {
126            logger.debug("flushed");
127        }
128    }
129
130    protected void createRiverContext(String riverType, String riverName, Map<String, Object> mySettings) {
131        String strategy = XContentMapValues.nodeStringValue(mySettings.get("strategy"), "simple");
132
133        String url = XContentMapValues.nodeStringValue(mySettings.get("url"), null);
134        String user = XContentMapValues.nodeStringValue(mySettings.get("user"), null);
135        String password = XContentMapValues.nodeStringValue(mySettings.get("password"), null);
136        List<SQLCommand> sql = SQLCommand.parse(mySettings);
137        String rounding = XContentMapValues.nodeStringValue(mySettings.get("rounding"), null);
138        int scale = XContentMapValues.nodeIntegerValue(mySettings.get("scale"), 2);
139        boolean autocommit = XContentMapValues.nodeBooleanValue(mySettings.get("autocommit"), Boolean.FALSE);
140        int fetchsize = url != null && url.startsWith("jdbc:mysql") ? Integer.MIN_VALUE :
141                XContentMapValues.nodeIntegerValue(mySettings.get("fetchsize"), 10);
142        int maxrows = XContentMapValues.nodeIntegerValue(mySettings.get("max_rows"), 0);
143        int maxretries = XContentMapValues.nodeIntegerValue(mySettings.get("max_retries"), 3);
144        TimeValue maxretrywait =
145                XContentMapValues.nodeTimeValue(mySettings.get("max_retries_wait"),
146                        TimeValue.timeValueSeconds(30));
147        String locale = XContentMapValues.nodeStringValue(mySettings.get("locale"),
148                LocaleUtil.fromLocale(Locale.getDefault()));
149        String resultSetType = XContentMapValues.nodeStringValue(mySettings.get("resultset_type"),
150                "TYPE_FORWARD_ONLY");
151        String resultSetConcurrency = XContentMapValues.nodeStringValue(mySettings.get("resultset_concurrency"),
152                "CONCUR_UPDATABLE");
153
154        RiverSource riverSource = RiverServiceLoader.findRiverSource(strategy);
155        logger.debug("found river source class {} for strategy {}", riverSource.getClass().getName(), strategy);
156        RiverMouth riverMouth = RiverServiceLoader.findRiverMouth(strategy);
157        logger.debug("found river mouth class {} for strategy {}", riverMouth.getClass().getName(), strategy);
158        RiverFlow riverFlow = RiverServiceLoader.findRiverFlow(strategy);
159        logger.debug("found river flow class {} for strategy {}", riverFlow.getClass().getName(), strategy);
160
161        String defaultIndex = XContentMapValues.nodeStringValue(mySettings.get("index"), "jdbc");
162        String defaultType = XContentMapValues.nodeStringValue(mySettings.get("type"), "jdbc");
163
164        logger.info("river default index/type {}/{}", defaultIndex, defaultType);
165
166        if (mySettings.containsKey("index_settings")) {
167            try {
168                ingest.setSettings(settingsBuilder().put(new JsonSettingsLoader()
169                        .load(jsonBuilder().map((Map<String, Object>) mySettings.get("index_settings")).string()))
170                        .build());
171                if (mySettings.containsKey("type_mapping")) {
172                    ingest.addMapping(defaultType,
173                        jsonBuilder().map((Map<String, Object>) mySettings.get("type_mapping")).string());
174                }
175            } catch (Exception e) {
176                logger.error(e.getMessage(),e);
177            }
178        }
179        try {
180            ingest.newIndex(defaultIndex).startBulk(defaultIndex);
181        } catch (IOException e) {
182            logger.error(e.getMessage(),e);
183        }
184        riverSource.setUrl(url)
185                .setUser(user)
186                .setPassword(password);
187        riverMouth.setIndex(defaultIndex).setType(defaultType).setIngest(ingest);
188        riverFlow.setFeeder(this);
189        this.riverContext = new RiverContext()
190                .setRiverName(riverName)
191                .setRiverSettings(mySettings)
192                .setRiverSource(riverSource)
193                .setRiverMouth(riverMouth)
194                .setRiverFlow(riverFlow)
195                .setLocale(locale)
196                .setRounding(rounding)
197                .setScale(scale)
198                .setStatements(sql)
199                .setAutoCommit(autocommit)
200                .setMaxRows(maxrows)
201                .setFetchSize(fetchsize)
202                .setRetries(maxretries)
203                .setMaxRetryWait(maxretrywait)
204                .setResultSetType(resultSetType)
205                .setResultSetConcurrency(resultSetConcurrency)
206                .contextualize();
207        logger.trace("JDBC feeder ready to start, context is {}", riverContext);
208    }
209
210    /**
211     * Close a single cycle. First, open connections in the river source are closed,
212     * then river mouth is closed. After this, set river state to disabled and save river state.
213     *
214     * @throws IOException if close failed
215     */
216    @Override
217    public void close() throws IOException {
218        super.close();
219        if (riverContext != null) {
220            riverContext.getRiverSource().closeReading();
221            logger.info("reading connection closed");
222            riverContext.getRiverSource().closeWriting();
223            logger.info("writing connection closed");
224            try {
225                logger.info("river mouth closing");
226                riverContext.getRiverMouth().close();
227                logger.info("river mouth closed");
228            } catch (IOException e) {
229                logger.error(e.getMessage(), e);
230            }
231            this.riverState = riverState
232                    .setEnabled(false)
233                    .setActive(false)
234                    .setTimestamp(new Date());
235            riverState.save(ingest.client());
236            logger.info("river state saved");
237        }
238    }
