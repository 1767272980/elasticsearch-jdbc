001package org.xbib.elasticsearch.plugin.support;
002
003import org.xbib.common.xcontent.ToXContent;
004import org.xbib.common.xcontent.XContentBuilder;
005
006import java.io.IOException;
007import java.util.List;
008import java.util.Map;
009import java.util.TreeMap;
010
011import static org.xbib.common.xcontent.XContentFactory.jsonBuilder;
012
013/**
014 * A basic indexable object. The indexable object can store meta data and core data.
015 */
016public class PlainIndexableObject implements IndexableObject, ToXContent {
017
018    private Map<String, String> meta;
019
020    private Map<String, Object> core;
021
022    public PlainIndexableObject() {
023        this.meta = new TreeMap<String, String>();
024        this.core = new TreeMap<String, Object>();
025
026    }
027
028    public IndexableObject optype(String optype) {
029        meta.put(ControlKeys._optype.name(), optype);
030        return this;
031    }
032
033    public String optype() {
034        return meta.get(ControlKeys._optype.name());
035    }
036
037    public IndexableObject index(String index) {
038        meta.put(ControlKeys._index.name(), index);
039        return this;
040    }
041
042    public String index() {
043        return meta.get(ControlKeys._index.name());
044    }
045
046    public IndexableObject type(String type) {
047        meta.put(ControlKeys._type.name(), type);
048        return this;
049    }
050
051    public String type() {
052        return meta.get(ControlKeys._type.name());
053    }
054
055    public IndexableObject id(String id) {
056        meta.put(ControlKeys._id.name(), id);
057        return this;
058    }
059
060    public String id() {
061        return meta.get(ControlKeys._id.name());
062    }
063
064    public IndexableObject meta(String key, String value) {
065        meta.put(key, value);
066        return this;
067    }
068
069    public String meta(String key) {
070        return meta.get(key);
071    }
072
073    public IndexableObject source(Map<String, Object> source) {
074        this.core = source;
075        return this;
076    }
077
078    public Map<String, Object> source() {
079        return core;
080    }
081
082    /**
083     * Build a string that can be used for indexing.
084     *
085     * @throws java.io.IOException when build gave an error
086     */
087    public String build() throws IOException {
088        XContentBuilder builder = jsonBuilder();
089        toXContent(builder, ToXContent.EMPTY_PARAMS);
090        return builder.string();
091    }
092
093    @Override
094    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
095        toXContent(builder, params, core);
096        return builder;
097    }
098
099    /**
100     * Recursive method to build XContent from a key/value map of Values
101     *
102     * @param builder the builder
103     * @param map     the map
104     * @return the XContent builder
105     * @throws java.io.IOException when method gave an error
106     */
107    @SuppressWarnings({"unchecked"})
108    protected XContentBuilder toXContent(XContentBuilder builder, Params params, Map<String, Object> map) throws IOException {
109        builder.startObject();
110        for (Map.Entry<String, Object> k : map.entrySet()) {
111            builder.field(k.getKey());
112            Object o = k.getValue();
113            if (o instanceof Values) {
114                Values v = (Values) o;
115                v.toXContent(builder, params);
116            } else if (o instanceof Map) {
117                toXContent(builder, params, (Map<String, Object>) o);
118            } else if (o instanceof List) {
119                toXContent(builder, params, (List) o);
120            } else {
121                try {
122                    builder.value(o);
123                } catch (Exception e) {
124                    throw new IOException("unknown object class:" + o.getClass().getName());
125                }
126            }
127        }
128        builder.endObject();
129        return builder;
130    }
131
132    @SuppressWarnings({"unchecked"})
133    protected XContentBuilder toXContent(XContentBuilder builder, Params params, List list) throws IOException {
134        builder.startArray();
135        for (Object o : list) {
136            if (o instanceof Values) {
137                Values v = (Values) o;
138                v.toXContent(builder, ToXContent.EMPTY_PARAMS);
139            } else if (o instanceof Map) {
140                toXContent(builder, params, (Map<String, Object>) o);
141            } else if (o instanceof List) {
142                toXContent(builder, params, (List) o);
143            } else {
144                try {
145                    builder.value(o);
146                } catch (Exception e) {
147                    throw new IOException("unknown object class:" + o.getClass().getName());
148                }
149            }
150        }
151        builder.endArray();
152        return builder;
153    }
154
155    public boolean isEmpty() {
156        return optype() == null && index() == null && type() == null && id() == null && core.isEmpty();
157    }
158
159    @Override
160    public String toString() {
161        return "[" + optype() + "/" + index() + "/" + type() + "/" + id() + "]->" + core;
162    }
163
164    @Override
165    public int compareTo(IndexableObject o) {
166        if (o == null) {
167            return -1;
168        }
169        String s1 = optype() + '/' + index() + '/' + type() + '/' + id();
170        String s2 = o.optype() + '/' + o.index() + '/' + o.type() + '/' + o.id();
171        return s1.compareTo(s2);
172    }
173
174    @Override
175    public boolean equals(Object o) {
176        if (o == null) {
177            return false;
178        }
179        if (!(o instanceof IndexableObject)) {
180            return false;
181        }
182        IndexableObject c = (IndexableObject) o;
183        return equal(optype(), c.optype()) &&
184                equal(index(), c.index()) &&
185                equal(type(), c.type()) &&
186                equal(id(), c.id());
187    }
188
189    @Override
190    public int hashCode() {
191        int hash = 3;
192        hash = 37 * hash + (optype() != null ? optype().hashCode() : 0);
193        hash = 37 * hash + (index() != null ? index().hashCode() : 0);
194        hash = 37 * hash + (type() != null ? type().hashCode() : 0);
195        hash = 37 * hash + (id() != null ? id().hashCode() : 0);
196        return hash;
197    }
198
199    private boolean equal(Object a, Object b) {
200        if (a == null && b == null) {
201            return true;
202        }
203        if (a != null && b != null) {
204            if (a.equals(b)) {
205                return true;
206            }
207        }
208        return false;
209    }
210}
