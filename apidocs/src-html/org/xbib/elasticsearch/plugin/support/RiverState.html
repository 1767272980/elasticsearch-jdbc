189    public RiverState save(Client client) throws IOException {
190        if (coordinateIndex == null || coordinateType == null || coordinateId == null) {
191            return this;
192        }
193        XContentBuilder builder = jsonBuilder();
194        builder = toXContent(builder, ToXContent.EMPTY_PARAMS);
195        if (logger.isDebugEnabled()) {
196            logger.debug("saving state={}", builder.string());
197        }
198        client.prepareIndex()
199                .setIndex(coordinateIndex)
200                .setType(coordinateType)
201                .setId(coordinateId)
202                .setSource(builder.string())
203                .execute()
204                .actionGet();
205        // refresh index to make this visible
206        client.admin().indices().prepareRefresh(coordinateIndex).execute().actionGet();
207        if (logger.isDebugEnabled()) {
208            logger.debug("state saved");
209        }
210        return this;
211    }
212
213    public RiverState load(Client client) throws IOException {
214        if (coordinateIndex == null || coordinateType == null || coordinateId == null) {
215            return this;
216        }
217        GetResponse get = null;
218        try {
219            get = client.prepareGet(coordinateIndex, coordinateType, coordinateId).execute().actionGet();
220        } catch (Exception e) {
221            logger.warn("state not found: {}/{}/{}", coordinateIndex, coordinateType, coordinateId);
222        }
223        if (get != null && get.isExists()) {
224            if (logger.isDebugEnabled()) {
225                logger.debug("load state={}", get.getSourceAsString());
226            }
227            XContentParser parser = XContentFactory.xContent(XContentType.JSON)
228                    .createParser(get.getSourceAsBytes());
229            fromXContent(parser);
230        } else {
231            counter = 0L;
232        }
233        return this;
234    }
235
236    public void fromXContent(XContentParser parser) throws IOException {
237        DateMathParser dateParser = new DateMathParser(Joda.forPattern("dateOptionalTime"), TimeUnit.MILLISECONDS);
238        String currentFieldName = null;
239        XContentParser.Token token;
240        while ((token = parser.nextToken()) != END_OBJECT) {
241            if (token == null) {
242                break;
243            } else if (token == FIELD_NAME) {
244                currentFieldName = parser.currentName();
245            } else if (token == VALUE_NULL || token == VALUE_STRING
246                    || token == VALUE_BOOLEAN || token == VALUE_NUMBER) {
247                switch (currentFieldName) {
248                    case "name":
249                        setName(parser.text());
250                        break;
251                    case "type":
252                        setType(parser.text());
253                        break;
254                    case "started":
255                        try {
256                            this.started = new Date(dateParser.parse(parser.text(), 0));
257                        } catch (Exception e) {
258                            // ignore
259                        }
260                        break;
261                    case "timestamp":
262                        try {
263                            setTimestamp(new Date(dateParser.parse(parser.text(), 0)));
264                        } catch (Exception e) {
265                            // ignore
266                        }
267                        break;
268                    case "counter":
269                        try {
270                            setCounter(parser.longValue());
271                        } catch (Exception e) {
272                            // ignore
273                        }
274                        break;
275                    case "enabled":
276                        setEnabled(parser.booleanValue());
277                        break;
278                    case "active":
279                        setActive(parser.booleanValue());
280                        break;
281                }
282            } else if (token == START_OBJECT) {
283                if ("custom".equals(currentFieldName)) {
284                    setCustom(parser.map());
285                } else if ("settings".equals(currentFieldName)) {
286                    setSettings(ImmutableSettings.readSettingsFromMap(parser.map()));
287                }
288            }
289        }
290    }

292    @Override
293    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
294        builder.startObject()
295                .fieldIfNotNull("name", name)
296                .fieldIfNotNull("type", type)
297                .fieldIfNotNull("enabled", enabled)
298                .fieldIfNotNull("started", started)
299                .fieldIfNotNull("timestamp", timestamp)
300                .fieldIfNotNull("counter", counter)
301                .fieldIfNotNull("active", active)
302                .fieldIfNotNull("custom", custom)
303                .fieldIfNotNull("settings", settings != null ? settings.getAsMap() : null)
304                .endObject();
305        return builder;
306    }
307
308    public String toString() {
309        try {
310            XContentBuilder builder = jsonBuilder();
311            toXContent(builder, XContentBuilder.EMPTY_PARAMS);
312            return builder.string();
313        } catch (IOException e) {
314            return "";
315        }
316    }
317
318    @Override
319    public void readFrom(StreamInput in) throws IOException {
320        name = in.readOptionalString();
321        type = in.readOptionalString();
322        started = new Date(in.readLong());
323        timestamp = new Date(in.readLong());
324        enabled = in.readBoolean();
325        active = in.readBoolean();
326        counter = in.readLong();
327        custom = in.readMap();
328        boolean b = in.readBoolean();
329        if (b) {
330            settings = ImmutableSettings.readSettingsFromMap(in.readMap());
331        } else {
332            settings = null;
333        }
334    }
335
336    @Override
337    public void writeTo(StreamOutput out) throws IOException {
338        out.writeOptionalString(name);
339        out.writeOptionalString(type);
340        out.writeLong(started.getTime());
341        out.writeLong(timestamp.getTime());
342        out.writeBoolean(enabled);
343        out.writeBoolean(active);
344        out.writeLong(counter);
345        out.writeMap(custom);
346        if (settings != null) {
347            out.writeBoolean(true);
348            Map<String, Object> map = new HashMap<>();
349            ImmutableSettings.writeSettingsToMap(settings, map);
350            out.writeMap(map);
351        } else {
352            out.writeBoolean(false);
353        }
354    }
355
356    @Override
357    public int compareTo(RiverState o) {
358        return toString().compareTo(o.toString());
359    }
