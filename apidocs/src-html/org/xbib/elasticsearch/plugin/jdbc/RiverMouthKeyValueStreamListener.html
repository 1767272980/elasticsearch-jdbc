001package org.xbib.elasticsearch.plugin.jdbc;
002
003import org.xbib.elasticsearch.river.jdbc.RiverMouth;
004
005import java.io.IOException;
006
007/**
008 * This class consumes pairs from a key/value stream
009 * and transports them to the river mouth.
010 */
011public class RiverMouthKeyValueStreamListener<K,V> extends PlainKeyValueStreamListener<K, V> {
012
013    private RiverMouth output;
014
015    public RiverMouthKeyValueStreamListener<K,V> output(RiverMouth output) {
016        this.output = output;
017        return this;
018    }
019
020    public RiverMouthKeyValueStreamListener<K,V> shouldIgnoreNull(boolean shouldIgnoreNull) {
021        super.shouldIgnoreNull(shouldIgnoreNull);
022        return this;
023    }
024
025    /**
026     * The object is complete. Push it to the river mouth.
027     *
028     * @param object the object
029     * @return this value listener
030     * @throws java.io.IOException
031     */
032    public RiverMouthKeyValueStreamListener<K,V> end(IndexableObject object) throws IOException {
033        if (object.isEmpty()) {
034            return this;
035        }
036        if (output != null) {
037            if (object.optype() == null) {
038                output.index(object, false);
039            } else if ("index".equals(object.optype())) {
040                output.index(object, false);
041            } else if ("create".equals(object.optype())) {
042                output.index(object, true);
043            } else if ("delete".equals(object.optype())) {
044                output.delete(object);
045            } else {
046                throw new IllegalArgumentException("unknown optype: " + object.optype());
047            }
048        }
049        return this;
050    }
051
052}
