001package org.xbib.elasticsearch.plugin.jdbc;
002
003import org.elasticsearch.common.base.Strings;
004
005import java.util.Locale;
006
007public class LocaleUtil {
008
009    public static Locale toLocale(String localeString) {
010        if (Strings.isNullOrEmpty(localeString)) {
011            return Locale.getDefault();
012        }
013        int separatorCountry = localeString.indexOf('_');
014        char separator;
015        if (separatorCountry >= 0) {
016            separator = '_';
017        } else {
018            separatorCountry = localeString.indexOf('-');
019            separator = '-';
020        }
021        String language;
022        String country;
023        String variant;
024        if (separatorCountry < 0) {
025            language = localeString;
026            country = "";
027            variant = "";
028        } else {
029            language = localeString.substring(0, separatorCountry);
030            int separatorVariant = localeString.indexOf(separator, separatorCountry + 1);
031            if (separatorVariant < 0) {
032                country = localeString.substring(separatorCountry + 1);
033                variant = "";
034            } else {
035                country = localeString.substring(separatorCountry + 1, separatorVariant);
036                variant = localeString.substring(separatorVariant + 1);
037            }
038        }
039        return new Locale(language, country, variant);
040    }
041
042    public static String fromLocale(Locale locale) {
043        return locale.getLanguage() +
044                (Strings.isNullOrEmpty(locale.getCountry()) ? "_" + locale.getCountry() : "");
045    }
046}
