015/**
016 */
017public class PlainKeyValueStreamListener<K, V> implements KeyValueStreamListener<K, V> {
018
019    /**
020     * The current structured object
021     */
022    private IndexableObject current;
023    /**
024     * The object before the current object
025     */
026    private IndexableObject prev;
027
028    /**
029     * The keys of the values. They are examined for the Elasticsearch index
030     * attributes.
031     */
032    private List<K> keys;
033    /**
034     * The delimiter between the key names in the path, used for string split.
035     * Should not be modified.
036     */
037    private char delimiter = '.';
038
039    private boolean shouldAutoGenID;
040
041    private boolean shouldIgnoreNull = false;

043    /**
044     * Set custom delimiter
045     *
046     * @param delimiter the delimiter

049    public PlainKeyValueStreamListener delimiter(char delimiter) {
050        this.delimiter = delimiter;
051        return this;
052    }

055        this.shouldIgnoreNull = shouldIgnoreNull;

066        return this;

070    public KeyValueStreamListener<K, V> keyValue(K key, V value) throws IOException {
071        return this;

074    /**
075     * Set the keys.
076     *
077     * @param keys the keys
078     * @return this value listener
079     */
080    @Override
081    public KeyValueStreamListener<K, V> keys(List<K> keys) throws IOException {
082        this.keys = keys;
083        this.shouldAutoGenID = shouldAutoGenID();
084        return this;
085    }

087    /**
088     * Receive values.
089     *
090     * @param values the values
091     * @return this value listener
092     * @throws java.io.IOException when value processing giv an error
093     */
094    @Override
095    public KeyValueStreamListener<K, V> values(List<V> values) throws IOException {
096        boolean hasSource = false;
097        if (current == null) {
098            current = newObject();
099        }
100        if (prev == null) {
101            prev = newObject();
102        }
103        // execute meta operations
104        for (int i = 0; i < values.size(); i++) {
105            // v may be null, then continue
106            V v = values.get(i);
107            if (v == null) {
108                continue;
109            }
110            K k = keys.get(i);
111            map(k, v, current);
112            if (ControlKeys._source.name().equals(k)) {
113                hasSource = true;
114            }
115        }
116        if (hasSource) {
117            end(current);
118            current = newObject();
119            return this;
120        }
121        // switch to next structured object if current is not equal to previous
122        if (!current.equals(prev) || current.isEmpty() || shouldAutoGenID) {
123            prev.source(current.source()); // "steal" source
124            end(prev); // here, the element is being prepared for bulk indexing
125            prev = current;
126            current = newObject();
127        }
128        // create current object from values by sequentially merging the values
129        for (int i = 0; i < keys.size(); i++) {
130            Map map = null;
131            try {
132                // JSON content?
133                map = JsonXContent.jsonXContent.createParser(values.get(i).toString()).mapAndClose();
134            } catch (Exception e) {
135                // ignore
136            }
137            Object v = map != null && map.size() > 0 ? map : values.get(i);
138            Map<String, Object> m = merge(current.source(), keys.get(i), v);
139            current.source(m);
140        }
141        return this;
142    }
143
144    protected void map(K k, V v, IndexableObject current) throws IOException {
145        if (ControlKeys._optype.name().equals(k)) {
146            current.optype(v.toString());
147        } else if (ControlKeys._index.name().equals(k)) {
148            current.index(v.toString());
149        } else if (ControlKeys._type.name().equals(k)) {
150            current.type(v.toString());
151        } else if (ControlKeys._id.name().equals(k)) {
152            current.id(v.toString());
153        } else if (ControlKeys._version.name().equals(k)) {
154            current.meta(k.toString(), v.toString());
155        } else if (ControlKeys._routing.name().equals(k)) {
156            current.meta(k.toString(), v.toString());
157        } else if (ControlKeys._parent.name().equals(k)) {
158            current.meta(k.toString(), v.toString());
159        } else if (ControlKeys._timestamp.name().equals(k)) {
160            current.meta(k.toString(), v.toString());
161        } else if (ControlKeys._ttl.name().equals(k)) {
162            current.meta(k.toString(), v.toString());
163        } else if (ControlKeys._job.name().equals(k)) {
164            current.meta(k.toString(), v.toString());
165        } else if (ControlKeys._source.name().equals(k)) {
166            current.source(JsonXContent.jsonXContent.createParser(v.toString()).mapAndClose());
167        }
168    }
169
170    /**
171     * End of values.
172     *
173     * @return this value listener
174     * @throws java.io.IOException
175     */
176    public KeyValueStreamListener<K, V> end() throws IOException {
177        if (prev != null) {
178            prev.source(current.source());
179            end(prev);
180        }
181        prev = newObject();
182        current = newObject();
183        return this;
184    }

186    /**
187     * The object is complete
188     *
189     * @param object the object
190     * @return this value listener
191     * @throws java.io.IOException when ending the object gives an error
192     */
193    public KeyValueStreamListener<K, V> end(IndexableObject object) throws IOException {
194        return this;
195    }

197    /**
198     * Merge key/value pair to a map holding a JSON object. The key consists of
199     * a path pointing to the value position in the JSON object. The key,
200     * representing a path, is divided into head/tail. The recursion terminates
201     * if there is only a head and no tail. In this case, the value is added as
202     * a tuple to the map. If the head key exists, the merge process is
203     * continued by following the path represented by the key. If the path does
204     * not exist, a new map is created. A conflict arises if there is no map at
205     * a head key position. Then, the prefix given in the path is considered
206     * illegal.
207     */
208    private final static Set<String> controlKeys = ControlKeys.makeSet();
209
210    @SuppressWarnings({"unchecked"})
211    protected Map<String, Object> merge(Map<String, Object> map, Object k, Object value) {
212        String key = k.toString();
213        if (controlKeys.contains(key)) {
214            return map;
215        }
216        int i = key.indexOf(delimiter);
217        String index = null;
218        boolean isSequence = false;
219        Matcher matcher = p.matcher(key);
220        if (matcher.matches()) {
221            isSequence = key.indexOf("[") < i;
222        }
223        if (i <= 0 || isSequence) {
224            isSequence = matcher.matches();
225            String head = key;
226            if (isSequence) {
227                head = matcher.group(1);
228                index = matcher.group(2);
229            }
230            if (index == null || index.isEmpty()) {
231                map.put(head, new Values(map.get(head), value, isSequence));
232            } else {
233                if (!map.containsKey(head)) {
234                    map.put(head, new LinkedList());
235                }
236                Object o = map.get(head);
237                if (o instanceof List) {
238                    List l = (List) o;
239                    int j = l.isEmpty() ? -1 : l.size() - 1;
240                    if (j >= 0) {
241                        Map<String, Object> m = (Map<String, Object>) l.get(j);
242                        int delimiterOffset = index.indexOf(delimiter);
243                        if (delimiterOffset >= 0) {
244                            if (!containsKeyInDotNotation(index, m)) {
245                                l.set(j, merge(m, index, value)); // append
246                            } else {
247                                l.add(merge(new LinkedHashMap(), index, value));
248                            }
249                        } else {
250                            if (!m.containsKey(index)) {
251                                l.set(j, merge(m, index, value)); // append
252                            } else {
253                                l.add(merge(new LinkedHashMap(), index, value));
254                            }
255                        }
256                    } else {
257                        l.add(merge(new LinkedHashMap(), index, value));
258                    }
259                }
260            }
261        } else {
262            String head = key.substring(0, i);
263            matcher = p.matcher(head);
264            isSequence = matcher.matches();
265            if (isSequence) {
266                head = matcher.group(1);
267            }
268            String tail = key.substring(i + 1);
269            if (map.containsKey(head)) {
270                Object o = map.get(head);
271                if (o instanceof Map) {
272                    merge((Map<String, Object>) o, tail, value);
273                } else if (o instanceof Values) {
274                    // head of Values a Map?
275                    o = ((Values) o).getValues()[0];
276                    if (o instanceof Map) {
277                        merge((Map<String, Object>) o, tail, value);
278                    } else {
279                        throw new IllegalArgumentException("illegal head: " + head);
280                    }
281                } else {
282                    throw new IllegalArgumentException("illegal head: " + head);
283                }
284            } else {
285                Map<String, Object> m = new LinkedHashMap<String, Object>();
286                map.put(head, m);
287                merge(m, tail, value);
288            }
289        }
290        return map;
291    }
292
293    private final static Pattern p = Pattern.compile("^(.*)\\[(.*?)\\]$");
294
295    /**
296     * Create a new structured object
297     *
298     * @return a new structured object
299     */
300    private IndexableObject newObject() {
301        return new PlainIndexableObject().ignoreNull(shouldIgnoreNull);
302    }
303
304    private boolean shouldAutoGenID() {
305        boolean b = true;
306        for (K key : keys) {
307            if (ControlKeys._id.name().equals(key)) {
308                b = false;
309                break;
310            }
311        }
312        return b;
313    }
314
315    private boolean containsKeyInDotNotation(String key, Map mapToCheck) {
316        int delimiterIndex = key.indexOf(delimiter);
317        if (delimiterIndex < 0) {
318            return mapToCheck.containsKey(key);
319        } else {
320            String head = key.substring(0, delimiterIndex);
321            String tail = key.substring(delimiterIndex + 1);
322            Object nextObject = mapToCheck.get(head);
323            if (nextObject != null) {
324                if (nextObject instanceof Map) {
325                    Map nextMap = (Map) nextObject;
326                    return containsKeyInDotNotation(tail, nextMap);
327                }
328                return false;
329            } else {
330                return false;
331            }
332        }
333    }
