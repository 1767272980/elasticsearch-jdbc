237    public RiverContext setRounding(String rounding) {
238        if ("ceiling".equalsIgnoreCase(rounding)) {
239            this.rounding = BigDecimal.ROUND_CEILING;
240        } else if ("down".equalsIgnoreCase(rounding)) {
241            this.rounding = BigDecimal.ROUND_DOWN;
242        } else if ("floor".equalsIgnoreCase(rounding)) {
243            this.rounding = BigDecimal.ROUND_FLOOR;
244        } else if ("halfdown".equalsIgnoreCase(rounding)) {
245            this.rounding = BigDecimal.ROUND_HALF_DOWN;
246        } else if ("halfeven".equalsIgnoreCase(rounding)) {
247            this.rounding = BigDecimal.ROUND_HALF_EVEN;
248        } else if ("halfup".equalsIgnoreCase(rounding)) {
249            this.rounding = BigDecimal.ROUND_HALF_UP;
250        } else if ("unnecessary".equalsIgnoreCase(rounding)) {
251            this.rounding = BigDecimal.ROUND_UNNECESSARY;
252        } else if ("up".equalsIgnoreCase(rounding)) {
253            this.rounding = BigDecimal.ROUND_UP;
254        }
255        return this;
256    }

352    public RiverContext contextualize() {
353        if (source != null) {
354            source.setRiverContext(this);
355        }
356        if (mouth != null) {
357            mouth.setRiverContext(this);
358        }
359        if (flow != null) {
360            flow.setRiverContext(this);
361        }
362        return this;
363    }

365    public Map<String, Object> asMap() {
366        try {
367            XContentBuilder builder = jsonBuilder()
368                    .startObject()
369                    .field("rivername", name)
370                    .field("settings", settings)
371                    .field("locale", LocaleUtil.fromLocale(locale))
372                    .field("job", job)
373                    .field("sql", sql)
374                    .field("autocommit", autocommit)
375                    .field("fetchsize", fetchSize)
376                    .field("maxrows", maxRows)
377                    .field("retries", retries)
378                    .field("maxretrywait", maxretrywait)
379                    .field("resultsetconcurrency", resultSetConcurrency)
380                    .field("resultsettype", resultSetType)
381                    .field("rounding", rounding)
382                    .field("scale", scale)
383                    .field("shouldignorenull", shouldIgnoreNull)
384                    .field("columnCreatedAt", columnCreatedAt)
385                    .field("columnUpdatedAt", columnUpdatedAt)
386                    .field("columnDeletedAt", columnDeletedAt)
387                    .field("columnEscape", columnEscape)
388                    .endObject();
389            return XContentHelper.convertToMap(builder.bytes(), true).v2();
390        } catch (IOException e) {
391            // should really not happen
392            return new HashMap<String, Object>();
393        }
394    }
395
396    @Override
