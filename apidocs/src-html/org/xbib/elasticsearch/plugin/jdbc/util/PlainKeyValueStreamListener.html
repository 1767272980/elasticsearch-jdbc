018import org.elasticsearch.common.xcontent.json.JsonXContent;
019import org.xbib.elasticsearch.plugin.jdbc.keyvalue.KeyValueStreamListener;

021import java.io.IOException;

030public class PlainKeyValueStreamListener<K, V> implements KeyValueStreamListener<K, V> {
031
032    private final static Pattern p = Pattern.compile("^(.*)\\[(.*?)\\]$");
033
034    /**
035     * The current structured object
036     */
037    private IndexableObject current;
038    /**
039     * The object before the current object
040     */
041    private IndexableObject prev;
042
043    /**
044     * The keys of the values. They are examined for the Elasticsearch index
045     * attributes.
046     */
047    private List<K> keys;
048    /**
049     * The delimiter between the key names in the path, used for string split.
050     * Should not be modified.
051     */
052    private char delimiter = '.';
053
054    private boolean shouldAutoGenID;
055
056    private boolean shouldIgnoreNull = false;
057
058    /**
059     * Set custom delimiter
060     *
061     * @param delimiter the delimiter
062     * @return this listener

064    public PlainKeyValueStreamListener delimiter(char delimiter) {
065        this.delimiter = delimiter;
066        return this;
067    }

069    public PlainKeyValueStreamListener shouldIgnoreNull(boolean shouldIgnoreNull) {
070        this.shouldIgnoreNull = shouldIgnoreNull;

089    /**
090     * Set the keys.
091     *
092     * @param keys the keys
093     * @return this value listener
094     */
095    @Override
096    public KeyValueStreamListener<K, V> keys(List<K> keys) throws IOException {
097        this.keys = keys;
098        this.shouldAutoGenID = shouldAutoGenID();
099        return this;
100    }

102    /**
103     * Receive values.
104     *
105     * @param values the values
106     * @return this value listener
107     * @throws java.io.IOException when value processing giv an error
108     */
109    @Override
110    public KeyValueStreamListener<K, V> values(List<V> values) throws IOException {
111        boolean hasSource = false;
112        if (current == null) {
113            current = newObject();
114        }
115        if (prev == null) {
116            prev = newObject();
117        }
118        // execute meta operations
119        for (int i = 0; i < keys.size() && i < values.size(); i++) {
120            // v may be null, then continue
121            V v = values.get(i);
122            if (v == null) {
123                continue;
124            }
125            K k = keys.get(i);
126            map(k, v, current);
127            if (ControlKeys._source.name().equals(k)) {
128                hasSource = true;
129            }
130        }
131        if (hasSource) {
132            end(current);
133            current = newObject();
134            return this;
135        }
136        // switch to next structured object if current is not equal to previous
137        if (!current.equals(prev) || current.isEmpty() || shouldAutoGenID) {
138            prev.source(current.source()); // "steal" source
139            end(prev); // here, the element is being prepared for bulk indexing
140            prev = current;
141            current = newObject();
142        }
143        // create current object from values by sequentially merging the values
144        for (int i = 0; i < keys.size() && i < values.size(); i++) {
145            Map map = null;
146            try {
147                // JSON content?
148                map = JsonXContent.jsonXContent.createParser(values.get(i).toString()).mapAndClose();
149            } catch (Exception e) {
150                // ignore
151            }
152            Object v = map != null && map.size() > 0 ? map : values.get(i);
153            Map<String, Object> m = merge(current.source(), keys.get(i), v);
154            current.source(m);
155        }
156        return this;
157    }
158
159    protected void map(K k, V v, IndexableObject current) throws IOException {
160        if (ControlKeys._optype.name().equals(k)) {
161            current.optype(v.toString());
162        } else if (ControlKeys._index.name().equals(k)) {
163            current.index(v.toString());
164        } else if (ControlKeys._type.name().equals(k)) {
165            current.type(v.toString());
166        } else if (ControlKeys._id.name().equals(k)) {
167            current.id(v.toString());
168        } else if (ControlKeys._version.name().equals(k)) {
169            current.meta(k.toString(), v.toString());
170        } else if (ControlKeys._routing.name().equals(k)) {
171            current.meta(k.toString(), v.toString());
172        } else if (ControlKeys._parent.name().equals(k)) {
173            current.meta(k.toString(), v.toString());
174        } else if (ControlKeys._timestamp.name().equals(k)) {
175            current.meta(k.toString(), v.toString());
176        } else if (ControlKeys._ttl.name().equals(k)) {
177            current.meta(k.toString(), v.toString());
178        } else if (ControlKeys._job.name().equals(k)) {
179            current.meta(k.toString(), v.toString());
180        } else if (ControlKeys._source.name().equals(k)) {
181            current.source(JsonXContent.jsonXContent.createParser(v.toString()).mapAndClose());
182        }
183    }
184
185    /**
186     * End of values.
187     *
188     * @return this value listener
189     * @throws java.io.IOException
190     */
191    public KeyValueStreamListener<K, V> end() throws IOException {
192        if (prev != null) {
193            prev.source(current.source());
194            end(prev);
195        }
196        prev = newObject();
197        current = newObject();
198        return this;
199    }

201    /**
202     * The object is complete
203     *
204     * @param object the object
205     * @return this value listener
206     * @throws java.io.IOException when ending the object gives an error
207     */
208    public KeyValueStreamListener<K, V> end(IndexableObject object) throws IOException {
209        return this;
210    }

212    /**
213     * Merge key/value pair to a map holding a JSON object. The key consists of
214     * a path pointing to the value position in the JSON object. The key,
215     * representing a path, is divided into head/tail. The recursion terminates
216     * if there is only a head and no tail. In this case, the value is added as
217     * a tuple to the map. If the head key exists, the merge process is
218     * continued by following the path represented by the key. If the path does
219     * not exist, a new map is created. A conflict arises if there is no map at
220     * a head key position. Then, the prefix given in the path is considered
221     * illegal.
222     */
223    private final static Set<String> controlKeys = ControlKeys.makeSet();
224
225    @SuppressWarnings({"unchecked"})
226    protected Map<String, Object> merge(Map<String, Object> map, Object k, Object value) {
227        String key = k.toString();
228        if (controlKeys.contains(key)) {
229            return map;
230        }
231        int i = key.indexOf(delimiter);
232        String index = null;
233        boolean isSequence = false;
234        Matcher matcher = p.matcher(key);
235        if (matcher.matches()) {
236            isSequence = key.indexOf("[") < i;
237        }
238        if (i <= 0 || isSequence) {
239            isSequence = matcher.matches();
240            String head = key;
241            if (isSequence) {
242                head = matcher.group(1);
243                index = matcher.group(2);
244            }
245            if (index == null || index.isEmpty()) {
246                map.put(head, new Values(map.get(head), value, isSequence));
247            } else {
248                if (!map.containsKey(head)) {
249                    map.put(head, new LinkedList());
250                }
251                Object o = map.get(head);
252                if (o instanceof List) {
253                    List l = (List) o;
254                    int j = l.isEmpty() ? -1 : l.size() - 1;
255                    if (j >= 0) {
256                        Map<String, Object> m = (Map<String, Object>) l.get(j);
257                        int delimiterOffset = index.indexOf(delimiter);
258                        if (delimiterOffset >= 0) {
259                            if (!containsKeyInDotNotation(index, m)) {
260                                l.set(j, merge(m, index, value)); // append
261                            } else {
262                                l.add(merge(new LinkedHashMap(), index, value));
263                            }
264                        } else {
265                            if (!m.containsKey(index)) {
266                                l.set(j, merge(m, index, value)); // append
267                            } else {
268                                l.add(merge(new LinkedHashMap(), index, value));
269                            }
270                        }
271                    } else {
272                        l.add(merge(new LinkedHashMap(), index, value));
273                    }
274                }
275            }
276        } else {
277            String head = key.substring(0, i);
278            matcher = p.matcher(head);
279            isSequence = matcher.matches();
280            if (isSequence) {
281                head = matcher.group(1);
282            }
283            String tail = key.substring(i + 1);
284            if (map.containsKey(head)) {
285                Object o = map.get(head);
286                if (o instanceof Map) {
287                    merge((Map<String, Object>) o, tail, value);
288                } else if (o instanceof Values) {
289                    // head of Values a Map?
290                    o = ((Values) o).getValues()[0];
291                    if (o instanceof Map) {
292                        merge((Map<String, Object>) o, tail, value);
293                    } else {
294                        throw new IllegalArgumentException("illegal head: " + head);
295                    }
296                } else {
297                    throw new IllegalArgumentException("illegal head: " + head);
298                }
299            } else {
300                Map<String, Object> m = new LinkedHashMap<String, Object>();
301                map.put(head, m);
302                merge(m, tail, value);
303            }
304        }
305        return map;
306    }

308    /**
309     * Create a new structured object
310     *
311     * @return a new structured object
312     */
313    private IndexableObject newObject() {
314        return new PlainIndexableObject().ignoreNull(shouldIgnoreNull);
315    }
316
317    private boolean shouldAutoGenID() {
318        boolean b = true;
319        for (K key : keys) {
320            if (ControlKeys._id.name().equals(key)) {
321                b = false;
322                break;
323            }
324        }
325        return b;
326    }
327
328    private boolean containsKeyInDotNotation(String key, Map mapToCheck) {
329        int delimiterIndex = key.indexOf(delimiter);
330        if (delimiterIndex < 0) {
331            return mapToCheck.containsKey(key);
332        } else {
333            String head = key.substring(0, delimiterIndex);
334            String tail = key.substring(delimiterIndex + 1);
335            Object nextObject = mapToCheck.get(head);
336            if (nextObject != null) {
337                if (nextObject instanceof Map) {
338                    Map nextMap = (Map) nextObject;
339                    return containsKeyInDotNotation(tail, nextMap);
340                }
341                return false;
342            } else {
343                return false;
344            }
345        }
346    }
