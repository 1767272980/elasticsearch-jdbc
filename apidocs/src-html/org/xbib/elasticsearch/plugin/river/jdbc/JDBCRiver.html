001package org.xbib.elasticsearch.plugin.river.jdbc;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.util.concurrent.EsExecutors;
006import org.elasticsearch.river.AbstractRiverComponent;
007import org.elasticsearch.river.RiverName;
008import org.elasticsearch.river.RiverSettings;
009import org.xbib.common.settings.Settings;
010import org.xbib.common.xcontent.support.XContentMapValues;
011import org.xbib.elasticsearch.plugin.feeder.Feeder;
012import org.xbib.elasticsearch.plugin.river.RunnableRiver;
013import org.xbib.elasticsearch.plugin.river.StatefulRiver;
014import org.xbib.elasticsearch.plugin.support.RiverServiceLoader;
015import org.xbib.elasticsearch.plugin.support.RiverState;
016import org.xbib.elasticsearch.river.jdbc.RiverFlow;
017import org.xbib.logging.Logger;
018import org.xbib.logging.LoggerFactory;
019
020import java.io.IOException;
021import java.util.Map;
022
023import static org.xbib.common.settings.ImmutableSettings.settingsBuilder;
024
025/**
026 * JDBC river
027 */
028public class JDBCRiver extends AbstractRiverComponent implements RunnableRiver, StatefulRiver {
029
030    private final Logger logger = LoggerFactory.getLogger(JDBCRiver.class.getName());
031
032    private final Client client;
033
034    private final Feeder feeder;
035
036    private volatile Thread riverThread;
037
038    private volatile boolean closed;
039
040    @Inject
041    @SuppressWarnings({"unchecked"})
042    public JDBCRiver(RiverName riverName,
043                     RiverSettings riverSettings,
044                     Client client) {
045        super(riverName, riverSettings);
046        this.client = client;
047        Map<String, Object> spec = (Map<String, Object>) riverSettings.settings().get("jdbc");
048        Settings mySettings = settingsBuilder().loadFromMap(spec).build();
049        String strategy = XContentMapValues.nodeStringValue(spec.get("strategy"), "simple");
050        RiverFlow riverFlow = RiverServiceLoader.findRiverFlow(strategy);
051        logger.debug("found river flow class {} for strategy {}", riverFlow.getClass().getName(), strategy);
052        this.feeder = riverFlow.getFeeder();
053        feeder.setType("jdbc").setSpec(riverSettings.settings()).setSettings(mySettings);
054    }
055
056    @Override
057    public void start() {
058        feeder.setClient(client);
059        this.riverThread = EsExecutors.daemonThreadFactory(settings.globalSettings(),
060                "river(" + riverName().getType() + "/" + riverName().getName() + ")")
061                .newThread(feeder);
062        feeder.schedule(riverThread);
063    }
064
065    @Override
066    public void close() {
067        if (closed) {
068            return;
069        }
070        closed = true;
071        logger.info("closing river({}/{})", riverName.getType(), riverName.getName());
072        try {
073            feeder.close();
074        } catch (IOException e) {
075            logger.error(e.getMessage(), e);
076        }
077        if (riverThread != null) {
078            riverThread.interrupt();
079        }
080    }
081
082    @Override
083    public void run() {
084        feeder.run();
085    }
086
087    @Override
088    public RiverState getRiverState() {
089        return feeder.getRiverState();
090    }
091}
