002    package org.xbib.elasticsearch.plugin.river.jdbc;
003    
004    import org.elasticsearch.common.inject.Inject;
005    import org.elasticsearch.plugins.AbstractPlugin;
006    import org.elasticsearch.rest.RestModule;
007    import org.elasticsearch.river.RiversModule;
008    
009    import org.xbib.elasticsearch.rest.action.RestJDBCRiverCreateAction;
010    import org.xbib.elasticsearch.rest.action.RestJDBCRiverInduceAction;
011    import org.xbib.elasticsearch.rest.action.RestJDBCRiverStateAction;
012    import org.xbib.elasticsearch.river.jdbc.JDBCRiver;
013    import org.xbib.elasticsearch.river.jdbc.JDBCRiverModule;
014    
015    /**
016     * JDBC River plugin. This plugin is the enrty point for the
017     * Elasticsearch river service. The onModule() methods are used
018     * by reflection. The JDBC River Module - a simple one which just
019     * binds a JDBCRiver instance to a River interface - and the
020     * REST move is configured. Also, a concurrent bulk move
021     * is registered.
022     */
023    public class JDBCRiverPlugin extends AbstractPlugin {
024    
025        @Inject
026        public JDBCRiverPlugin() {
027        }
028    
029        @Override
030        public String name() {
031            return JDBCRiver.NAME + "-"
032                    + Build.getInstance().getVersion() + "-"
033                    + Build.getInstance().getShortHash();
034        }
035    
036        @Override
037        public String description() {
038            return "JDBC River";
039        }
040    
041        /**
042         * Register the river to Elasticsearch
043         *
044         * @param module the rivers module
045         */
046        public void onModule(RiversModule module) {
047            module.registerRiver(JDBCRiver.TYPE, JDBCRiverModule.class);
048        }
049    
050        /**
051         * Register the REST actions of this river
052         *
053         * @param module the REST module
054         */
055        public void onModule(RestModule module) {
056            module.addRestAction(RestJDBCRiverCreateAction.class);
057            module.addRestAction(RestJDBCRiverInduceAction.class);
058            module.addRestAction(RestJDBCRiverStateAction.class);
059        }
060    
061    }
