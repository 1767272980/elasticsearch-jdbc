045/**
046 * The JDBC River REST fire move. The river can be fired once to run
047 * when this move is called from REST.
048 * <p>
049 * Example:<br/>
050 * <code>
051 * curl -XPOST 'localhost:9200/_river/my_jdbc_river/_induce'
052 * </code>
053 * @author JÃ¶rg Prante <joergprante@gmail.com>
054 * @author pdegeus
055 */
056public class RestJDBCRiverInduceAction extends BaseRestHandler {
057
058    private final RiversService riversService;
059
060    @Inject
061    public RestJDBCRiverInduceAction(Settings settings, Client client,
062                                     RestController controller, Injector injector) {
063        super(settings, client);
064        this.riversService = injector.getInstance(RiversService.class);
065        controller.registerHandler(RestRequest.Method.POST, "/_river/{river}/_induce", this);
066    }
067
068    @Override
069    public void handleRequest(final RestRequest request, RestChannel channel) {
070
071        //Get and check river name parameter
072        String riverName = request.param("river");
073        if (riverName == null || riverName.isEmpty()) {
074            respond(false, request, channel, "Parameter 'river' is required", RestStatus.BAD_REQUEST);
075            return;
076        }
077
078        //Retrieve the registered rivers using reflection (UGLY HACK!!)
079        //TODO: Obtain the rivers using public API
080        ImmutableMap<RiverName, River> rivers;
081        try {
082            Field field = RiversService.class.getDeclaredField("rivers");
083            field.setAccessible(true);
084            rivers = (ImmutableMap<RiverName, River>) field.get(riversService);
085        } catch (NoSuchFieldException e) {
086            errorResponse(request, channel, e);
087            return;
088        } catch (IllegalAccessException e) {
089            errorResponse(request, channel, e);
090            return;
091        }
092
093        boolean found = false;
094        for (Map.Entry<RiverName, River> entry : rivers.entrySet()) {
095            RiverName name = entry.getKey();
096
097            if (name.getName().equals(riverName)) {
098                if (!name.getType().equals(JDBCRiver.TYPE)) {
099                    respond(
100                        false, request, channel,
101                        "River '" + riverName + "' is not a jdbc-river, but has type " + name.getType(),
102                        RestStatus.UNPROCESSABLE_ENTITY
103                    );
104                    return;
105                }
106
107                JDBCRiver jdbcRiver = (JDBCRiver) entry.getValue();
108                jdbcRiver.induce();
109                found = true;
110                break;
111            }
112        }
113
114        String error = found ? null : "River not found: " + riverName;
115        respond(found, request, channel, error, RestStatus.OK);
116    }
117
118    private void respond(boolean success, RestRequest request, RestChannel channel, String error, RestStatus status) {
119        try {
120            XContentBuilder builder = RestXContentBuilder.restContentBuilder(request);
121            builder.startObject();
122            builder.field("success", success);
123            if (error != null) {
124                builder.field("error", error);
125            }
126            builder.endObject();
127            channel.sendResponse(new XContentRestResponse(request, status, builder));
128        } catch (IOException e) {
129            errorResponse(request, channel, e);
130        }
131    }
132
133    private void errorResponse(RestRequest request, RestChannel channel, Throwable e) {
134        try {
135            channel.sendResponse(new XContentThrowableRestResponse(request, e));
136        } catch (IOException e1) {
137            logger.error("Failed to send failure response", e1);
138        }
139    }
