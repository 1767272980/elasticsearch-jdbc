001    package org.xbib.elasticsearch.rest.action;
002    
003    import org.elasticsearch.client.Client;
004    import org.elasticsearch.common.inject.Inject;
005    import org.elasticsearch.common.inject.Injector;
006    import org.elasticsearch.common.settings.Settings;
007    import org.elasticsearch.common.xcontent.ToXContent;
008    import org.elasticsearch.common.xcontent.XContentBuilder;
009    import org.elasticsearch.rest.RestChannel;
010    import org.elasticsearch.rest.RestController;
011    import org.elasticsearch.rest.RestRequest;
012    import org.elasticsearch.rest.RestStatus;
013    import org.elasticsearch.rest.XContentRestResponse;
014    import org.elasticsearch.rest.XContentThrowableRestResponse;
015    import org.elasticsearch.rest.action.support.RestXContentBuilder;
016    import org.elasticsearch.river.River;
017    import org.elasticsearch.river.RiverName;
018    
019    import org.xbib.elasticsearch.river.jdbc.JDBCRiver;
020    
021    import java.io.IOException;
022    import java.util.Map;
023    
024    public class RestJDBCRiverStateAction extends AbstractRestRiverAction {
025    
026        @Inject
027        public RestJDBCRiverStateAction(Settings settings, Client client,
028                                        RestController controller,
029                                        Injector injector) {
030            super(settings, client, injector);
031            controller.registerHandler(RestRequest.Method.GET, "/_river/jdbc/{river}/state", this);
032        }
033    
034        @Override
035        public void handleRequest(RestRequest request, RestChannel channel) {
036            try {
037                String riverName = request.param("river");
038                boolean found = false;
039                for (Map.Entry<RiverName, River> entry : rivers(injector).entrySet()) {
040                    RiverName name = entry.getKey();
041                    if (name.getName().equals(riverName)) {
042                        if (!name.getType().equals(JDBCRiver.TYPE)) {
043                            respond(false, request, channel,
044                                    "River '" + riverName + "' is not a jdbc-river, but has type " + name.getType(),
045                                    RestStatus.UNPROCESSABLE_ENTITY
046                            );
047                            return;
048                        }
049                        JDBCRiver jdbcRiver = (JDBCRiver) entry.getValue();
050                        XContentBuilder builder = RestXContentBuilder.restContentBuilder(request);
051                        builder.startObject();
052                        builder.field("success", true);
053                        builder.field("state");
054                        jdbcRiver.riverFlow().riverState().toXContent(builder, ToXContent.EMPTY_PARAMS);
055                        builder.endObject();
056                        channel.sendResponse(new XContentRestResponse(request, RestStatus.OK, builder));
057                        found = true;
058                        break;
059                    }
060                }
061                String error = found ? null : "River not found: " + riverName;
062                respond(found, request, channel, error, RestStatus.OK);
063            } catch (IOException ioe) {
064                try {
065                    channel.sendResponse(new XContentThrowableRestResponse(request, ioe));
066                } catch (IOException e) {
067                    logger.error("unable to send response to client");
068                }
069            }
070        }
071    
072    }
