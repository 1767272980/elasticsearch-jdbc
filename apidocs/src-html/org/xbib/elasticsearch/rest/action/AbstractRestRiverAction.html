001    package org.xbib.elasticsearch.rest.action;
002    
003    import java.io.IOException;
004    import java.lang.reflect.Field;
005    
006    import org.elasticsearch.client.Client;
007    import org.elasticsearch.common.collect.ImmutableMap;
008    import org.elasticsearch.common.inject.Injector;
009    import org.elasticsearch.common.settings.Settings;
010    import org.elasticsearch.common.xcontent.XContentBuilder;
011    import org.elasticsearch.rest.BaseRestHandler;
012    import org.elasticsearch.rest.RestChannel;
013    import org.elasticsearch.rest.RestRequest;
014    import org.elasticsearch.rest.RestStatus;
015    import org.elasticsearch.rest.XContentRestResponse;
016    import org.elasticsearch.rest.XContentThrowableRestResponse;
017    import org.elasticsearch.rest.action.support.RestXContentBuilder;
018    import org.elasticsearch.river.River;
019    import org.elasticsearch.river.RiverName;
020    import org.elasticsearch.river.RiversService;
021    
022    public abstract class AbstractRestRiverAction extends BaseRestHandler {
023    
024        protected Injector injector;
025    
026        public AbstractRestRiverAction(Settings settings, Client client, Injector injector) {
027            super(settings, client);
028            this.injector = injector;
029        }
030    
031        protected void respond(boolean success, RestRequest request, RestChannel channel, String error, RestStatus status) {
032            try {
033                XContentBuilder builder = RestXContentBuilder.restContentBuilder(request);
034                builder.startObject();
035                builder.field("success", success);
036                if (error != null) {
037                    builder.field("error", error);
038                }
039                builder.endObject();
040                channel.sendResponse(new XContentRestResponse(request, status, builder));
041            } catch (IOException e) {
042                errorResponse(request, channel, e);
043            }
044        }
045    
046        protected void errorResponse(RestRequest request, RestChannel channel, Throwable e) {
047            try {
048                channel.sendResponse(new XContentThrowableRestResponse(request, e));
049            } catch (IOException e1) {
050                logger.error("Failed to send failure response", e1);
051            }
052        }
053    
054        /**
055         * Retrieve the registered rivers using reflection (UGLY HACK!!)
056         * TODO: Obtain the rivers using public API
057         * @param injector injector
058         * @return map of rivers or null if not possible
059         */
060        protected ImmutableMap<RiverName, River> rivers(Injector injector) {
061            RiversService riversService = injector.getInstance(RiversService.class);
062            try {
063                Field field = RiversService.class.getDeclaredField("rivers");
064                field.setAccessible(true);
065                return (ImmutableMap<RiverName, River>) field.get(riversService);
066            } catch (NoSuchFieldException e) {
067                return null;
068            } catch (IllegalAccessException e) {
069                return null;
070            }
071        }
072    }
