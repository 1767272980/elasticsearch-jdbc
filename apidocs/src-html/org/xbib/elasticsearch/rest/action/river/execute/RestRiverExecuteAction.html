001package org.xbib.elasticsearch.rest.action.river.execute;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.settings.Settings;
006import org.elasticsearch.common.xcontent.XContentBuilder;
007import org.elasticsearch.rest.BaseRestHandler;
008import org.elasticsearch.rest.BytesRestResponse;
009import org.elasticsearch.rest.RestChannel;
010import org.elasticsearch.rest.RestController;
011import org.elasticsearch.rest.RestRequest;
012import org.elasticsearch.rest.RestResponse;
013import org.elasticsearch.rest.action.support.RestBuilderListener;
014import org.xbib.elasticsearch.action.river.jdbc.execute.RiverExecuteAction;
015import org.xbib.elasticsearch.action.river.jdbc.execute.RiverExecuteRequest;
016import org.xbib.elasticsearch.action.river.jdbc.execute.RiverExecuteResponse;
017
018import static org.elasticsearch.rest.RestStatus.OK;
019
020/**
021 * Run a river. The river can be executed once with such a call. Example:
022 * <p/>
023 * curl -XPOST 'localhost:9200/_river/my_jdbc_river/_execute'
024 */
025public class RestRiverExecuteAction extends BaseRestHandler {
026
027    @Inject
028    public RestRiverExecuteAction(Settings settings, Client client, RestController controller) {
029        super(settings, client);
030        controller.registerHandler(RestRequest.Method.POST, "/_river/jdbc/{riverName}/_execute", this);
031    }
032
033    @Override
034    public void handleRequest(final RestRequest request, final RestChannel channel, final Client client) {
035        String riverName = request.param("riverName");
036        String riverType = "jdbc";
037        RiverExecuteRequest riverExecuteRequest = new RiverExecuteRequest()
038                .setRiverName(riverName)
039                .setRiverType(riverType);
040        client.admin().cluster().execute(RiverExecuteAction.INSTANCE, riverExecuteRequest, new RestBuilderListener<RiverExecuteResponse>(channel) {
041            @Override
042            public RestResponse buildResponse(RiverExecuteResponse riverExecuteResponse, XContentBuilder builder) throws Exception {
043                boolean isExecuted = false;
044                for (int i = 0; i < riverExecuteResponse.isExecuted().length; i++) {
045                    isExecuted = isExecuted || riverExecuteResponse.isExecuted()[i];
046                }
047                builder.field("executed", isExecuted);
048                return new BytesRestResponse(OK, builder);
049            }
050        });
051    }
052
053}
