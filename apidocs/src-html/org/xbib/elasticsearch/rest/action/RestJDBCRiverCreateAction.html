001    package org.xbib.elasticsearch.rest.action;
002    
003    import java.io.IOException;
004    
005    import org.elasticsearch.client.Client;
006    import org.elasticsearch.common.inject.Inject;
007    import org.elasticsearch.common.inject.Injector;
008    import org.elasticsearch.common.settings.ImmutableSettings;
009    import org.elasticsearch.common.settings.Settings;
010    import org.elasticsearch.common.xcontent.XContentBuilder;
011    import org.elasticsearch.common.xcontent.XContentHelper;
012    import org.elasticsearch.rest.RestChannel;
013    import org.elasticsearch.rest.RestController;
014    import org.elasticsearch.rest.RestRequest;
015    import org.elasticsearch.rest.XContentRestResponse;
016    import org.elasticsearch.rest.XContentThrowableRestResponse;
017    import org.elasticsearch.river.RiverName;
018    import org.elasticsearch.river.RiverSettings;
019    
020    import org.xbib.elasticsearch.river.jdbc.JDBCRiver;
021    
022    import static org.elasticsearch.rest.RestStatus.OK;
023    import static org.elasticsearch.rest.action.support.RestXContentBuilder.restContentBuilder;
024    
025    public class RestJDBCRiverCreateAction extends AbstractRestRiverAction {
026    
027        private Settings settings = ImmutableSettings.EMPTY;
028    
029        @Inject
030        public RestJDBCRiverCreateAction(Settings settings, Client client,
031                                         RestController controller, Injector injector) {
032            super(settings, client, injector);
033            this.settings = settings;
034            controller.registerHandler(RestRequest.Method.POST, "/_river/jdbc/{river}/create", this);
035        }
036    
037        @Override
038        public void handleRequest(RestRequest request, RestChannel channel) {
039            try {
040                String riverName = request.param("river");
041                RiverSettings riverSettings = new RiverSettings(settings,
042                        XContentHelper.convertToMap(request.content(), true).v2())  ;
043                JDBCRiver river = new JDBCRiver(new RiverName("jdbc", riverName), riverSettings, client);
044                river.start();
045                XContentBuilder builder = restContentBuilder(request);
046                builder.startObject()
047                        .field("active", river.riverFlow().isActive())
048                        .endObject();
049                channel.sendResponse(new XContentRestResponse(request, OK, builder));
050            } catch (IOException ioe) {
051                try {
052                    channel.sendResponse(new XContentThrowableRestResponse(request, ioe));
053                } catch (IOException e) {
054                    logger.error("unable to send response to client");
055                }
056            }
057        }
058    
059    }
