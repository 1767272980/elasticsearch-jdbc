001package org.xbib.pipeline;
002
003import java.util.Iterator;
004import java.util.LinkedHashMap;
005import java.util.Map;
006
007/**
008 * Basic pipeline for pressing pipeline requests.
009 * This abstract class can be used for creating custom Pipeline classes.
010 *
011 * @param <R> the pipeline request type
012 * @param <E> the pipeline error type
013 */
014public abstract class AbstractPipeline<R extends PipelineRequest, E extends PipelineException>
015        implements Pipeline<Boolean, R>, PipelineRequestListener<Boolean, R>,
016        PipelineErrorListener<Boolean, R, E> {
017
018    /**
019     * A list of request listeners for processing requests
020     */
021    private Map<String, PipelineRequestListener<Boolean, R>> requestListeners =
022            new LinkedHashMap<String, PipelineRequestListener<Boolean, R>>();
023
024    /**
025     * A list of error listeners for processing errors
026     */
027    private Map<String, PipelineErrorListener<Boolean, R, E>> errorListeners =
028            new LinkedHashMap<String, PipelineErrorListener<Boolean, R, E>>();
029
030
031    private volatile boolean isRunning;
032
033    public boolean isRunning() {
034        return isRunning;
035    }
036
037    /**
038     * Add a pipeline request listener to the pipeline. The listener is called each time
039     * this pipeline processes a new request.
040     *
041     * @param name     the listener name
042     * @param listener the listener
043     * @return this pipeline
044     */
045    public Pipeline<Boolean, R> add(String name, PipelineRequestListener<Boolean, R> listener) {
046        if (name != null) {
047            this.requestListeners.put(name, listener);
048        }
049        return this;
050    }
051
052    /**
053     * Add a pipeline request listener to the pipeline. The listener is called each time
054     * this pipeline processes a new request.
055     *
056     * @param name     the listener name
057     * @param listener the listener
058     * @return this pipeline
059     */
060    public Pipeline<Boolean, R> add(String name, PipelineErrorListener<Boolean, R, E> listener) {
061        if (name != null) {
062            this.errorListeners.put(name, listener);
063        }
064        return this;
065    }
066
067    /**
068     * Call this thread. Iterate over all request and pass them to request listeners.
069     * At least, this pipeline itself can listen to requests and handle errors.
070     * Only PipelineExceptions are handled for each listener. Other execptions will quit the
071     * pipeline request executions.
072     *
073     * @return a metric about the pipeline request executions.
074     * @throws Exception if pipeline execution was sborted by a non-PipelineException
075     */
076    @Override
077    public Boolean call() throws Exception {
078        Iterator<R> it = this;
079        isRunning = true;
080        while (it.hasNext()) {
081            R r = it.next();
082            // add ourselves if not already done
083            requestListeners.put(null, this);
084            errorListeners.put(null, this);
085            for (PipelineRequestListener<Boolean, R> requestListener : requestListeners.values()) {
086                try {
087                    requestListener.newRequest(this, r);
088                } catch (PipelineException e) {
089                    for (PipelineErrorListener<Boolean, R, E> errorListener : errorListeners.values()) {
090                        errorListener.error(this, r, (E) e);
091                    }
092                }
093            }
094        }
095        close();
096        isRunning = false;
097        return true;
098    }
099
100    /**
101     * Removing pipeline requests is not supported.
102     */
103    @Override
104    public void remove() {
105        throw new UnsupportedOperationException("remove not supported");
106    }
107
108    /**
109     * A new request for the pipeline is processed.
110     *
111     * @param pipeline the pipeline
112     * @param request  the pipeline request
113     */
114    public abstract void newRequest(Pipeline<Boolean, R> pipeline, R request);
115
116    /**
117     * A PipelineException occured.
118     *
119     * @param pipeline the pipeline
120     * @param request  the pipeline request
121     * @param error    the pipeline error
122     */
123    public abstract void error(Pipeline<Boolean, R> pipeline, R request, E error);
124
125}
