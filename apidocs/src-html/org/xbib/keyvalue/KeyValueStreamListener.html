032package org.xbib.keyvalue;
033
034import java.io.IOException;
035import java.util.List;
036
037public interface KeyValueStreamListener<K,V> {
038
039    /**
040     * Begin a key/value sequence
041     *
042     * @return this value listener
043     * @throws java.io.IOException
044     */
045    KeyValueStreamListener<K,V> begin() throws IOException;
046
047    /**
048     * Receive key/value pair
049     * @param key the key
050     * @param value the value
051     */
052    KeyValueStreamListener<K,V> keyValue(K key, V value) throws IOException;
053
054    /**
055     * Declare the keys for the values
056     *
057     * @param keys the keys
058     * @return this ValueListener
059     */
060    KeyValueStreamListener<K,V> keys(List<K> keys) throws IOException;
061
062    /**
063     * Receive values for the declared keys
064     *
065     * @param values the values
066     * @return this ValueListener
067     * @throws java.io.IOException
068     */
069    KeyValueStreamListener<K,V> values(List<V> values) throws IOException;
070
071    /**
072     * End a key/value sequence
073     *
074     * @return this listener
075     * @throws java.io.IOException
076     */
077    KeyValueStreamListener<K,V> end() throws IOException;
078
079}
